<!-- @segment snp1.1:1.1 -->

Yo uppatitaṁ vineti kodhaṁ,

<!-- @segment snp1.1:1.2 -->

Visaṭaṁ sappavisaṁva osadhehi;

<!-- @segment snp1.1:1.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:1.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:2.1 -->

Yo rāgamudacchidā asesaṁ,

<!-- @segment snp1.1:2.2 -->

Bhisapupphaṁva saroruhaṁ vigayha;

<!-- @segment snp1.1:2.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:2.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:3.1 -->

Yo taṇhamudacchidā asesaṁ,

<!-- @segment snp1.1:3.2 -->

Saritaṁ sīghasaraṁ visosayitvā;

<!-- @segment snp1.1:3.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:3.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:4.1 -->

Yo mānamudabbadhī asesaṁ,

<!-- @segment snp1.1:4.2 -->

Naḷasetuṁva sudubbalaṁ mahogho;

<!-- @segment snp1.1:4.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:4.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:5.1 -->

Yo nājjhagamā bhavesu sāraṁ,

<!-- @segment snp1.1:5.2 -->

Vicinaṁ pupphamivā udumbaresu;

<!-- @segment snp1.1:5.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:5.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:6.1 -->

Yassantarato na santi kopā,

<!-- @segment snp1.1:6.2 -->

Itibhavābhavatañca vītivatto;

<!-- @segment snp1.1:6.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:6.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:7.1 -->

Yassa vitakkā vidhūpitā,

<!-- @segment snp1.1:7.2 -->

Ajjhattaṁ suvikappitā asesā;

<!-- @segment snp1.1:7.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:7.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:8.1 -->

Yo nāccasārī na paccasārī,

<!-- @segment snp1.1:8.2 -->

Sabbaṁ accagamā imaṁ papañcaṁ;

<!-- @segment snp1.1:8.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:8.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:9.1 -->

Yo nāccasārī na paccasārī,

<!-- @segment snp1.1:9.2 -->

Sabbaṁ vitathamidanti ñatva loke;

<!-- @segment snp1.1:9.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:9.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:10.1 -->

Yo nāccasārī na paccasārī,

<!-- @segment snp1.1:10.2 -->

Sabbaṁ vitathamidanti vītalobho;

<!-- @segment snp1.1:10.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:10.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:11.1 -->

Yo nāccasārī na paccasārī,

<!-- @segment snp1.1:11.2 -->

Sabbaṁ vitathamidanti vītarāgo;

<!-- @segment snp1.1:11.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:11.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:12.1 -->

Yo nāccasārī na paccasārī,

<!-- @segment snp1.1:12.2 -->

Sabbaṁ vitathamidanti vītadoso;

<!-- @segment snp1.1:12.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:12.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:13.1 -->

Yo nāccasārī na paccasārī,

<!-- @segment snp1.1:13.2 -->

Sabbaṁ vitathamidanti vītamoho;

<!-- @segment snp1.1:13.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:13.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:14.1 -->

Yassānusayā na santi keci,

<!-- @segment snp1.1:14.2 -->

Mūlā ca akusalā samūhatāse;

<!-- @segment snp1.1:14.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:14.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:15.1 -->

Yassa darathajā na santi keci,

<!-- @segment snp1.1:15.2 -->

Oraṁ āgamanāya paccayāse;

<!-- @segment snp1.1:15.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:15.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:16.1 -->

Yassa vanathajā na santi keci,

<!-- @segment snp1.1:16.2 -->

Vinibandhāya bhavāya hetukappā;

<!-- @segment snp1.1:16.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:16.4 -->

Urago jiṇṇamivattacaṁ purāṇaṁ.

<!-- @segment snp1.1:17.1 -->

Yo nīvaraṇe pahāya pañca,

<!-- @segment snp1.1:17.2 -->

Anigho tiṇṇakathaṅkatho visallo;

<!-- @segment snp1.1:17.3 -->

So bhikkhu jahāti orapāraṁ,

<!-- @segment snp1.1:17.4 -->

Urago jiṇṇamivattacaṁ purāṇanti.

<!-- @segment snp1.1:17.5 -->

Uragasuttaṁ paṭhamaṁ.
