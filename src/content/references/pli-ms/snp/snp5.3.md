<!-- @segment snp5.3:1.1 -->

“Kodha santusito loke,

<!-- @segment snp5.3:1.2 -->

(iccāyasmā tissametteyyo)

<!-- @segment snp5.3:1.3 -->

Kassa no santi iñjitā;

<!-- @segment snp5.3:1.4 -->

Ko ubhantamabhiññāya,

<!-- @segment snp5.3:1.5 -->

Majjhe mantā na lippati;

<!-- @segment snp5.3:1.6 -->

Kaṁ brūsi mahāpurisoti,

<!-- @segment snp5.3:1.7 -->

Ko idha sibbinimaccagā”.

<!-- @segment snp5.3:2.1 -->

“Kāmesu brahmacariyavā,

<!-- @segment snp5.3:2.2 -->

(metteyyāti bhagavā)

<!-- @segment snp5.3:2.3 -->

Vītataṇho sadā sato;

<!-- @segment snp5.3:2.4 -->

Saṅkhāya nibbuto bhikkhu,

<!-- @segment snp5.3:2.5 -->

Tassa no santi iñjitā.

<!-- @segment snp5.3:3.1 -->

So ubhantamabhiññāya,

<!-- @segment snp5.3:3.2 -->

Majjhe mantā na lippati;

<!-- @segment snp5.3:3.3 -->

Taṁ brūmi mahāpurisoti,

<!-- @segment snp5.3:3.4 -->

So idha sibbinimaccagā”ti.

<!-- @segment snp5.3:3.5 -->

Tissametteyyamāṇavapucchā dutiyā.
