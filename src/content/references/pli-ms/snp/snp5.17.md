<!-- @segment snp5.18:1.1 -->

Idamavoca bhagavā magadhesu viharanto pāsāṇake cetiye,

<!-- @segment snp5.18:1.2 -->

paricārakasoḷasānaṁ brāhmaṇānaṁ ajjhiṭṭho puṭṭho puṭṭho pañhaṁ byākāsi.

<!-- @segment snp5.18:1.3 -->

Ekamekassa cepi pañhassa atthamaññāya dhammamaññāya dhammānudhammaṁ paṭipajjeyya, gaccheyyeva jarāmaraṇassa pāraṁ.

<!-- @segment snp5.18:1.4 -->

Pāraṅgamanīyā ime dhammāti, tasmā imassa dhammapariyāyassa pārāyananteva adhivacanaṁ.

<!-- @segment snp5.18:2.1 -->

Ajito tissametteyyo,

<!-- @segment snp5.18:2.2 -->

puṇṇako atha mettagū;

<!-- @segment snp5.18:2.3 -->

Dhotako upasīvo ca,

<!-- @segment snp5.18:2.4 -->

nando ca atha hemako.

<!-- @segment snp5.18:3.1 -->

Todeyya-kappā dubhayo,

<!-- @segment snp5.18:3.2 -->

jatukaṇṇī ca paṇḍito;

<!-- @segment snp5.18:3.3 -->

Bhadrāvudho udayo ca,

<!-- @segment snp5.18:3.4 -->

posālo cāpi brāhmaṇo;

<!-- @segment snp5.18:3.5 -->

Mogharājā ca medhāvī,

<!-- @segment snp5.18:3.6 -->

piṅgiyo ca mahāisi.

<!-- @segment snp5.18:4.1 -->

Ete buddhaṁ upāgacchuṁ,

<!-- @segment snp5.18:4.2 -->

sampannacaraṇaṁ isiṁ;

<!-- @segment snp5.18:4.3 -->

Pucchantā nipuṇe pañhe,

<!-- @segment snp5.18:4.4 -->

buddhaseṭṭhaṁ upāgamuṁ.

<!-- @segment snp5.18:5.1 -->

Tesaṁ buddho pabyākāsi,

<!-- @segment snp5.18:5.2 -->

pañhe puṭṭho yathātathaṁ;

<!-- @segment snp5.18:5.3 -->

Pañhānaṁ veyyākaraṇena,

<!-- @segment snp5.18:5.4 -->

tosesi brāhmaṇe muni.

<!-- @segment snp5.18:6.1 -->

Te tositā cakkhumatā,

<!-- @segment snp5.18:6.2 -->

buddhenādiccabandhunā;

<!-- @segment snp5.18:6.3 -->

Brahmacariyamacariṁsu,

<!-- @segment snp5.18:6.4 -->

varapaññassa santike.

<!-- @segment snp5.18:7.1 -->

Ekamekassa pañhassa,

<!-- @segment snp5.18:7.2 -->

yathā buddhena desitaṁ;

<!-- @segment snp5.18:7.3 -->

Tathā yo paṭipajjeyya,

<!-- @segment snp5.18:7.4 -->

gacche pāraṁ apārato.

<!-- @segment snp5.18:8.1 -->

Apārā pāraṁ gaccheyya,

<!-- @segment snp5.18:8.2 -->

bhāvento maggamuttamaṁ;

<!-- @segment snp5.18:8.3 -->

Maggo so pāraṁ gamanāya,

<!-- @segment snp5.18:8.4 -->

tasmā pārāyanaṁ iti.

<!-- @segment snp5.19:1.1 -->

“Pārāyanamanugāyissaṁ,

<!-- @segment snp5.19:1.2 -->

(iccāyasmā piṅgiyo)

<!-- @segment snp5.19:1.3 -->

Yathāddakkhi tathākkhāsi;

<!-- @segment snp5.19:1.4 -->

Vimalo bhūrimedhaso,

<!-- @segment snp5.19:1.5 -->

Nikkāmo nibbano nāgo;

<!-- @segment snp5.19:1.6 -->

Kissa hetu musā bhaṇe.

<!-- @segment snp5.19:2.1 -->

Pahīnamalamohassa,

<!-- @segment snp5.19:2.2 -->

mānamakkhappahāyino;

<!-- @segment snp5.19:2.3 -->

Handāhaṁ kittayissāmi,

<!-- @segment snp5.19:2.4 -->

giraṁ vaṇṇūpasañhitaṁ.

<!-- @segment snp5.19:3.1 -->

Tamonudo buddho samantacakkhu,

<!-- @segment snp5.19:3.2 -->

Lokantagū sabbabhavātivatto;

<!-- @segment snp5.19:3.3 -->

Anāsavo sabbadukkhappahīno,

<!-- @segment snp5.19:3.4 -->

Saccavhayo brahme upāsito me.

<!-- @segment snp5.19:4.1 -->

Dijo yathā kubbanakaṁ pahāya,

<!-- @segment snp5.19:4.2 -->

Bahupphalaṁ kānanamāvaseyya;

<!-- @segment snp5.19:4.3 -->

Evampahaṁ appadasse pahāya,

<!-- @segment snp5.19:4.4 -->

Mahodadhiṁ haṁsoriva ajjhapatto.

<!-- @segment snp5.19:5.1 -->

Ye me pubbe viyākaṁsu,

<!-- @segment snp5.19:5.2 -->

Huraṁ gotamasāsanā;

<!-- @segment snp5.19:5.3 -->

Iccāsi iti bhavissati,

<!-- @segment snp5.19:5.4 -->

Sabbaṁ taṁ itihītihaṁ;

<!-- @segment snp5.19:5.5 -->

Sabbaṁ taṁ takkavaḍḍhanaṁ.

<!-- @segment snp5.19:6.1 -->

Eko tamanudāsino,

<!-- @segment snp5.19:6.2 -->

jutimā so pabhaṅkaro;

<!-- @segment snp5.19:6.3 -->

Gotamo bhūripaññāṇo,

<!-- @segment snp5.19:6.4 -->

gotamo bhūrimedhaso.

<!-- @segment snp5.19:7.1 -->

Yo me dhammamadesesi,

<!-- @segment snp5.19:7.2 -->

Sandiṭṭhikamakālikaṁ;

<!-- @segment snp5.19:7.3 -->

Taṇhakkhayamanītikaṁ,

<!-- @segment snp5.19:7.4 -->

Yassa natthi upamā kvaci”.

<!-- @segment snp5.19:8.1 -->

“Kiṁ nu tamhā vippavasasi,

<!-- @segment snp5.19:8.2 -->

muhuttamapi piṅgiya;

<!-- @segment snp5.19:8.3 -->

Gotamā bhūripaññāṇā,

<!-- @segment snp5.19:8.4 -->

gotamā bhūrimedhasā.

<!-- @segment snp5.19:9.1 -->

Yo te dhammamadesesi,

<!-- @segment snp5.19:9.2 -->

Sandiṭṭhikamakālikaṁ;

<!-- @segment snp5.19:9.3 -->

Taṇhakkhayamanītikaṁ,

<!-- @segment snp5.19:9.4 -->

Yassa natthi upamā kvaci”.

<!-- @segment snp5.19:10.1 -->

“Nāhaṁ tamhā vippavasāmi,

<!-- @segment snp5.19:10.2 -->

Muhuttamapi brāhmaṇa;

<!-- @segment snp5.19:10.3 -->

Gotamā bhūripaññāṇā,

<!-- @segment snp5.19:10.4 -->

Gotamā bhūrimedhasā.

<!-- @segment snp5.19:11.1 -->

Yo me dhammamadesesi,

<!-- @segment snp5.19:11.2 -->

Sandiṭṭhikamakālikaṁ;

<!-- @segment snp5.19:11.3 -->

Taṇhakkhayamanītikaṁ,

<!-- @segment snp5.19:11.4 -->

Yassa natthi upamā kvaci.

<!-- @segment snp5.19:12.1 -->

Passāmi naṁ manasā cakkhunāva,

<!-- @segment snp5.19:12.2 -->

Rattindivaṁ brāhmaṇa appamatto;

<!-- @segment snp5.19:12.3 -->

Namassamāno vivasemi rattiṁ,

<!-- @segment snp5.19:12.4 -->

Teneva maññāmi avippavāsaṁ.

<!-- @segment snp5.19:13.1 -->

Saddhā ca pīti ca mano sati ca,

<!-- @segment snp5.19:13.2 -->

Nāpentime gotamasāsanamhā;

<!-- @segment snp5.19:13.3 -->

Yaṁ yaṁ disaṁ vajati bhūripañño,

<!-- @segment snp5.19:13.4 -->

Sa tena teneva natohamasmi.

<!-- @segment snp5.19:14.1 -->

Jiṇṇassa me dubbalathāmakassa,

<!-- @segment snp5.19:14.2 -->

Teneva kāyo na paleti tattha;

<!-- @segment snp5.19:14.3 -->

Saṅkappayantāya vajāmi niccaṁ,

<!-- @segment snp5.19:14.4 -->

Mano hi me brāhmaṇa tena yutto.

<!-- @segment snp5.19:15.1 -->

Paṅke sayāno pariphandamāno,

<!-- @segment snp5.19:15.2 -->

Dīpā dīpaṁ upaplaviṁ;

<!-- @segment snp5.19:15.3 -->

Athaddasāsiṁ sambuddhaṁ,

<!-- @segment snp5.19:15.4 -->

Oghatiṇṇamanāsavaṁ”.

<!-- @segment snp5.19:16.1 -->

“Yathā ahū vakkali muttasaddho,

<!-- @segment snp5.19:16.2 -->

Bhadrāvudho āḷavigotamo ca;

<!-- @segment snp5.19:16.3 -->

Evamevaṁ tvampi pamuñcassu saddhaṁ,

<!-- @segment snp5.19:16.4 -->

Gamissasi tvaṁ piṅgiya maccudheyyassa pāraṁ”.

<!-- @segment snp5.19:17.1 -->

“Esa bhiyyo pasīdāmi,

<!-- @segment snp5.19:17.2 -->

Sutvāna munino vaco;

<!-- @segment snp5.19:17.3 -->

Vivaṭṭacchado sambuddho,

<!-- @segment snp5.19:17.4 -->

Akhilo paṭibhānavā.

<!-- @segment snp5.19:18.1 -->

Adhideve abhiññāya,

<!-- @segment snp5.19:18.2 -->

Sabbaṁ vedi varovaraṁ;

<!-- @segment snp5.19:18.3 -->

Pañhānantakaro satthā,

<!-- @segment snp5.19:18.4 -->

Kaṅkhīnaṁ paṭijānataṁ.

<!-- @segment snp5.19:19.1 -->

Asaṁhīraṁ asaṅkuppaṁ,

<!-- @segment snp5.19:19.2 -->

Yassa natthi upamā kvaci;

<!-- @segment snp5.19:19.3 -->

Addhā gamissāmi na mettha kaṅkhā,

<!-- @segment snp5.19:19.4 -->

Evaṁ maṁ dhārehi adhimuttacittan”ti.
