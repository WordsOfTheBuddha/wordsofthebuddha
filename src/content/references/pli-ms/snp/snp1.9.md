<!-- @segment snp1.9:1.1 -->

“Ajja pannaraso uposatho,

<!-- @segment snp1.9:1.2 -->

(iti sātāgiro yakkho)

<!-- @segment snp1.9:1.3 -->

Dibbā ratti upaṭṭhitā;

<!-- @segment snp1.9:1.4 -->

Anomanāmaṁ satthāraṁ,

<!-- @segment snp1.9:1.5 -->

Handa passāma gotamaṁ”.

<!-- @segment snp1.9:2.1 -->

“Kacci mano supaṇihito,

<!-- @segment snp1.9:2.2 -->

(iti hemavato yakkho)

<!-- @segment snp1.9:2.3 -->

Sabbabhūtesu tādino;

<!-- @segment snp1.9:2.4 -->

Kacci iṭṭhe aniṭṭhe ca,

<!-- @segment snp1.9:2.5 -->

Saṅkappassa vasīkatā”.

<!-- @segment snp1.9:3.1 -->

“Mano cassa supaṇihito,

<!-- @segment snp1.9:3.2 -->

(iti sātāgiro yakkho)

<!-- @segment snp1.9:3.3 -->

Sabbabhūtesu tādino;

<!-- @segment snp1.9:3.4 -->

Atho iṭṭhe aniṭṭhe ca,

<!-- @segment snp1.9:3.5 -->

Saṅkappassa vasīkatā”.

<!-- @segment snp1.9:4.1 -->

“Kacci adinnaṁ nādiyati,

<!-- @segment snp1.9:4.2 -->

(iti hemavato yakkho)

<!-- @segment snp1.9:4.3 -->

Kacci pāṇesu saññato;

<!-- @segment snp1.9:4.4 -->

Kacci ārā pamādamhā,

<!-- @segment snp1.9:4.5 -->

Kacci jhānaṁ na riñcati”.

<!-- @segment snp1.9:5.1 -->

“Na so adinnaṁ ādiyati,

<!-- @segment snp1.9:5.2 -->

(iti sātāgiro yakkho)

<!-- @segment snp1.9:5.3 -->

Atho pāṇesu saññato;

<!-- @segment snp1.9:5.4 -->

Atho ārā pamādamhā,

<!-- @segment snp1.9:5.5 -->

Buddho jhānaṁ na riñcati”.

<!-- @segment snp1.9:6.1 -->

“Kacci musā na bhaṇati,

<!-- @segment snp1.9:6.2 -->

(iti hemavato yakkho)

<!-- @segment snp1.9:6.3 -->

Kacci na khīṇabyappatho;

<!-- @segment snp1.9:6.4 -->

Kacci vebhūtiyaṁ nāha,

<!-- @segment snp1.9:6.5 -->

Kacci samphaṁ na bhāsati”.

<!-- @segment snp1.9:7.1 -->

“Musā ca so na bhaṇati,

<!-- @segment snp1.9:7.2 -->

(iti sātāgiro yakkho)

<!-- @segment snp1.9:7.3 -->

Atho na khīṇabyappatho;

<!-- @segment snp1.9:7.4 -->

Atho vebhūtiyaṁ nāha,

<!-- @segment snp1.9:7.5 -->

Mantā atthañca bhāsati”.

<!-- @segment snp1.9:8.1 -->

“Kacci na rajjati kāmesu,

<!-- @segment snp1.9:8.2 -->

(iti hemavato yakkho)

<!-- @segment snp1.9:8.3 -->

Kacci cittaṁ anāvilaṁ;

<!-- @segment snp1.9:8.4 -->

Kacci mohaṁ atikkanto,

<!-- @segment snp1.9:8.5 -->

Kacci dhammesu cakkhumā”.

<!-- @segment snp1.9:9.1 -->

“Na so rajjati kāmesu,

<!-- @segment snp1.9:9.2 -->

(iti sātāgiro yakkho)

<!-- @segment snp1.9:9.3 -->

Atho cittaṁ anāvilaṁ;

<!-- @segment snp1.9:9.4 -->

Sabbamohaṁ atikkanto,

<!-- @segment snp1.9:9.5 -->

Buddho dhammesu cakkhumā”.

<!-- @segment snp1.9:10.1 -->

“Kacci vijjāya sampanno,

<!-- @segment snp1.9:10.2 -->

(iti hemavato yakkho)

<!-- @segment snp1.9:10.3 -->

Kacci saṁsuddhacāraṇo;

<!-- @segment snp1.9:10.4 -->

Kaccissa āsavā khīṇā,

<!-- @segment snp1.9:10.5 -->

Kacci natthi punabbhavo”.

<!-- @segment snp1.9:11.1 -->

“Vijjāya ceva sampanno,

<!-- @segment snp1.9:11.2 -->

(iti sātāgiro yakkho)

<!-- @segment snp1.9:11.3 -->

Atho saṁsuddhacāraṇo;

<!-- @segment snp1.9:11.4 -->

Sabbassa āsavā khīṇā,

<!-- @segment snp1.9:11.5 -->

Natthi tassa punabbhavo”.

<!-- @segment snp1.9:12.1 -->

“Sampannaṁ munino cittaṁ,

<!-- @segment snp1.9:12.2 -->

Kammunā byappathena ca;

<!-- @segment snp1.9:12.3 -->

Vijjācaraṇasampannaṁ,

<!-- @segment snp1.9:12.4 -->

Dhammato naṁ pasaṁsati”.

<!-- @segment snp1.9:13.1 -->

“Sampannaṁ munino cittaṁ,

<!-- @segment snp1.9:13.2 -->

Kammunā byappathena ca;

<!-- @segment snp1.9:13.3 -->

Vijjācaraṇasampannaṁ,

<!-- @segment snp1.9:13.4 -->

Dhammato anumodasi.

<!-- @segment snp1.9:14.1 -->

Sampannaṁ munino cittaṁ,

<!-- @segment snp1.9:14.2 -->

Kammunā byappathena ca;

<!-- @segment snp1.9:14.3 -->

Vijjācaraṇasampannaṁ,

<!-- @segment snp1.9:14.4 -->

Handa passāma gotamaṁ”.

<!-- @segment snp1.9:15.1 -->

“Eṇijaṅghaṁ kisaṁ vīraṁ,

<!-- @segment snp1.9:15.2 -->

Appāhāraṁ alolupaṁ;

<!-- @segment snp1.9:15.3 -->

Muniṁ vanasmiṁ jhāyantaṁ,

<!-- @segment snp1.9:15.4 -->

Ehi passāma gotamaṁ.

<!-- @segment snp1.9:16.1 -->

Sīhaṁvekacaraṁ nāgaṁ,

<!-- @segment snp1.9:16.2 -->

Kāmesu anapekkhinaṁ;

<!-- @segment snp1.9:16.3 -->

Upasaṅkamma pucchāma,

<!-- @segment snp1.9:16.4 -->

Maccupāsappamocanaṁ.

<!-- @segment snp1.9:17.1 -->

Akkhātāraṁ pavattāraṁ,

<!-- @segment snp1.9:17.2 -->

Sabbadhammāna pāraguṁ;

<!-- @segment snp1.9:17.3 -->

Buddhaṁ verabhayātītaṁ,

<!-- @segment snp1.9:17.4 -->

Mayaṁ pucchāma gotamaṁ”.

<!-- @segment snp1.9:18.1 -->

“Kismiṁ loko samuppanno,

<!-- @segment snp1.9:18.2 -->

(iti hemavato yakkho)

<!-- @segment snp1.9:18.3 -->

Kismiṁ kubbati santhavaṁ;

<!-- @segment snp1.9:18.4 -->

Kissa loko upādāya,

<!-- @segment snp1.9:18.5 -->

Kismiṁ loko vihaññati”.

<!-- @segment snp1.9:19.1 -->

“Chasu loko samuppanno,

<!-- @segment snp1.9:19.2 -->

(hemavatāti bhagavā)

<!-- @segment snp1.9:19.3 -->

Chasu kubbati santhavaṁ;

<!-- @segment snp1.9:19.4 -->

Channameva upādāya,

<!-- @segment snp1.9:19.5 -->

Chasu loko vihaññati”.

<!-- @segment snp1.9:20.1 -->

“Katamaṁ taṁ upādānaṁ,

<!-- @segment snp1.9:20.2 -->

Yattha loko vihaññati;

<!-- @segment snp1.9:20.3 -->

Niyyānaṁ pucchito brūhi,

<!-- @segment snp1.9:20.4 -->

Kathaṁ dukkhā pamuccati”.

<!-- @segment snp1.9:21.1 -->

“Pañca kāmaguṇā loke,

<!-- @segment snp1.9:21.2 -->

Manochaṭṭhā paveditā;

<!-- @segment snp1.9:21.3 -->

Ettha chandaṁ virājetvā,

<!-- @segment snp1.9:21.4 -->

Evaṁ dukkhā pamuccati.

<!-- @segment snp1.9:22.1 -->

Etaṁ lokassa niyyānaṁ,

<!-- @segment snp1.9:22.2 -->

Akkhātaṁ vo yathātathaṁ;

<!-- @segment snp1.9:22.3 -->

Etaṁ vo ahamakkhāmi,

<!-- @segment snp1.9:22.4 -->

Evaṁ dukkhā pamuccati”.

<!-- @segment snp1.9:23.1 -->

“Ko sūdha tarati oghaṁ,

<!-- @segment snp1.9:23.2 -->

Kodha tarati aṇṇavaṁ;

<!-- @segment snp1.9:23.3 -->

Appatiṭṭhe anālambe,

<!-- @segment snp1.9:23.4 -->

Ko gambhīre na sīdati”.

<!-- @segment snp1.9:24.1 -->

“Sabbadā sīlasampanno,

<!-- @segment snp1.9:24.2 -->

Paññavā susamāhito;

<!-- @segment snp1.9:24.3 -->

Ajjhattacintī satimā,

<!-- @segment snp1.9:24.4 -->

Oghaṁ tarati duttaraṁ.

<!-- @segment snp1.9:25.1 -->

Virato kāmasaññāya,

<!-- @segment snp1.9:25.2 -->

Sabbasaṁyojanātigo;

<!-- @segment snp1.9:25.3 -->

Nandībhavaparikkhīṇo,

<!-- @segment snp1.9:25.4 -->

So gambhīre na sīdati”.

<!-- @segment snp1.9:26.1 -->

“Gambhīrapaññaṁ nipuṇatthadassiṁ,

<!-- @segment snp1.9:26.2 -->

Akiñcanaṁ kāmabhave asattaṁ;

<!-- @segment snp1.9:26.3 -->

Taṁ passatha sabbadhi vippamuttaṁ,

<!-- @segment snp1.9:26.4 -->

Dibbe pathe kamamānaṁ mahesiṁ.

<!-- @segment snp1.9:27.1 -->

Anomanāmaṁ nipuṇatthadassiṁ,

<!-- @segment snp1.9:27.2 -->

Paññādadaṁ kāmālaye asattaṁ;

<!-- @segment snp1.9:27.3 -->

Taṁ passatha sabbaviduṁ sumedhaṁ,

<!-- @segment snp1.9:27.4 -->

Ariye pathe kamamānaṁ mahesiṁ.

<!-- @segment snp1.9:28.1 -->

Sudiṭṭhaṁ vata no ajja,

<!-- @segment snp1.9:28.2 -->

Suppabhātaṁ suhuṭṭhitaṁ;

<!-- @segment snp1.9:28.3 -->

Yaṁ addasāma sambuddhaṁ,

<!-- @segment snp1.9:28.4 -->

Oghatiṇṇamanāsavaṁ.

<!-- @segment snp1.9:29.1 -->

Ime dasasatā yakkhā,

<!-- @segment snp1.9:29.2 -->

Iddhimanto yasassino;

<!-- @segment snp1.9:29.3 -->

Sabbe taṁ saraṇaṁ yanti,

<!-- @segment snp1.9:29.4 -->

Tvaṁ no satthā anuttaro.

<!-- @segment snp1.9:30.1 -->

Te mayaṁ vicarissāma,

<!-- @segment snp1.9:30.2 -->

Gāmā gāmaṁ nagā nagaṁ;

<!-- @segment snp1.9:30.3 -->

Namassamānā sambuddhaṁ,

<!-- @segment snp1.9:30.4 -->

Dhammassa ca sudhammatan”ti.

<!-- @segment snp1.9:30.5 -->

Hemavatasuttaṁ navamaṁ.
