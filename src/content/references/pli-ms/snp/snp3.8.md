<!-- @segment snp3.8:1.1 -->

Animittamanaññātaṁ,

<!-- @segment snp3.8:1.2 -->

maccānaṁ idha jīvitaṁ;

<!-- @segment snp3.8:1.3 -->

Kasirañca parittañca,

<!-- @segment snp3.8:1.4 -->

tañca dukkhena saṁyutaṁ.

<!-- @segment snp3.8:2.1 -->

Na hi so upakkamo atthi,

<!-- @segment snp3.8:2.2 -->

yena jātā na miyyare;

<!-- @segment snp3.8:2.3 -->

Jarampi patvā maraṇaṁ,

<!-- @segment snp3.8:2.4 -->

evaṁdhammā hi pāṇino.

<!-- @segment snp3.8:3.1 -->

Phalānamiva pakkānaṁ,

<!-- @segment snp3.8:3.2 -->

pāto patanato bhayaṁ;

<!-- @segment snp3.8:3.3 -->

Evaṁ jātāna maccānaṁ,

<!-- @segment snp3.8:3.4 -->

niccaṁ maraṇato bhayaṁ.

<!-- @segment snp3.8:4.1 -->

Yathāpi kumbhakārassa,

<!-- @segment snp3.8:4.2 -->

katā mattikabhājanā;

<!-- @segment snp3.8:4.3 -->

Sabbe bhedanapariyantā,

<!-- @segment snp3.8:4.4 -->

evaṁ maccāna jīvitaṁ.

<!-- @segment snp3.8:5.1 -->

Daharā ca mahantā ca,

<!-- @segment snp3.8:5.2 -->

ye bālā ye ca paṇḍitā;

<!-- @segment snp3.8:5.3 -->

Sabbe maccuvasaṁ yanti,

<!-- @segment snp3.8:5.4 -->

sabbe maccuparāyaṇā.

<!-- @segment snp3.8:6.1 -->

Tesaṁ maccuparetānaṁ,

<!-- @segment snp3.8:6.2 -->

gacchataṁ paralokato;

<!-- @segment snp3.8:6.3 -->

Na pitā tāyate puttaṁ,

<!-- @segment snp3.8:6.4 -->

ñātī vā pana ñātake.

<!-- @segment snp3.8:7.1 -->

Pekkhataṁyeva ñātīnaṁ,

<!-- @segment snp3.8:7.2 -->

passa lālapataṁ puthu;

<!-- @segment snp3.8:7.3 -->

Ekamekova maccānaṁ,

<!-- @segment snp3.8:7.4 -->

govajjho viya nīyati.

<!-- @segment snp3.8:8.1 -->

Evamabbhāhato loko,

<!-- @segment snp3.8:8.2 -->

Maccunā ca jarāya ca;

<!-- @segment snp3.8:8.3 -->

Tasmā dhīrā na socanti,

<!-- @segment snp3.8:8.4 -->

Viditvā lokapariyāyaṁ.

<!-- @segment snp3.8:9.1 -->

Yassa maggaṁ na jānāsi,

<!-- @segment snp3.8:9.2 -->

āgatassa gatassa vā;

<!-- @segment snp3.8:9.3 -->

Ubho ante asampassaṁ,

<!-- @segment snp3.8:9.4 -->

niratthaṁ paridevasi.

<!-- @segment snp3.8:10.1 -->

Paridevayamāno ce,

<!-- @segment snp3.8:10.2 -->

Kiñcidatthaṁ udabbahe;

<!-- @segment snp3.8:10.3 -->

Sammūḷho hiṁsamattānaṁ,

<!-- @segment snp3.8:10.4 -->

Kayirā ce naṁ vicakkhaṇo.

<!-- @segment snp3.8:11.1 -->

Na hi ruṇṇena sokena,

<!-- @segment snp3.8:11.2 -->

Santiṁ pappoti cetaso;

<!-- @segment snp3.8:11.3 -->

Bhiyyassuppajjate dukkhaṁ,

<!-- @segment snp3.8:11.4 -->

Sarīraṁ cupahaññati.

<!-- @segment snp3.8:12.1 -->

Kiso vivaṇṇo bhavati,

<!-- @segment snp3.8:12.2 -->

Hiṁsamattānamattanā;

<!-- @segment snp3.8:12.3 -->

Na tena petā pālenti,

<!-- @segment snp3.8:12.4 -->

Niratthā paridevanā.

<!-- @segment snp3.8:13.1 -->

Sokamappajahaṁ jantu,

<!-- @segment snp3.8:13.2 -->

Bhiyyo dukkhaṁ nigacchati;

<!-- @segment snp3.8:13.3 -->

Anutthunanto kālaṅkataṁ,

<!-- @segment snp3.8:13.4 -->

Sokassa vasamanvagū.

<!-- @segment snp3.8:14.1 -->

Aññepi passa gamine,

<!-- @segment snp3.8:14.2 -->

Yathākammupage nare;

<!-- @segment snp3.8:14.3 -->

Maccuno vasamāgamma,

<!-- @segment snp3.8:14.4 -->

Phandantevidha pāṇino.

<!-- @segment snp3.8:15.1 -->

Yena yena hi maññanti,

<!-- @segment snp3.8:15.2 -->

Tato taṁ hoti aññathā;

<!-- @segment snp3.8:15.3 -->

Etādiso vinābhāvo,

<!-- @segment snp3.8:15.4 -->

Passa lokassa pariyāyaṁ.

<!-- @segment snp3.8:16.1 -->

Api vassasataṁ jīve,

<!-- @segment snp3.8:16.2 -->

Bhiyyo vā pana māṇavo;

<!-- @segment snp3.8:16.3 -->

Ñātisaṅghā vinā hoti,

<!-- @segment snp3.8:16.4 -->

Jahāti idha jīvitaṁ.

<!-- @segment snp3.8:17.1 -->

Tasmā arahato sutvā,

<!-- @segment snp3.8:17.2 -->

Vineyya paridevitaṁ;

<!-- @segment snp3.8:17.3 -->

Petaṁ kālaṅkataṁ disvā,

<!-- @segment snp3.8:17.4 -->

Neso labbhā mayā iti.

<!-- @segment snp3.8:18.1 -->

Yathā saraṇamādittaṁ,

<!-- @segment snp3.8:18.2 -->

Vārinā parinibbaye;

<!-- @segment snp3.8:18.3 -->

Evampi dhīro sapañño,

<!-- @segment snp3.8:18.4 -->

Paṇḍito kusalo naro;

<!-- @segment snp3.8:18.5 -->

Khippamuppatitaṁ sokaṁ,

<!-- @segment snp3.8:18.6 -->

Vāto tūlaṁva dhaṁsaye.

<!-- @segment snp3.8:19.1 -->

Paridevaṁ pajappañca,

<!-- @segment snp3.8:19.2 -->

Domanassañca attano;

<!-- @segment snp3.8:19.3 -->

Attano sukhamesāno,

<!-- @segment snp3.8:19.4 -->

Abbahe sallamattano.

<!-- @segment snp3.8:20.1 -->

Abbūḷhasallo asito,

<!-- @segment snp3.8:20.2 -->

santiṁ pappuyya cetaso;

<!-- @segment snp3.8:20.3 -->

Sabbasokaṁ atikkanto,

<!-- @segment snp3.8:20.4 -->

asoko hoti nibbutoti.

<!-- @segment snp3.8:20.5 -->

Sallasuttaṁ aṭṭhamaṁ.
