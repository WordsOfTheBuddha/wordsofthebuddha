<!-- @segment snp5.14:1.1 -->

“Jhāyiṁ virajamāsīnaṁ,

<!-- @segment snp5.14:1.2 -->

(iccāyasmā udayo)

<!-- @segment snp5.14:1.3 -->

Katakiccaṁ anāsavaṁ;

<!-- @segment snp5.14:1.4 -->

Pāraguṁ sabbadhammānaṁ,

<!-- @segment snp5.14:1.5 -->

Atthi pañhena āgamaṁ;

<!-- @segment snp5.14:1.6 -->

Aññāvimokkhaṁ pabrūhi,

<!-- @segment snp5.14:1.7 -->

Avijjāya pabhedanaṁ”.

<!-- @segment snp5.14:2.1 -->

“Pahānaṁ kāmacchandānaṁ,

<!-- @segment snp5.14:2.2 -->

(udayāti bhagavā)

<!-- @segment snp5.14:2.3 -->

Domanassāna cūbhayaṁ;

<!-- @segment snp5.14:2.4 -->

Thinassa ca panūdanaṁ,

<!-- @segment snp5.14:2.5 -->

Kukkuccānaṁ nivāraṇaṁ.

<!-- @segment snp5.14:3.1 -->

Upekkhāsatisaṁsuddhaṁ,

<!-- @segment snp5.14:3.2 -->

dhammatakkapurejavaṁ;

<!-- @segment snp5.14:3.3 -->

Aññāvimokkhaṁ pabrūmi,

<!-- @segment snp5.14:3.4 -->

avijjāya pabhedanaṁ”.

<!-- @segment snp5.14:4.1 -->

“Kiṁsu saṁyojano loko,

<!-- @segment snp5.14:4.2 -->

kiṁsu tassa vicāraṇaṁ;

<!-- @segment snp5.14:4.3 -->

Kissassa vippahānena,

<!-- @segment snp5.14:4.4 -->

nibbānaṁ iti vuccati”.

<!-- @segment snp5.14:5.1 -->

“Nandisaṁyojano loko,

<!-- @segment snp5.14:5.2 -->

vitakkassa vicāraṇaṁ;

<!-- @segment snp5.14:5.3 -->

Taṇhāya vippahānena,

<!-- @segment snp5.14:5.4 -->

nibbānaṁ iti vuccati”.

<!-- @segment snp5.14:6.1 -->

“Kathaṁ satassa carato,

<!-- @segment snp5.14:6.2 -->

viññāṇaṁ uparujjhati;

<!-- @segment snp5.14:6.3 -->

Bhagavantaṁ puṭṭhumāgamma,

<!-- @segment snp5.14:6.4 -->

taṁ suṇoma vaco tava”.

<!-- @segment snp5.14:7.1 -->

“Ajjhattañca bahiddhā ca,

<!-- @segment snp5.14:7.2 -->

Vedanaṁ nābhinandato;

<!-- @segment snp5.14:7.3 -->

Evaṁ satassa carato,

<!-- @segment snp5.14:7.4 -->

Viññāṇaṁ uparujjhatī”ti.

<!-- @segment snp5.14:7.5 -->

Udayamāṇavapucchā terasamā.
