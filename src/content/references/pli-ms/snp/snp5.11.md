<!-- @segment snp5.11:1.1 -->

“Majjhe sarasmiṁ tiṭṭhataṁ,

<!-- @segment snp5.11:1.2 -->

(iccāyasmā kappo)

<!-- @segment snp5.11:1.3 -->

Oghe jāte mahabbhaye;

<!-- @segment snp5.11:1.4 -->

Jarāmaccuparetānaṁ,

<!-- @segment snp5.11:1.5 -->

Dīpaṁ pabrūhi mārisa;

<!-- @segment snp5.11:1.6 -->

Tvañca me dīpamakkhāhi,

<!-- @segment snp5.11:1.7 -->

Yathāyidaṁ nāparaṁ siyā”.

<!-- @segment snp5.11:2.1 -->

“Majjhe sarasmiṁ tiṭṭhataṁ,

<!-- @segment snp5.11:2.2 -->

(kappāti bhagavā)

<!-- @segment snp5.11:2.3 -->

Oghe jāte mahabbhaye;

<!-- @segment snp5.11:2.4 -->

Jarāmaccuparetānaṁ,

<!-- @segment snp5.11:2.5 -->

Dīpaṁ pabrūmi kappa te.

<!-- @segment snp5.11:3.1 -->

Akiñcanaṁ anādānaṁ,

<!-- @segment snp5.11:3.2 -->

etaṁ dīpaṁ anāparaṁ;

<!-- @segment snp5.11:3.3 -->

Nibbānaṁ iti naṁ brūmi,

<!-- @segment snp5.11:3.4 -->

jarāmaccuparikkhayaṁ.

<!-- @segment snp5.11:4.1 -->

Etadaññāya ye satā,

<!-- @segment snp5.11:4.2 -->

Diṭṭhadhammābhinibbutā;

<!-- @segment snp5.11:4.3 -->

Na te māravasānugā,

<!-- @segment snp5.11:4.4 -->

Na te mārassa paddhagū”ti.

<!-- @segment snp5.11:4.5 -->

Kappamāṇavapucchā dasamā.
