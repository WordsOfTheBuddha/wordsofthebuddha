<!-- @segment snp2.3:1.1 -->

Hiriṁ tarantaṁ vijigucchamānaṁ,

<!-- @segment snp2.3:1.2 -->

Tavāhamasmi iti bhāsamānaṁ;

<!-- @segment snp2.3:1.3 -->

Sayhāni kammāni anādiyantaṁ,

<!-- @segment snp2.3:1.4 -->

Neso mamanti iti naṁ vijaññā.

<!-- @segment snp2.3:2.1 -->

Ananvayaṁ piyaṁ vācaṁ,

<!-- @segment snp2.3:2.2 -->

yo mittesu pakubbati;

<!-- @segment snp2.3:2.3 -->

Akarontaṁ bhāsamānaṁ,

<!-- @segment snp2.3:2.4 -->

parijānanti paṇḍitā.

<!-- @segment snp2.3:3.1 -->

Na so mitto yo sadā appamatto,

<!-- @segment snp2.3:3.2 -->

Bhedāsaṅkī randhamevānupassī;

<!-- @segment snp2.3:3.3 -->

Yasmiñca seti urasīva putto,

<!-- @segment snp2.3:3.4 -->

Sa ve mitto yo parehi abhejjo.

<!-- @segment snp2.3:4.1 -->

Pāmujjakaraṇaṁ ṭhānaṁ,

<!-- @segment snp2.3:4.2 -->

pasaṁsāvahanaṁ sukhaṁ;

<!-- @segment snp2.3:4.3 -->

Phalānisaṁso bhāveti,

<!-- @segment snp2.3:4.4 -->

vahanto porisaṁ dhuraṁ.

<!-- @segment snp2.3:5.1 -->

Pavivekarasaṁ pitvā,

<!-- @segment snp2.3:5.2 -->

Rasaṁ upasamassa ca;

<!-- @segment snp2.3:5.3 -->

Niddaro hoti nippāpo,

<!-- @segment snp2.3:5.4 -->

Dhammapītirasaṁ pivanti.

<!-- @segment snp2.3:5.5 -->

Hirisuttaṁ tatiyaṁ.
