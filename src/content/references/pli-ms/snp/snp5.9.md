<!-- @segment snp5.9:1.1 -->

“Ye me pubbe viyākaṁsu,

<!-- @segment snp5.9:1.2 -->

(iccāyasmā hemako)

<!-- @segment snp5.9:1.3 -->

Huraṁ gotamasāsanā;

<!-- @segment snp5.9:1.4 -->

Iccāsi iti bhavissati,

<!-- @segment snp5.9:1.5 -->

Sabbaṁ taṁ itihītihaṁ;

<!-- @segment snp5.9:1.6 -->

Sabbaṁ taṁ takkavaḍḍhanaṁ,

<!-- @segment snp5.9:1.7 -->

Nāhaṁ tattha abhiramiṁ.

<!-- @segment snp5.9:2.1 -->

Tvañca me dhammamakkhāhi,

<!-- @segment snp5.9:2.2 -->

taṇhānigghātanaṁ muni;

<!-- @segment snp5.9:2.3 -->

Yaṁ viditvā sato caraṁ,

<!-- @segment snp5.9:2.4 -->

tare loke visattikaṁ”.

<!-- @segment snp5.9:3.1 -->

“Idha diṭṭhasutamutaviññātesu,

<!-- @segment snp5.9:3.2 -->

Piyarūpesu hemaka;

<!-- @segment snp5.9:3.3 -->

Chandarāgavinodanaṁ,

<!-- @segment snp5.9:3.4 -->

Nibbānapadamaccutaṁ.

<!-- @segment snp5.9:4.1 -->

Etadaññāya ye satā,

<!-- @segment snp5.9:4.2 -->

Diṭṭhadhammābhinibbutā;

<!-- @segment snp5.9:4.3 -->

Upasantā ca te sadā,

<!-- @segment snp5.9:4.4 -->

Tiṇṇā loke visattikan”ti.

<!-- @segment snp5.9:4.5 -->

Hemakamāṇavapucchā aṭṭhamā.
