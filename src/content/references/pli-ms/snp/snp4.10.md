<!-- @segment snp4.10:1.1 -->

“Kathaṁdassī kathaṁsīlo,

<!-- @segment snp4.10:1.2 -->

upasantoti vuccati;

<!-- @segment snp4.10:1.3 -->

Taṁ me gotama pabrūhi,

<!-- @segment snp4.10:1.4 -->

pucchito uttamaṁ naraṁ”.

<!-- @segment snp4.10:2.1 -->

“Vītataṇho purā bhedā,

<!-- @segment snp4.10:2.2 -->

(iti bhagavā)

<!-- @segment snp4.10:2.3 -->

Pubbamantamanissito;

<!-- @segment snp4.10:2.4 -->

Vemajjhe nupasaṅkheyyo,

<!-- @segment snp4.10:2.5 -->

Tassa natthi purakkhataṁ.

<!-- @segment snp4.10:3.1 -->

Akkodhano asantāsī,

<!-- @segment snp4.10:3.2 -->

avikatthī akukkuco;

<!-- @segment snp4.10:3.3 -->

Mantabhāṇī anuddhato,

<!-- @segment snp4.10:3.4 -->

sa ve vācāyato muni.

<!-- @segment snp4.10:4.1 -->

Nirāsatti anāgate,

<!-- @segment snp4.10:4.2 -->

atītaṁ nānusocati;

<!-- @segment snp4.10:4.3 -->

Vivekadassī phassesu,

<!-- @segment snp4.10:4.4 -->

diṭṭhīsu ca na nīyati.

<!-- @segment snp4.10:5.1 -->

Patilīno akuhako,

<!-- @segment snp4.10:5.2 -->

apihālu amaccharī;

<!-- @segment snp4.10:5.3 -->

Appagabbho ajeguccho,

<!-- @segment snp4.10:5.4 -->

pesuṇeyye ca no yuto.

<!-- @segment snp4.10:6.1 -->

Sātiyesu anassāvī,

<!-- @segment snp4.10:6.2 -->

atimāne ca no yuto;

<!-- @segment snp4.10:6.3 -->

Saṇho ca paṭibhānavā,

<!-- @segment snp4.10:6.4 -->

na saddho na virajjati.

<!-- @segment snp4.10:7.1 -->

Lābhakamyā na sikkhati,

<!-- @segment snp4.10:7.2 -->

alābhe ca na kuppati;

<!-- @segment snp4.10:7.3 -->

Aviruddho ca taṇhāya,

<!-- @segment snp4.10:7.4 -->

rasesu nānugijjhati.

<!-- @segment snp4.10:8.1 -->

Upekkhako sadā sato,

<!-- @segment snp4.10:8.2 -->

na loke maññate samaṁ;

<!-- @segment snp4.10:8.3 -->

Na visesī na nīceyyo,

<!-- @segment snp4.10:8.4 -->

tassa no santi ussadā.

<!-- @segment snp4.10:9.1 -->

Yassa nissayanā natthi,

<!-- @segment snp4.10:9.2 -->

ñatvā dhammaṁ anissito;

<!-- @segment snp4.10:9.3 -->

Bhavāya vibhavāya vā,

<!-- @segment snp4.10:9.4 -->

taṇhā yassa na vijjati.

<!-- @segment snp4.10:10.1 -->

Taṁ brūmi upasantoti,

<!-- @segment snp4.10:10.2 -->

Kāmesu anapekkhinaṁ;

<!-- @segment snp4.10:10.3 -->

Ganthā tassa na vijjanti,

<!-- @segment snp4.10:10.4 -->

Atarī so visattikaṁ.

<!-- @segment snp4.10:11.1 -->

Na tassa puttā pasavo,

<!-- @segment snp4.10:11.2 -->

Khettaṁ vatthuñca vijjati;

<!-- @segment snp4.10:11.3 -->

Attā vāpi nirattā vā,

<!-- @segment snp4.10:11.4 -->

Na tasmiṁ upalabbhati.

<!-- @segment snp4.10:12.1 -->

Yena naṁ vajjuṁ puthujjanā,

<!-- @segment snp4.10:12.2 -->

Atho samaṇabrāhmaṇā;

<!-- @segment snp4.10:12.3 -->

Taṁ tassa apurakkhataṁ,

<!-- @segment snp4.10:12.4 -->

Tasmā vādesu nejati.

<!-- @segment snp4.10:13.1 -->

Vītagedho amaccharī,

<!-- @segment snp4.10:13.2 -->

Na ussesu vadate muni;

<!-- @segment snp4.10:13.3 -->

Na samesu na omesu,

<!-- @segment snp4.10:13.4 -->

Kappaṁ neti akappiyo.

<!-- @segment snp4.10:14.1 -->

Yassa loke sakaṁ natthi,

<!-- @segment snp4.10:14.2 -->

Asatā ca na socati;

<!-- @segment snp4.10:14.3 -->

Dhammesu ca na gacchati,

<!-- @segment snp4.10:14.4 -->

Sa ve santoti vuccatī”ti.

<!-- @segment snp4.10:14.5 -->

Purābhedasuttaṁ dasamaṁ.
