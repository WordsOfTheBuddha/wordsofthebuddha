<!-- @segment snp5.15:1.1 -->

“Yo atītaṁ ādisati,

<!-- @segment snp5.15:1.2 -->

(iccāyasmā posālo)

<!-- @segment snp5.15:1.3 -->

Anejo chinnasaṁsayo;

<!-- @segment snp5.15:1.4 -->

Pāraguṁ sabbadhammānaṁ,

<!-- @segment snp5.15:1.5 -->

Atthi pañhena āgamaṁ.

<!-- @segment snp5.15:2.1 -->

Vibhūtarūpasaññissa,

<!-- @segment snp5.15:2.2 -->

sabbakāyappahāyino;

<!-- @segment snp5.15:2.3 -->

Ajjhattañca bahiddhā ca,

<!-- @segment snp5.15:2.4 -->

natthi kiñcīti passato;

<!-- @segment snp5.15:2.5 -->

Ñāṇaṁ sakkānupucchāmi,

<!-- @segment snp5.15:2.6 -->

kathaṁ neyyo tathāvidho”.

<!-- @segment snp5.15:3.1 -->

“Viññāṇaṭṭhitiyo sabbā,

<!-- @segment snp5.15:3.2 -->

(posālāti bhagavā)

<!-- @segment snp5.15:3.3 -->

Abhijānaṁ tathāgato;

<!-- @segment snp5.15:3.4 -->

Tiṭṭhantamenaṁ jānāti,

<!-- @segment snp5.15:3.5 -->

Vimuttaṁ tapparāyaṇaṁ.

<!-- @segment snp5.15:4.1 -->

Ākiñcaññasambhavaṁ ñatvā,

<!-- @segment snp5.15:4.2 -->

Nandī saṁyojanaṁ iti;

<!-- @segment snp5.15:4.3 -->

Evametaṁ abhiññāya,

<!-- @segment snp5.15:4.4 -->

Tato tattha vipassati;

<!-- @segment snp5.15:4.5 -->

Etaṁ ñāṇaṁ tathaṁ tassa,

<!-- @segment snp5.15:4.6 -->

Brāhmaṇassa vusīmato”ti.

<!-- @segment snp5.15:4.7 -->

Posālamāṇavapucchā cuddasamā.
