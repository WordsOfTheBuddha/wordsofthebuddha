<!-- @segment snp4.1:1.1 -->

Kāmaṁ kāmayamānassa,

<!-- @segment snp4.1:1.2 -->

tassa ce taṁ samijjhati;

<!-- @segment snp4.1:1.3 -->

Addhā pītimano hoti,

<!-- @segment snp4.1:1.4 -->

laddhā macco yadicchati.

<!-- @segment snp4.1:2.1 -->

Tassa ce kāmayānassa,

<!-- @segment snp4.1:2.2 -->

chandajātassa jantuno;

<!-- @segment snp4.1:2.3 -->

Te kāmā parihāyanti,

<!-- @segment snp4.1:2.4 -->

sallaviddhova ruppati.

<!-- @segment snp4.1:3.1 -->

Yo kāme parivajjeti,

<!-- @segment snp4.1:3.2 -->

sappasseva padā siro;

<!-- @segment snp4.1:3.3 -->

Somaṁ visattikaṁ loke,

<!-- @segment snp4.1:3.4 -->

sato samativattati.

<!-- @segment snp4.1:4.1 -->

Khettaṁ vatthuṁ hiraññaṁ vā,

<!-- @segment snp4.1:4.2 -->

gavāssaṁ dāsaporisaṁ;

<!-- @segment snp4.1:4.3 -->

Thiyo bandhū puthu kāme,

<!-- @segment snp4.1:4.4 -->

yo naro anugijjhati.

<!-- @segment snp4.1:5.1 -->

Abalā naṁ balīyanti,

<!-- @segment snp4.1:5.2 -->

maddantenaṁ parissayā;

<!-- @segment snp4.1:5.3 -->

Tato naṁ dukkhamanveti,

<!-- @segment snp4.1:5.4 -->

nāvaṁ bhinnamivodakaṁ.

<!-- @segment snp4.1:6.1 -->

Tasmā jantu sadā sato,

<!-- @segment snp4.1:6.2 -->

Kāmāni parivajjaye;

<!-- @segment snp4.1:6.3 -->

Te pahāya tare oghaṁ,

<!-- @segment snp4.1:6.4 -->

Nāvaṁ sitvāva pāragūti.

<!-- @segment snp4.1:6.5 -->

Kāmasuttaṁ paṭhamaṁ.
