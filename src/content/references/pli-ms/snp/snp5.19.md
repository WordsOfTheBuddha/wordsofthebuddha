<!-- @segment snp5.19:1.1 -->

“Pārāyanamanugāyissaṁ,

<!-- @segment snp5.19:1.2 -->

(iccāyasmā piṅgiyo)

<!-- @segment snp5.19:1.3 -->

Yathāddakkhi tathākkhāsi;

<!-- @segment snp5.19:1.4 -->

Vimalo bhūrimedhaso,

<!-- @segment snp5.19:1.5 -->

Nikkāmo nibbano nāgo;

<!-- @segment snp5.19:1.6 -->

Kissa hetu musā bhaṇe.

<!-- @segment snp5.19:2.1 -->

Pahīnamalamohassa,

<!-- @segment snp5.19:2.2 -->

mānamakkhappahāyino;

<!-- @segment snp5.19:2.3 -->

Handāhaṁ kittayissāmi,

<!-- @segment snp5.19:2.4 -->

giraṁ vaṇṇūpasañhitaṁ.

<!-- @segment snp5.19:3.1 -->

Tamonudo buddho samantacakkhu,

<!-- @segment snp5.19:3.2 -->

Lokantagū sabbabhavātivatto;

<!-- @segment snp5.19:3.3 -->

Anāsavo sabbadukkhappahīno,

<!-- @segment snp5.19:3.4 -->

Saccavhayo brahme upāsito me.

<!-- @segment snp5.19:4.1 -->

Dijo yathā kubbanakaṁ pahāya,

<!-- @segment snp5.19:4.2 -->

Bahupphalaṁ kānanamāvaseyya;

<!-- @segment snp5.19:4.3 -->

Evampahaṁ appadasse pahāya,

<!-- @segment snp5.19:4.4 -->

Mahodadhiṁ haṁsoriva ajjhapatto.

<!-- @segment snp5.19:5.1 -->

Ye me pubbe viyākaṁsu,

<!-- @segment snp5.19:5.2 -->

Huraṁ gotamasāsanā;

<!-- @segment snp5.19:5.3 -->

Iccāsi iti bhavissati,

<!-- @segment snp5.19:5.4 -->

Sabbaṁ taṁ itihītihaṁ;

<!-- @segment snp5.19:5.5 -->

Sabbaṁ taṁ takkavaḍḍhanaṁ.

<!-- @segment snp5.19:6.1 -->

Eko tamanudāsino,

<!-- @segment snp5.19:6.2 -->

jutimā so pabhaṅkaro;

<!-- @segment snp5.19:6.3 -->

Gotamo bhūripaññāṇo,

<!-- @segment snp5.19:6.4 -->

gotamo bhūrimedhaso.

<!-- @segment snp5.19:7.1 -->

Yo me dhammamadesesi,

<!-- @segment snp5.19:7.2 -->

Sandiṭṭhikamakālikaṁ;

<!-- @segment snp5.19:7.3 -->

Taṇhakkhayamanītikaṁ,

<!-- @segment snp5.19:7.4 -->

Yassa natthi upamā kvaci”.

<!-- @segment snp5.19:8.1 -->

“Kiṁ nu tamhā vippavasasi,

<!-- @segment snp5.19:8.2 -->

muhuttamapi piṅgiya;

<!-- @segment snp5.19:8.3 -->

Gotamā bhūripaññāṇā,

<!-- @segment snp5.19:8.4 -->

gotamā bhūrimedhasā.

<!-- @segment snp5.19:9.1 -->

Yo te dhammamadesesi,

<!-- @segment snp5.19:9.2 -->

Sandiṭṭhikamakālikaṁ;

<!-- @segment snp5.19:9.3 -->

Taṇhakkhayamanītikaṁ,

<!-- @segment snp5.19:9.4 -->

Yassa natthi upamā kvaci”.

<!-- @segment snp5.19:10.1 -->

“Nāhaṁ tamhā vippavasāmi,

<!-- @segment snp5.19:10.2 -->

Muhuttamapi brāhmaṇa;

<!-- @segment snp5.19:10.3 -->

Gotamā bhūripaññāṇā,

<!-- @segment snp5.19:10.4 -->

Gotamā bhūrimedhasā.

<!-- @segment snp5.19:11.1 -->

Yo me dhammamadesesi,

<!-- @segment snp5.19:11.2 -->

Sandiṭṭhikamakālikaṁ;

<!-- @segment snp5.19:11.3 -->

Taṇhakkhayamanītikaṁ,

<!-- @segment snp5.19:11.4 -->

Yassa natthi upamā kvaci.

<!-- @segment snp5.19:12.1 -->

Passāmi naṁ manasā cakkhunāva,

<!-- @segment snp5.19:12.2 -->

Rattindivaṁ brāhmaṇa appamatto;

<!-- @segment snp5.19:12.3 -->

Namassamāno vivasemi rattiṁ,

<!-- @segment snp5.19:12.4 -->

Teneva maññāmi avippavāsaṁ.

<!-- @segment snp5.19:13.1 -->

Saddhā ca pīti ca mano sati ca,

<!-- @segment snp5.19:13.2 -->

Nāpentime gotamasāsanamhā;

<!-- @segment snp5.19:13.3 -->

Yaṁ yaṁ disaṁ vajati bhūripañño,

<!-- @segment snp5.19:13.4 -->

Sa tena teneva natohamasmi.

<!-- @segment snp5.19:14.1 -->

Jiṇṇassa me dubbalathāmakassa,

<!-- @segment snp5.19:14.2 -->

Teneva kāyo na paleti tattha;

<!-- @segment snp5.19:14.3 -->

Saṅkappayantāya vajāmi niccaṁ,

<!-- @segment snp5.19:14.4 -->

Mano hi me brāhmaṇa tena yutto.

<!-- @segment snp5.19:15.1 -->

Paṅke sayāno pariphandamāno,

<!-- @segment snp5.19:15.2 -->

Dīpā dīpaṁ upaplaviṁ;

<!-- @segment snp5.19:15.3 -->

Athaddasāsiṁ sambuddhaṁ,

<!-- @segment snp5.19:15.4 -->

Oghatiṇṇamanāsavaṁ”.

<!-- @segment snp5.19:16.1 -->

“Yathā ahū vakkali muttasaddho,

<!-- @segment snp5.19:16.2 -->

Bhadrāvudho āḷavigotamo ca;

<!-- @segment snp5.19:16.3 -->

Evamevaṁ tvampi pamuñcassu saddhaṁ,

<!-- @segment snp5.19:16.4 -->

Gamissasi tvaṁ piṅgiya maccudheyyassa pāraṁ”.

<!-- @segment snp5.19:17.1 -->

“Esa bhiyyo pasīdāmi,

<!-- @segment snp5.19:17.2 -->

Sutvāna munino vaco;

<!-- @segment snp5.19:17.3 -->

Vivaṭṭacchado sambuddho,

<!-- @segment snp5.19:17.4 -->

Akhilo paṭibhānavā.

<!-- @segment snp5.19:18.1 -->

Adhideve abhiññāya,

<!-- @segment snp5.19:18.2 -->

Sabbaṁ vedi varovaraṁ;

<!-- @segment snp5.19:18.3 -->

Pañhānantakaro satthā,

<!-- @segment snp5.19:18.4 -->

Kaṅkhīnaṁ paṭijānataṁ.

<!-- @segment snp5.19:19.1 -->

Asaṁhīraṁ asaṅkuppaṁ,

<!-- @segment snp5.19:19.2 -->

Yassa natthi upamā kvaci;

<!-- @segment snp5.19:19.3 -->

Addhā gamissāmi na mettha kaṅkhā,

<!-- @segment snp5.19:19.4 -->

Evaṁ maṁ dhārehi adhimuttacittan”ti.

<!-- @segment snp5.19:19.5 -->

Pārāyanavaggo pañcamo.

<!-- @segment snp5.19:19.6 -->

Suttuddānaṁ

<!-- @segment snp5.19:20.1 -->

Urago dhaniyopi ca,

<!-- @segment snp5.19:20.2 -->

Khaggavisāṇo kasi ca;

<!-- @segment snp5.19:20.3 -->

Cundo bhavo punadeva,

<!-- @segment snp5.19:20.4 -->

Vasalo ca karaṇīyañca;

<!-- @segment snp5.19:20.5 -->

Hemavato atha yakkho,

<!-- @segment snp5.19:20.6 -->

Vijayasuttaṁ munisuttavaranti.

<!-- @segment snp5.19:21.1 -->

Paṭhamakaṭṭhavaro varavaggo,

<!-- @segment snp5.19:21.2 -->

Dvādasasuttadharo suvibhatto;

<!-- @segment snp5.19:21.3 -->

Desito cakkhumatā vimalena,

<!-- @segment snp5.19:21.4 -->

Suyyati vaggavaro uragoti.

<!-- @segment snp5.19:22.1 -->

Ratanāmagandho hirimaṅgalanāmo,

<!-- @segment snp5.19:22.2 -->

Sucilomakapilo ca brāhmaṇadhammo;

<!-- @segment snp5.19:22.3 -->

Nāvā kiṁsīlauṭṭhahano ca,

<!-- @segment snp5.19:22.4 -->

Rāhulo ca punapi vaṅgīso.

<!-- @segment snp5.19:23.1 -->

Sammāparibbājanīyopi cettha,

<!-- @segment snp5.19:23.2 -->

Dhammikasuttavaro suvibhatto;

<!-- @segment snp5.19:23.3 -->

Cuddasasuttadharo dutiyamhi,

<!-- @segment snp5.19:23.4 -->

Cūḷakavaggavaroti tamāhu.

<!-- @segment snp5.19:24.1 -->

Pabbajjapadhānasubhāsitanāmo,

<!-- @segment snp5.19:24.2 -->

Pūraḷāso punadeva māgho ca;

<!-- @segment snp5.19:24.3 -->

Sabhiyaṁ keṇiyameva sallanāmo,

<!-- @segment snp5.19:24.4 -->

Vāseṭṭhavaro kālikopi ca.

<!-- @segment snp5.19:25.1 -->

Nālakasuttavaro suvibhatto,

<!-- @segment snp5.19:25.2 -->

Taṁ anupassī tathā punadeva;

<!-- @segment snp5.19:25.3 -->

Dvādasasuttadharo tatiyamhi,

<!-- @segment snp5.19:25.4 -->

Suyyati vaggavaro mahānāmo.

<!-- @segment snp5.19:26.1 -->

Kāmaguhaṭṭhakaduṭṭhakanāmā,

<!-- @segment snp5.19:26.2 -->

Suddhavaro paramaṭṭhakanāmo;

<!-- @segment snp5.19:26.3 -->

Jarā mettiyavaro suvibhatto,

<!-- @segment snp5.19:26.4 -->

Pasūramāgaṇḍiyā purābhedo.

<!-- @segment snp5.19:27.1 -->

Kalahavivādo ubho viyuhā ca,

<!-- @segment snp5.19:27.2 -->

Tuvaṭakaattadaṇḍasāriputtā;

<!-- @segment snp5.19:27.3 -->

Soḷasasuttadharo catutthamhi,

<!-- @segment snp5.19:27.4 -->

Aṭṭhakavaggavaroti tamāhu.

<!-- @segment snp5.19:28.1 -->

Magadhe janapade ramaṇīye,

<!-- @segment snp5.19:28.2 -->

Desavare katapuññanivese;

<!-- @segment snp5.19:28.3 -->

Pāsāṇakacetiyavare suvibhatte,

<!-- @segment snp5.19:28.4 -->

Vasi bhagavā gaṇaseṭṭho.

<!-- @segment snp5.19:29.1 -->

Ubhayavāsamāgatiyamhi,

<!-- @segment snp5.19:29.2 -->

Dvādasayojaniyā parisāya;

<!-- @segment snp5.19:29.3 -->

Soḷasabrāhmaṇānaṁ kira puṭṭho,

<!-- @segment snp5.19:29.4 -->

Pucchāya soḷasapañhakammiyā;

<!-- @segment snp5.19:29.5 -->

Nippakāsayi dhammamadāsi.

<!-- @segment snp5.19:30.1 -->

Atthapakāsakabyañjanapuṇṇaṁ,

<!-- @segment snp5.19:30.2 -->

Dhammamadesesi parakhemajaniyaṁ;

<!-- @segment snp5.19:30.3 -->

Lokahitāya jino dvipadaggo,

<!-- @segment snp5.19:30.4 -->

Suttavaraṁ bahudhammavicitraṁ;

<!-- @segment snp5.19:30.5 -->

Sabbakilesapamocanahetuṁ,

<!-- @segment snp5.19:30.6 -->

Desayi suttavaraṁ dvipadaggo.

<!-- @segment snp5.19:31.1 -->

Byañjanamatthapadaṁ samayuttaṁ,

<!-- @segment snp5.19:31.2 -->

Akkharasaññitaopamagāḷhaṁ;

<!-- @segment snp5.19:31.3 -->

Lokavicāraṇañāṇapabhaggaṁ,

<!-- @segment snp5.19:31.4 -->

Desayi suttavaraṁ dvipadaggo.

<!-- @segment snp5.19:32.1 -->

Rāgamale amalaṁ vimalaggaṁ,

<!-- @segment snp5.19:32.2 -->

Dosamale amalaṁ vimalaggaṁ;

<!-- @segment snp5.19:32.3 -->

Mohamale amalaṁ vimalaggaṁ,

<!-- @segment snp5.19:32.4 -->

Lokavicāraṇañāṇapabhaggaṁ;

<!-- @segment snp5.19:32.5 -->

Desayi suttavaraṁ dvipadaggo.

<!-- @segment snp5.19:33.1 -->

Klesamale amalaṁ vimalaggaṁ,

<!-- @segment snp5.19:33.2 -->

Duccaritamale amalaṁ vimalaggaṁ;

<!-- @segment snp5.19:33.3 -->

Lokavicāraṇañāṇapabhaggaṁ,

<!-- @segment snp5.19:33.4 -->

Desayi suttavaraṁ dvipadaggo.

<!-- @segment snp5.19:34.1 -->

Āsavabandhanayogākilesaṁ,

<!-- @segment snp5.19:34.2 -->

Nīvaraṇāni ca tīṇi malāni;

<!-- @segment snp5.19:34.3 -->

Tassa kilesapamocanahetuṁ,

<!-- @segment snp5.19:34.4 -->

Desayi suttavaraṁ dvipadaggo.

<!-- @segment snp5.19:35.1 -->

Nimmalasabbakilesapanūdaṁ,

<!-- @segment snp5.19:35.2 -->

Rāgavirāgamanejamasokaṁ;

<!-- @segment snp5.19:35.3 -->

Santapaṇītasududdasadhammaṁ,

<!-- @segment snp5.19:35.4 -->

Desayi suttavaraṁ dvipadaggo.

<!-- @segment snp5.19:36.1 -->

Rāgañca dosakamabhañjitasantaṁ,

<!-- @segment snp5.19:36.2 -->

Yonicatuggatipañcaviññāṇaṁ;

<!-- @segment snp5.19:36.3 -->

Taṇhāratacchadanatāṇalatāpamokkhaṁ,

<!-- @segment snp5.19:37.1 -->

Gambhīraduddasasaṇhanipuṇaṁ,

<!-- @segment snp5.19:37.2 -->

Paṇḍitavedaniyaṁ nipuṇatthaṁ;

<!-- @segment snp5.19:37.4 -->

Desayi suttavaraṁ dvipadaggo.

<!-- @segment snp5.19:38.1 -->
