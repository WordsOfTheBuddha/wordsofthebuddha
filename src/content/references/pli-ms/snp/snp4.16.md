<!-- @segment snp4.16:1.1 -->

“Na me diṭṭho ito pubbe,

<!-- @segment snp4.16:1.2 -->

(iccāyasmā sāriputto)

<!-- @segment snp4.16:1.3 -->

Na suto uda kassaci;

<!-- @segment snp4.16:1.4 -->

Evaṁ vagguvado satthā,

<!-- @segment snp4.16:1.5 -->

Tusitā gaṇimāgato.

<!-- @segment snp4.16:2.1 -->

Sadevakassa lokassa,

<!-- @segment snp4.16:2.2 -->

yathā dissati cakkhumā;

<!-- @segment snp4.16:2.3 -->

Sabbaṁ tamaṁ vinodetvā,

<!-- @segment snp4.16:2.4 -->

ekova ratimajjhagā.

<!-- @segment snp4.16:3.1 -->

Taṁ buddhaṁ asitaṁ tādiṁ,

<!-- @segment snp4.16:3.2 -->

akuhaṁ gaṇimāgataṁ;

<!-- @segment snp4.16:3.3 -->

Bahūnamidha baddhānaṁ,

<!-- @segment snp4.16:3.4 -->

atthi pañhena āgamaṁ.

<!-- @segment snp4.16:4.1 -->

Bhikkhuno vijigucchato,

<!-- @segment snp4.16:4.2 -->

bhajato rittamāsanaṁ;

<!-- @segment snp4.16:4.3 -->

Rukkhamūlaṁ susānaṁ vā,

<!-- @segment snp4.16:4.4 -->

pabbatānaṁ guhāsu vā.

<!-- @segment snp4.16:5.1 -->

Uccāvacesu sayanesu,

<!-- @segment snp4.16:5.2 -->

kīvanto tattha bheravā;

<!-- @segment snp4.16:5.3 -->

Yehi bhikkhu na vedheyya,

<!-- @segment snp4.16:5.4 -->

nigghose sayanāsane.

<!-- @segment snp4.16:6.1 -->

Katī parissayā loke,

<!-- @segment snp4.16:6.2 -->

gacchato agataṁ disaṁ;

<!-- @segment snp4.16:6.3 -->

Ye bhikkhu abhisambhave,

<!-- @segment snp4.16:6.4 -->

pantamhi sayanāsane.

<!-- @segment snp4.16:7.1 -->

Kyāssa byappathayo assu,

<!-- @segment snp4.16:7.2 -->

kyāssassu idha gocarā;

<!-- @segment snp4.16:7.3 -->

Kāni sīlabbatānāssu,

<!-- @segment snp4.16:7.4 -->

pahitattassa bhikkhuno.

<!-- @segment snp4.16:8.1 -->

Kaṁ so sikkhaṁ samādāya,

<!-- @segment snp4.16:8.2 -->

ekodi nipako sato;

<!-- @segment snp4.16:8.3 -->

Kammāro rajatasseva,

<!-- @segment snp4.16:8.4 -->

niddhame malamattano”.

<!-- @segment snp4.16:9.1 -->

“Vijigucchamānassa yadidaṁ phāsu,

<!-- @segment snp4.16:9.2 -->

(sāriputtāti bhagavā)

<!-- @segment snp4.16:9.3 -->

Rittāsanaṁ sayanaṁ sevato ce;

<!-- @segment snp4.16:9.4 -->

Sambodhikāmassa yathānudhammaṁ,

<!-- @segment snp4.16:9.5 -->

Taṁ te pavakkhāmi yathā pajānaṁ.

<!-- @segment snp4.16:10.1 -->

Pañcannaṁ dhīro bhayānaṁ na bhāye,

<!-- @segment snp4.16:10.2 -->

Bhikkhu sato sapariyantacārī;

<!-- @segment snp4.16:10.3 -->

Ḍaṁsādhipātānaṁ sarīsapānaṁ,

<!-- @segment snp4.16:10.4 -->

Manussaphassānaṁ catuppadānaṁ.

<!-- @segment snp4.16:11.1 -->

Paradhammikānampi na santaseyya,

<!-- @segment snp4.16:11.2 -->

Disvāpi tesaṁ bahubheravāni;

<!-- @segment snp4.16:11.3 -->

Athāparāni abhisambhaveyya,

<!-- @segment snp4.16:11.4 -->

Parissayāni kusalānuesī.

<!-- @segment snp4.16:12.1 -->

Ātaṅkaphassena khudāya phuṭṭho,

<!-- @segment snp4.16:12.2 -->

Sītaṁ athuṇhaṁ adhivāsayeyya;

<!-- @segment snp4.16:12.3 -->

So tehi phuṭṭho bahudhā anoko,

<!-- @segment snp4.16:12.4 -->

Viriyaṁ parakkammadaḷhaṁ kareyya.

<!-- @segment snp4.16:13.1 -->

Theyyaṁ na kāre na musā bhaṇeyya,

<!-- @segment snp4.16:13.2 -->

Mettāya phasse tasathāvarāni;

<!-- @segment snp4.16:13.3 -->

Yadāvilattaṁ manaso vijaññā,

<!-- @segment snp4.16:13.4 -->

Kaṇhassa pakkhoti vinodayeyya.

<!-- @segment snp4.16:14.1 -->

Kodhātimānassa vasaṁ na gacche,

<!-- @segment snp4.16:14.2 -->

Mūlampi tesaṁ palikhañña tiṭṭhe;

<!-- @segment snp4.16:14.3 -->

Athappiyaṁ vā pana appiyaṁ vā,

<!-- @segment snp4.16:14.4 -->

Addhā bhavanto abhisambhaveyya.

<!-- @segment snp4.16:15.1 -->

Paññaṁ purakkhatvā kalyāṇapīti,

<!-- @segment snp4.16:15.2 -->

Vikkhambhaye tāni parissayāni;

<!-- @segment snp4.16:15.3 -->

Aratiṁ sahetha sayanamhi pante,

<!-- @segment snp4.16:15.4 -->

Caturo sahetha paridevadhamme.

<!-- @segment snp4.16:16.1 -->

Kiṁsū asissāmi kuva vā asissaṁ,

<!-- @segment snp4.16:16.2 -->

Dukkhaṁ vata settha kvajja sessaṁ;

<!-- @segment snp4.16:16.3 -->

Ete vitakke paridevaneyye,

<!-- @segment snp4.16:16.4 -->

Vinayetha sekho aniketacārī.

<!-- @segment snp4.16:17.1 -->

Annañca laddhā vasanañca kāle,

<!-- @segment snp4.16:17.2 -->

Mattaṁ so jaññā idha tosanatthaṁ;

<!-- @segment snp4.16:17.3 -->

So tesu gutto yatacāri gāme,

<!-- @segment snp4.16:17.4 -->

Rusitopi vācaṁ pharusaṁ na vajjā.

<!-- @segment snp4.16:18.1 -->

Okkhittacakkhu na ca pādalolo,

<!-- @segment snp4.16:18.2 -->

Jhānānuyutto bahujāgarassa;

<!-- @segment snp4.16:18.3 -->

Upekkhamārabbha samāhitatto,

<!-- @segment snp4.16:18.4 -->

Takkāsayaṁ kukkucciyūpachinde.

<!-- @segment snp4.16:19.1 -->

Cudito vacībhi satimābhinande,

<!-- @segment snp4.16:19.2 -->

Sabrahmacārīsu khilaṁ pabhinde;

<!-- @segment snp4.16:19.3 -->

Vācaṁ pamuñce kusalaṁ nātivelaṁ,

<!-- @segment snp4.16:19.4 -->

Janavādadhammāya na cetayeyya.

<!-- @segment snp4.16:20.1 -->

Athāparaṁ pañca rajāni loke,

<!-- @segment snp4.16:20.2 -->

Yesaṁ satīmā vinayāya sikkhe;

<!-- @segment snp4.16:20.3 -->

Rūpesu saddesu atho rasesu,

<!-- @segment snp4.16:20.4 -->

Gandhesu phassesu sahetha rāgaṁ.

<!-- @segment snp4.16:21.1 -->

Etesu dhammesu vineyya chandaṁ,

<!-- @segment snp4.16:21.2 -->

Bhikkhu satimā suvimuttacitto;

<!-- @segment snp4.16:21.3 -->

Kālena so sammā dhammaṁ parivīmaṁsamāno,

<!-- @segment snp4.16:21.4 -->

Ekodibhūto vihane tamaṁ so”ti.

<!-- @segment snp4.16:21.5 -->

Sāriputtasuttaṁ soḷasamaṁ.

<!-- @segment snp4.16:21.6 -->

Aṭṭhakavaggo catuttho.

<!-- @segment snp4.16:22.0 -->

Tassuddānaṁ

<!-- @segment snp4.16:22.1 -->

Kāmaṁ guhañca duṭṭhā ca,

<!-- @segment snp4.16:22.2 -->

suddhañca paramā jarā;

<!-- @segment snp4.16:22.3 -->

Metteyyo ca pasūro ca,

<!-- @segment snp4.16:22.4 -->

māgaṇḍi purābhedanaṁ.

<!-- @segment snp4.16:23.1 -->

Kalahaṁ dve ca byūhāni,

<!-- @segment snp4.16:23.2 -->

punadeva tuvaṭṭakaṁ;

<!-- @segment snp4.16:23.3 -->

Attadaṇḍavaraṁ suttaṁ,

<!-- @segment snp4.16:23.4 -->

therapuṭṭhena soḷasa;

<!-- @segment snp4.16:23.5 -->

Iti etāni suttāni,

<!-- @segment snp4.16:23.6 -->

sabbānaṭṭhakavaggikāti.
