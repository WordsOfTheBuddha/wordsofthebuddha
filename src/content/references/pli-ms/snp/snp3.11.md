<!-- @segment snp3.11:1.1 -->

Ānandajāte tidasagaṇe patīte,

<!-- @segment snp3.11:1.2 -->

Sakkañca indaṁ sucivasane ca deve;

<!-- @segment snp3.11:1.3 -->

Dussaṁ gahetvā atiriva thomayante,

<!-- @segment snp3.11:1.4 -->

Asito isi addasa divāvihāre.

<!-- @segment snp3.11:2.1 -->

Disvāna deve muditamane udagge,

<!-- @segment snp3.11:2.2 -->

Cittiṁ karitvāna idamavoca tattha;

<!-- @segment snp3.11:2.3 -->

“Kiṁ devasaṅgho atiriva kalyarūpo,

<!-- @segment snp3.11:2.4 -->

Dussaṁ gahetvā ramayatha kiṁ paṭicca.

<!-- @segment snp3.11:3.1 -->

Yadāpi āsī asurehi saṅgamo,

<!-- @segment snp3.11:3.2 -->

Jayo surānaṁ asurā parājitā;

<!-- @segment snp3.11:3.3 -->

Tadāpi netādiso lomahaṁsano,

<!-- @segment snp3.11:3.4 -->

Kimabbhutaṁ daṭṭhu marū pamoditā.

<!-- @segment snp3.11:4.1 -->

Seḷenti gāyanti ca vādayanti ca,

<!-- @segment snp3.11:4.2 -->

Bhujāni phoṭenti ca naccayanti ca;

<!-- @segment snp3.11:4.3 -->

Pucchāmi vohaṁ merumuddhavāsine,

<!-- @segment snp3.11:4.4 -->

Dhunātha me saṁsayaṁ khippa mārisā”.

<!-- @segment snp3.11:5.1 -->

“So bodhisatto ratanavaro atulyo,

<!-- @segment snp3.11:5.2 -->

Manussaloke hitasukhatthāya jāto;

<!-- @segment snp3.11:5.3 -->

Sakyāna gāme janapade lumbineyye,

<!-- @segment snp3.11:5.4 -->

Tenamha tuṭṭhā atiriva kalyarūpā.

<!-- @segment snp3.11:6.1 -->

So sabbasattuttamo aggapuggalo,

<!-- @segment snp3.11:6.2 -->

Narāsabho sabbapajānamuttamo;

<!-- @segment snp3.11:6.3 -->

Vattessati cakkamisivhaye vane,

<!-- @segment snp3.11:6.4 -->

Nadaṁva sīho balavā migābhibhū”.

<!-- @segment snp3.11:7.1 -->

Taṁ saddaṁ sutvā turitamavasarī so,

<!-- @segment snp3.11:7.2 -->

Suddhodanassa tada bhavanaṁ upāvisi;

<!-- @segment snp3.11:7.3 -->

Nisajja tattha idamavocāsi sakye,

<!-- @segment snp3.11:7.4 -->

“Kuhiṁ kumāro ahamapi daṭṭhukāmo”.

<!-- @segment snp3.11:8.1 -->

Tato kumāraṁ jalitamiva suvaṇṇaṁ,

<!-- @segment snp3.11:8.2 -->

Ukkāmukheva sukusalasampahaṭṭhaṁ;

<!-- @segment snp3.11:8.3 -->

Daddallamānaṁ siriyā anomavaṇṇaṁ,

<!-- @segment snp3.11:8.4 -->

Dassesu puttaṁ asitavhayassa sakyā.

<!-- @segment snp3.11:9.1 -->

Disvā kumāraṁ sikhimiva pajjalantaṁ,

<!-- @segment snp3.11:9.2 -->

Tārāsabhaṁva nabhasigamaṁ visuddhaṁ;

<!-- @segment snp3.11:9.3 -->

Suriyaṁ tapantaṁ saradarivabbhamuttaṁ,

<!-- @segment snp3.11:9.4 -->

Ānandajāto vipulamalattha pītiṁ.

<!-- @segment snp3.11:10.1 -->

Anekasākhañca sahassamaṇḍalaṁ,

<!-- @segment snp3.11:10.2 -->

Chattaṁ marū dhārayumantalikkhe;

<!-- @segment snp3.11:10.3 -->

Suvaṇṇadaṇḍā vītipatanti cāmarā,

<!-- @segment snp3.11:10.4 -->

Na dissare cāmarachattagāhakā.

<!-- @segment snp3.11:11.1 -->

Disvā jaṭī kaṇhasirivhayo isi,

<!-- @segment snp3.11:11.2 -->

Suvaṇṇanikkhaṁ viya paṇḍukambale;

<!-- @segment snp3.11:11.3 -->

Setañca chattaṁ dhariyanta muddhani,

<!-- @segment snp3.11:11.4 -->

Udaggacitto sumano paṭiggahe.

<!-- @segment snp3.11:12.1 -->

Paṭiggahetvā pana sakyapuṅgavaṁ,

<!-- @segment snp3.11:12.2 -->

Jigīsako lakkhaṇamantapāragū;

<!-- @segment snp3.11:12.3 -->

Pasannacitto giramabbhudīrayi,

<!-- @segment snp3.11:12.4 -->

“Anuttarāyaṁ dvipadānamuttamo”.

<!-- @segment snp3.11:13.1 -->

Athattano gamanamanussaranto,

<!-- @segment snp3.11:13.2 -->

Akalyarūpo gaḷayati assukāni;

<!-- @segment snp3.11:13.3 -->

Disvāna sakyā isimavocuṁ rudantaṁ,

<!-- @segment snp3.11:13.4 -->

“No ce kumāre bhavissati antarāyo”.

<!-- @segment snp3.11:14.1 -->

Disvāna sakye isimavoca akalye,

<!-- @segment snp3.11:14.2 -->

“Nāhaṁ kumāre ahitamanussarāmi;

<!-- @segment snp3.11:14.3 -->

Na cāpimassa bhavissati antarāyo,

<!-- @segment snp3.11:14.4 -->

Na orakāyaṁ adhimānasā bhavātha.

<!-- @segment snp3.11:15.1 -->

Sambodhiyaggaṁ phusissatāyaṁ kumāro,

<!-- @segment snp3.11:15.2 -->

So dhammacakkaṁ paramavisuddhadassī;

<!-- @segment snp3.11:15.3 -->

Vattessatāyaṁ bahujanahitānukampī,

<!-- @segment snp3.11:15.4 -->

Vitthārikassa bhavissati brahmacariyaṁ.

<!-- @segment snp3.11:16.1 -->

Mamañca āyu na ciramidhāvaseso,

<!-- @segment snp3.11:16.2 -->

Athantarā me bhavissati kālakiriyā;

<!-- @segment snp3.11:16.3 -->

Sohaṁ na sossaṁ asamadhurassa dhammaṁ,

<!-- @segment snp3.11:16.4 -->

Tenamhi aṭṭo byasanaṅgato aghāvī”.

<!-- @segment snp3.11:17.1 -->

So sākiyānaṁ vipulaṁ janetvā pītiṁ,

<!-- @segment snp3.11:17.2 -->

Antepuramhā niggamā brahmacārī;

<!-- @segment snp3.11:17.3 -->

So bhāgineyyaṁ sayaṁ anukampamāno,

<!-- @segment snp3.11:17.4 -->

Samādapesi asamadhurassa dhamme.

<!-- @segment snp3.11:18.1 -->

“Buddhoti ghosaṁ yada parato suṇāsi,

<!-- @segment snp3.11:18.2 -->

Sambodhipatto vivarati dhammamaggaṁ;

<!-- @segment snp3.11:18.3 -->

Gantvāna tattha samayaṁ paripucchamāno,

<!-- @segment snp3.11:18.4 -->

Carassu tasmiṁ bhagavati brahmacariyaṁ”.

<!-- @segment snp3.11:19.1 -->

Tenānusiṭṭho hitamanena tādinā,

<!-- @segment snp3.11:19.2 -->

Anāgate paramavisuddhadassinā;

<!-- @segment snp3.11:19.3 -->

So nālako upacitapuññasañcayo,

<!-- @segment snp3.11:19.4 -->

Jinaṁ patikkhaṁ parivasi rakkhitindriyo.

<!-- @segment snp3.11:20.1 -->

Sutvāna ghosaṁ jinavaracakkavattane,

<!-- @segment snp3.11:20.2 -->

Gantvāna disvā isinisabhaṁ pasanno;

<!-- @segment snp3.11:20.3 -->

Moneyyaseṭṭhaṁ munipavaraṁ apucchi,

<!-- @segment snp3.11:20.4 -->

Samāgate asitāvhayassa sāsaneti.

<!-- @segment snp3.11:20.5 -->

Vatthugāthā niṭṭhitā.

<!-- @segment snp3.11:21.1 -->

“Aññātametaṁ vacanaṁ,

<!-- @segment snp3.11:21.2 -->

Asitassa yathātathaṁ;

<!-- @segment snp3.11:21.3 -->

Taṁ taṁ gotama pucchāmi,

<!-- @segment snp3.11:21.4 -->

Sabbadhammāna pāraguṁ.

<!-- @segment snp3.11:22.1 -->

Anagāriyupetassa,

<!-- @segment snp3.11:22.2 -->

Bhikkhācariyaṁ jigīsato;

<!-- @segment snp3.11:22.3 -->

Muni pabrūhi me puṭṭho,

<!-- @segment snp3.11:22.4 -->

Moneyyaṁ uttamaṁ padaṁ”.

<!-- @segment snp3.11:23.1 -->

“Moneyyaṁ te upaññissaṁ,

<!-- @segment snp3.11:23.2 -->

(iti bhagavā)

<!-- @segment snp3.11:23.3 -->

Dukkaraṁ durabhisambhavaṁ;

<!-- @segment snp3.11:23.4 -->

Handa te naṁ pavakkhāmi,

<!-- @segment snp3.11:23.5 -->

Santhambhassu daḷho bhava.

<!-- @segment snp3.11:24.1 -->

Samānabhāgaṁ kubbetha,

<!-- @segment snp3.11:24.2 -->

Gāme akkuṭṭhavanditaṁ;

<!-- @segment snp3.11:24.3 -->

Manopadosaṁ rakkheyya,

<!-- @segment snp3.11:24.4 -->

Santo anuṇṇato care.

<!-- @segment snp3.11:25.1 -->

Uccāvacā niccharanti,

<!-- @segment snp3.11:25.2 -->

Dāye aggisikhūpamā;

<!-- @segment snp3.11:25.3 -->

Nāriyo muniṁ palobhenti,

<!-- @segment snp3.11:25.4 -->

Tā su taṁ mā palobhayuṁ.

<!-- @segment snp3.11:26.1 -->

Virato methunā dhammā,

<!-- @segment snp3.11:26.2 -->

Hitvā kāme paropare;

<!-- @segment snp3.11:26.3 -->

Aviruddho asāratto,

<!-- @segment snp3.11:26.4 -->

Pāṇesu tasathāvare.

<!-- @segment snp3.11:27.1 -->

Yathā ahaṁ tathā ete,

<!-- @segment snp3.11:27.2 -->

Yathā ete tathā ahaṁ;

<!-- @segment snp3.11:27.3 -->

Attānaṁ upamaṁ katvā,

<!-- @segment snp3.11:27.4 -->

Na haneyya na ghātaye.

<!-- @segment snp3.11:28.1 -->

Hitvā icchañca lobhañca,

<!-- @segment snp3.11:28.2 -->

Yattha satto puthujjano;

<!-- @segment snp3.11:28.3 -->

Cakkhumā paṭipajjeyya,

<!-- @segment snp3.11:28.4 -->

Tareyya narakaṁ imaṁ.

<!-- @segment snp3.11:29.1 -->

Ūnūdaro mitāhāro,

<!-- @segment snp3.11:29.2 -->

Appicchassa alolupo;

<!-- @segment snp3.11:29.3 -->

Sadā icchāya nicchāto,

<!-- @segment snp3.11:29.4 -->

Aniccho hoti nibbuto.

<!-- @segment snp3.11:30.1 -->

Sa piṇḍacāraṁ caritvā,

<!-- @segment snp3.11:30.2 -->

Vanantamabhihāraye;

<!-- @segment snp3.11:30.3 -->

Upaṭṭhito rukkhamūlasmiṁ,

<!-- @segment snp3.11:30.4 -->

Āsanūpagato muni.

<!-- @segment snp3.11:31.1 -->

Sa jhānapasuto dhīro,

<!-- @segment snp3.11:31.2 -->

Vanante ramito siyā;

<!-- @segment snp3.11:31.3 -->

Jhāyetha rukkhamūlasmiṁ,

<!-- @segment snp3.11:31.4 -->

Attānamabhitosayaṁ.

<!-- @segment snp3.11:32.1 -->

Tato ratyā vivasāne,

<!-- @segment snp3.11:32.2 -->

Gāmantamabhihāraye;

<!-- @segment snp3.11:32.3 -->

Avhānaṁ nābhinandeyya,

<!-- @segment snp3.11:32.4 -->

Abhihārañca gāmato.

<!-- @segment snp3.11:33.1 -->

Na munī gāmamāgamma,

<!-- @segment snp3.11:33.2 -->

Kulesu sahasā care;

<!-- @segment snp3.11:33.3 -->

Ghāsesanaṁ chinnakatho,

<!-- @segment snp3.11:33.4 -->

Na vācaṁ payutaṁ bhaṇe.

<!-- @segment snp3.11:34.1 -->

Alatthaṁ yadidaṁ sādhu,

<!-- @segment snp3.11:34.2 -->

Nālatthaṁ kusalaṁ iti;

<!-- @segment snp3.11:34.3 -->

Ubhayeneva so tādī,

<!-- @segment snp3.11:34.4 -->

Rukkhaṁvupanivattati.

<!-- @segment snp3.11:35.1 -->

Sa pattapāṇi vicaranto,

<!-- @segment snp3.11:35.2 -->

Amūgo mūgasammato;

<!-- @segment snp3.11:35.3 -->

Appaṁ dānaṁ na hīḷeyya,

<!-- @segment snp3.11:35.4 -->

Dātāraṁ nāvajāniyā.

<!-- @segment snp3.11:36.1 -->

Uccāvacā hi paṭipadā,

<!-- @segment snp3.11:36.2 -->

Samaṇena pakāsitā;

<!-- @segment snp3.11:36.3 -->

Na pāraṁ diguṇaṁ yanti,

<!-- @segment snp3.11:36.4 -->

Nayidaṁ ekaguṇaṁ mutaṁ.

<!-- @segment snp3.11:37.1 -->

Yassa ca visatā natthi,

<!-- @segment snp3.11:37.2 -->

Chinnasotassa bhikkhuno;

<!-- @segment snp3.11:37.3 -->

Kiccākiccappahīnassa,

<!-- @segment snp3.11:37.4 -->

Pariḷāho na vijjati.

<!-- @segment snp3.11:38.1 -->

Moneyyaṁ te upaññissaṁ,

<!-- @segment snp3.11:38.2 -->

Khuradhārūpamo bhave;

<!-- @segment snp3.11:38.3 -->

Jivhāya tālumāhacca,

<!-- @segment snp3.11:38.4 -->

Udare saññato siyā.

<!-- @segment snp3.11:39.1 -->

Alīnacitto ca siyā,

<!-- @segment snp3.11:39.2 -->

Na cāpi bahu cintaye;

<!-- @segment snp3.11:39.3 -->

Nirāmagandho asito,

<!-- @segment snp3.11:39.4 -->

Brahmacariyaparāyaṇo.

<!-- @segment snp3.11:40.1 -->

Ekāsanassa sikkhetha,

<!-- @segment snp3.11:40.2 -->

Samaṇūpāsanassa ca;

<!-- @segment snp3.11:40.3 -->

Ekattaṁ monamakkhātaṁ,

<!-- @segment snp3.11:40.4 -->

Eko ce abhiramissasi;

<!-- @segment snp3.11:40.5 -->

Atha bhāhisi dasadisā.

<!-- @segment snp3.11:41.1 -->

Sutvā dhīrānaṁ nighosaṁ,

<!-- @segment snp3.11:41.2 -->

Jhāyīnaṁ kāmacāginaṁ;

<!-- @segment snp3.11:41.3 -->

Tato hiriñca saddhañca,

<!-- @segment snp3.11:41.4 -->

Bhiyyo kubbetha māmako.

<!-- @segment snp3.11:42.1 -->

Taṁ nadīhi vijānātha,

<!-- @segment snp3.11:42.2 -->

Sobbhesu padaresu ca;

<!-- @segment snp3.11:42.3 -->

Saṇantā yanti kusobbhā,

<!-- @segment snp3.11:42.4 -->

Tuṇhī yanti mahodadhī.

<!-- @segment snp3.11:43.1 -->

Yadūnakaṁ taṁ saṇati,

<!-- @segment snp3.11:43.2 -->

Yaṁ pūraṁ santameva taṁ;

<!-- @segment snp3.11:43.3 -->

Aḍḍhakumbhūpamo bālo,

<!-- @segment snp3.11:43.4 -->

Rahado pūrova paṇḍito.

<!-- @segment snp3.11:44.1 -->

Yaṁ samaṇo bahuṁ bhāsati,

<!-- @segment snp3.11:44.2 -->

Upetaṁ atthasañhitaṁ;

<!-- @segment snp3.11:44.3 -->

Jānaṁ so dhammaṁ deseti,

<!-- @segment snp3.11:44.4 -->

Jānaṁ so bahu bhāsati.

<!-- @segment snp3.11:45.1 -->

Yo ca jānaṁ saṁyatatto,

<!-- @segment snp3.11:45.2 -->

Jānaṁ na bahu bhāsati;

<!-- @segment snp3.11:45.3 -->

Sa munī monamarahati,

<!-- @segment snp3.11:45.4 -->

Sa munī monamajjhagā”ti.

<!-- @segment snp3.11:45.5 -->

Nālakasuttaṁ ekādasamaṁ.
