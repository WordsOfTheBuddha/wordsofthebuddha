<!-- @segment snp4.15:1.1 -->

“Attadaṇḍā bhayaṁ jātaṁ,

<!-- @segment snp4.15:1.2 -->

janaṁ passatha medhagaṁ;

<!-- @segment snp4.15:1.3 -->

Saṁvegaṁ kittayissāmi,

<!-- @segment snp4.15:1.4 -->

yathā saṁvijitaṁ mayā.

<!-- @segment snp4.15:2.1 -->

Phandamānaṁ pajaṁ disvā,

<!-- @segment snp4.15:2.2 -->

macche appodake yathā;

<!-- @segment snp4.15:2.3 -->

Aññamaññehi byāruddhe,

<!-- @segment snp4.15:2.4 -->

disvā maṁ bhayamāvisi.

<!-- @segment snp4.15:3.1 -->

Samantamasāro loko,

<!-- @segment snp4.15:3.2 -->

disā sabbā sameritā;

<!-- @segment snp4.15:3.3 -->

Icchaṁ bhavanamattano,

<!-- @segment snp4.15:3.4 -->

nāddasāsiṁ anositaṁ.

<!-- @segment snp4.15:4.1 -->

Osāne tveva byāruddhe,

<!-- @segment snp4.15:4.2 -->

Disvā me aratī ahu;

<!-- @segment snp4.15:4.3 -->

Athettha sallamaddakkhiṁ,

<!-- @segment snp4.15:4.4 -->

Duddasaṁ hadayanissitaṁ.

<!-- @segment snp4.15:5.1 -->

Yena sallena otiṇṇo,

<!-- @segment snp4.15:5.2 -->

disā sabbā vidhāvati;

<!-- @segment snp4.15:5.3 -->

Tameva sallamabbuyha,

<!-- @segment snp4.15:5.4 -->

na dhāvati na sīdati.

<!-- @segment snp4.15:6.1 -->

Tattha sikkhānugīyanti,

<!-- @segment snp4.15:6.2 -->

yāni loke gadhitāni;

<!-- @segment snp4.15:6.3 -->

Na tesu pasuto siyā,

<!-- @segment snp4.15:6.4 -->

nibbijjha sabbaso kāme;

<!-- @segment snp4.15:6.5 -->

Sikkhe nibbānamattano.

<!-- @segment snp4.15:7.1 -->

Sacco siyā appagabbho,

<!-- @segment snp4.15:7.2 -->

amāyo rittapesuṇo;

<!-- @segment snp4.15:7.3 -->

Akkodhano lobhapāpaṁ,

<!-- @segment snp4.15:7.4 -->

vevicchaṁ vitare muni.

<!-- @segment snp4.15:8.1 -->

Niddaṁ tandiṁ sahe thīnaṁ,

<!-- @segment snp4.15:8.2 -->

pamādena na saṁvase;

<!-- @segment snp4.15:8.3 -->

Atimāne na tiṭṭheyya,

<!-- @segment snp4.15:8.4 -->

nibbānamanaso naro.

<!-- @segment snp4.15:9.1 -->

Mosavajje na nīyetha,

<!-- @segment snp4.15:9.2 -->

rūpe snehaṁ na kubbaye;

<!-- @segment snp4.15:9.3 -->

Mānañca parijāneyya,

<!-- @segment snp4.15:9.4 -->

sāhasā virato care.

<!-- @segment snp4.15:10.1 -->

Purāṇaṁ nābhinandeyya,

<!-- @segment snp4.15:10.2 -->

Nave khantiṁ na kubbaye;

<!-- @segment snp4.15:10.3 -->

Hiyyamāne na soceyya,

<!-- @segment snp4.15:10.4 -->

Ākāsaṁ na sito siyā.

<!-- @segment snp4.15:11.1 -->

Gedhaṁ brūmi mahoghoti,

<!-- @segment snp4.15:11.2 -->

Ājavaṁ brūmi jappanaṁ;

<!-- @segment snp4.15:11.3 -->

Ārammaṇaṁ pakappanaṁ,

<!-- @segment snp4.15:11.4 -->

Kāmapaṅko duraccayo.

<!-- @segment snp4.15:12.1 -->

Saccā avokkamma muni,

<!-- @segment snp4.15:12.2 -->

Thale tiṭṭhati brāhmaṇo;

<!-- @segment snp4.15:12.3 -->

Sabbaṁ so paṭinissajja,

<!-- @segment snp4.15:12.4 -->

Sa ve santoti vuccati.

<!-- @segment snp4.15:13.1 -->

Sa ve vidvā sa vedagū,

<!-- @segment snp4.15:13.2 -->

Ñatvā dhammaṁ anissito;

<!-- @segment snp4.15:13.3 -->

Sammā so loke iriyāno,

<!-- @segment snp4.15:13.4 -->

Na pihetīdha kassaci.

<!-- @segment snp4.15:14.1 -->

Yodha kāme accatari,

<!-- @segment snp4.15:14.2 -->

Saṅgaṁ loke duraccayaṁ;

<!-- @segment snp4.15:14.3 -->

Na so socati nājjheti,

<!-- @segment snp4.15:14.4 -->

Chinnasoto abandhano.

<!-- @segment snp4.15:15.1 -->

Yaṁ pubbe taṁ visosehi,

<!-- @segment snp4.15:15.2 -->

Pacchā te māhu kiñcanaṁ;

<!-- @segment snp4.15:15.3 -->

Majjhe ce no gahessasi,

<!-- @segment snp4.15:15.4 -->

Upasanto carissasi.

<!-- @segment snp4.15:16.1 -->

Sabbaso nāmarūpasmiṁ,

<!-- @segment snp4.15:16.2 -->

Yassa natthi mamāyitaṁ;

<!-- @segment snp4.15:16.3 -->

Asatā ca na socati,

<!-- @segment snp4.15:16.4 -->

Sa ve loke na jīyati.

<!-- @segment snp4.15:17.1 -->

Yassa natthi idaṁ meti,

<!-- @segment snp4.15:17.2 -->

Paresaṁ vāpi kiñcanaṁ;

<!-- @segment snp4.15:17.3 -->

Mamattaṁ so asaṁvindaṁ,

<!-- @segment snp4.15:17.4 -->

Natthi meti na socati.

<!-- @segment snp4.15:18.1 -->

Aniṭṭhurī ananugiddho,

<!-- @segment snp4.15:18.2 -->

Anejo sabbadhī samo;

<!-- @segment snp4.15:18.3 -->

Tamānisaṁsaṁ pabrūmi,

<!-- @segment snp4.15:18.4 -->

Pucchito avikampinaṁ.

<!-- @segment snp4.15:19.1 -->

Anejassa vijānato,

<!-- @segment snp4.15:19.2 -->

Natthi kāci nisaṅkhati;

<!-- @segment snp4.15:19.3 -->

Virato so viyārabbhā,

<!-- @segment snp4.15:19.4 -->

Khemaṁ passati sabbadhi.

<!-- @segment snp4.15:20.1 -->

Na samesu na omesu,

<!-- @segment snp4.15:20.2 -->

Na ussesu vadate muni;

<!-- @segment snp4.15:20.3 -->

Santo so vītamaccharo,

<!-- @segment snp4.15:20.4 -->

Nādeti na nirassatī”ti.

<!-- @segment snp4.15:20.5 -->

Attadaṇḍasuttaṁ pannarasamaṁ.
