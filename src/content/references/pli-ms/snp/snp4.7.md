<!-- @segment snp4.7:1.1 -->

“Methunamanuyuttassa,

<!-- @segment snp4.7:1.2 -->

(iccāyasmā tisso metteyyo)

<!-- @segment snp4.7:1.3 -->

Vighātaṁ brūhi mārisa;

<!-- @segment snp4.7:1.4 -->

Sutvāna tava sāsanaṁ,

<!-- @segment snp4.7:1.5 -->

Viveke sikkhissāmase”.

<!-- @segment snp4.7:2.1 -->

“Methunamanuyuttassa,

<!-- @segment snp4.7:2.2 -->

(metteyyāti bhagavā)

<!-- @segment snp4.7:2.3 -->

Mussate vāpi sāsanaṁ;

<!-- @segment snp4.7:2.4 -->

Micchā ca paṭipajjati,

<!-- @segment snp4.7:2.5 -->

Etaṁ tasmiṁ anāriyaṁ.

<!-- @segment snp4.7:3.1 -->

Eko pubbe caritvāna,

<!-- @segment snp4.7:3.2 -->

methunaṁ yo nisevati;

<!-- @segment snp4.7:3.3 -->

Yānaṁ bhantaṁva taṁ loke,

<!-- @segment snp4.7:3.4 -->

hīnamāhu puthujjanaṁ.

<!-- @segment snp4.7:4.1 -->

Yaso kitti ca yā pubbe,

<!-- @segment snp4.7:4.2 -->

hāyate vāpi tassa sā;

<!-- @segment snp4.7:4.3 -->

Etampi disvā sikkhetha,

<!-- @segment snp4.7:4.4 -->

methunaṁ vippahātave.

<!-- @segment snp4.7:5.1 -->

Saṅkappehi pareto so,

<!-- @segment snp4.7:5.2 -->

kapaṇo viya jhāyati;

<!-- @segment snp4.7:5.3 -->

Sutvā paresaṁ nigghosaṁ,

<!-- @segment snp4.7:5.4 -->

maṅku hoti tathāvidho.

<!-- @segment snp4.7:6.1 -->

Atha satthāni kurute,

<!-- @segment snp4.7:6.2 -->

paravādehi codito;

<!-- @segment snp4.7:6.3 -->

Esa khvassa mahāgedho,

<!-- @segment snp4.7:6.4 -->

mosavajjaṁ pagāhati.

<!-- @segment snp4.7:7.1 -->

Paṇḍitoti samaññāto,

<!-- @segment snp4.7:7.2 -->

ekacariyaṁ adhiṭṭhito;

<!-- @segment snp4.7:7.3 -->

Athāpi methune yutto,

<!-- @segment snp4.7:7.4 -->

mandova parikissati.

<!-- @segment snp4.7:8.1 -->

Etamādīnavaṁ ñatvā,

<!-- @segment snp4.7:8.2 -->

Muni pubbāpare idha;

<!-- @segment snp4.7:8.3 -->

Ekacariyaṁ daḷhaṁ kayirā,

<!-- @segment snp4.7:8.4 -->

Na nisevetha methunaṁ.

<!-- @segment snp4.7:9.1 -->

Vivekaññeva sikkhetha,

<!-- @segment snp4.7:9.2 -->

etaṁ ariyānamuttamaṁ;

<!-- @segment snp4.7:9.3 -->

Na tena seṭṭho maññetha,

<!-- @segment snp4.7:9.4 -->

sa ve nibbānasantike.

<!-- @segment snp4.7:10.1 -->

Rittassa munino carato,

<!-- @segment snp4.7:10.2 -->

Kāmesu anapekkhino;

<!-- @segment snp4.7:10.3 -->

Oghatiṇṇassa pihayanti,

<!-- @segment snp4.7:10.4 -->

Kāmesu gadhitā pajā”ti.

<!-- @segment snp4.7:10.5 -->

Tissametteyyasuttaṁ sattamaṁ.
