<!-- @segment snp5.2:1.1 -->

“Kenassu nivuto loko,

<!-- @segment snp5.2:1.2 -->

(iccāyasmā ajito)

<!-- @segment snp5.2:1.3 -->

Kenassu nappakāsati;

<!-- @segment snp5.2:1.4 -->

Kissābhilepanaṁ brūsi,

<!-- @segment snp5.2:1.5 -->

Kiṁsu tassa mahabbhayaṁ”.

<!-- @segment snp5.2:2.1 -->

“Avijjāya nivuto loko,

<!-- @segment snp5.2:2.2 -->

(ajitāti bhagavā)

<!-- @segment snp5.2:2.3 -->

Vevicchā pamādā nappakāsati;

<!-- @segment snp5.2:2.4 -->

Jappābhilepanaṁ brūmi,

<!-- @segment snp5.2:2.5 -->

Dukkhamassa mahabbhayaṁ”.

<!-- @segment snp5.2:3.1 -->

“Savanti sabbadhi sotā,

<!-- @segment snp5.2:3.2 -->

(iccāyasmā ajito)

<!-- @segment snp5.2:3.3 -->

Sotānaṁ kiṁ nivāraṇaṁ;

<!-- @segment snp5.2:3.4 -->

Sotānaṁ saṁvaraṁ brūhi,

<!-- @segment snp5.2:3.5 -->

Kena sotā pidhiyyare”.

<!-- @segment snp5.2:4.1 -->

“Yāni sotāni lokasmiṁ,

<!-- @segment snp5.2:4.2 -->

(ajitāti bhagavā)

<!-- @segment snp5.2:4.3 -->

Sati tesaṁ nivāraṇaṁ;

<!-- @segment snp5.2:4.4 -->

Sotānaṁ saṁvaraṁ brūmi,

<!-- @segment snp5.2:4.5 -->

Paññāyete pidhiyyare”.

<!-- @segment snp5.2:5.1 -->

“Paññā ceva sati yañca,

<!-- @segment snp5.2:5.2 -->

(iccāyasmā ajito)

<!-- @segment snp5.2:5.3 -->

Nāmarūpañca mārisa;

<!-- @segment snp5.2:5.4 -->

Etaṁ me puṭṭho pabrūhi,

<!-- @segment snp5.2:5.5 -->

Katthetaṁ uparujjhati”.

<!-- @segment snp5.2:6.1 -->

“Yametaṁ pañhaṁ apucchi,

<!-- @segment snp5.2:6.2 -->

ajita taṁ vadāmi te;

<!-- @segment snp5.2:6.3 -->

Yattha nāmañca rūpañca,

<!-- @segment snp5.2:6.4 -->

asesaṁ uparujjhati;

<!-- @segment snp5.2:6.5 -->

Viññāṇassa nirodhena,

<!-- @segment snp5.2:6.6 -->

etthetaṁ uparujjhati”.

<!-- @segment snp5.2:7.1 -->

“Ye ca saṅkhātadhammāse,

<!-- @segment snp5.2:7.2 -->

ye ca sekhā puthū idha;

<!-- @segment snp5.2:7.3 -->

Tesaṁ me nipako iriyaṁ,

<!-- @segment snp5.2:7.4 -->

puṭṭho pabrūhi mārisa”.

<!-- @segment snp5.2:8.1 -->

“Kāmesu nābhigijjheyya,

<!-- @segment snp5.2:8.2 -->

Manasānāvilo siyā;

<!-- @segment snp5.2:8.3 -->

Kusalo sabbadhammānaṁ,

<!-- @segment snp5.2:8.4 -->

Sato bhikkhu paribbaje”ti.

<!-- @segment snp5.2:8.5 -->

Ajitamāṇavapucchā paṭhamā.
