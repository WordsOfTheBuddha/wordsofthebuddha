<!-- @segment snp3.1:1.1 -->

“Pabbajjaṁ kittayissāmi,

<!-- @segment snp3.1:1.2 -->

yathā pabbaji cakkhumā;

<!-- @segment snp3.1:1.3 -->

Yathā vīmaṁsamāno so,

<!-- @segment snp3.1:1.4 -->

pabbajjaṁ samarocayi.

<!-- @segment snp3.1:2.1 -->

Sambādhoyaṁ gharāvāso,

<!-- @segment snp3.1:2.2 -->

rajassāyatanaṁ iti;

<!-- @segment snp3.1:2.3 -->

Abbhokāsova pabbajjā,

<!-- @segment snp3.1:2.4 -->

iti disvāna pabbaji.

<!-- @segment snp3.1:3.1 -->

Pabbajitvāna kāyena,

<!-- @segment snp3.1:3.2 -->

pāpakammaṁ vivajjayi;

<!-- @segment snp3.1:3.3 -->

Vacīduccaritaṁ hitvā,

<!-- @segment snp3.1:3.4 -->

ājīvaṁ parisodhayi.

<!-- @segment snp3.1:4.1 -->

Agamā rājagahaṁ buddho,

<!-- @segment snp3.1:4.2 -->

magadhānaṁ giribbajaṁ;

<!-- @segment snp3.1:4.3 -->

Piṇḍāya abhihāresi,

<!-- @segment snp3.1:4.4 -->

ākiṇṇavaralakkhaṇo”.

<!-- @segment snp3.1:5.1 -->

Tamaddasā bimbisāro,

<!-- @segment snp3.1:5.2 -->

pāsādasmiṁ patiṭṭhito;

<!-- @segment snp3.1:5.3 -->

Disvā lakkhaṇasampannaṁ,

<!-- @segment snp3.1:5.4 -->

imamatthaṁ abhāsatha.

<!-- @segment snp3.1:6.1 -->

“Imaṁ bhonto nisāmetha,

<!-- @segment snp3.1:6.2 -->

abhirūpo brahā suci;

<!-- @segment snp3.1:6.3 -->

Caraṇena ca sampanno,

<!-- @segment snp3.1:6.4 -->

yugamattañca pekkhati.

<!-- @segment snp3.1:7.1 -->

Okkhittacakkhu satimā,

<!-- @segment snp3.1:7.2 -->

nāyaṁ nīcakulāmiva;

<!-- @segment snp3.1:7.3 -->

Rājadūtābhidhāvantu,

<!-- @segment snp3.1:7.4 -->

kuhiṁ bhikkhu gamissati”.

<!-- @segment snp3.1:8.1 -->

Te pesitā rājadūtā,

<!-- @segment snp3.1:8.2 -->

piṭṭhito anubandhisuṁ;

<!-- @segment snp3.1:8.3 -->

Kuhiṁ gamissati bhikkhu,

<!-- @segment snp3.1:8.4 -->

kattha vāso bhavissati.

<!-- @segment snp3.1:9.1 -->

Sapadānaṁ caramāno,

<!-- @segment snp3.1:9.2 -->

guttadvāro susaṁvuto;

<!-- @segment snp3.1:9.3 -->

Khippaṁ pattaṁ apūresi,

<!-- @segment snp3.1:9.4 -->

sampajāno paṭissato.

<!-- @segment snp3.1:10.1 -->

Piṇḍacāraṁ caritvāna,

<!-- @segment snp3.1:10.2 -->

Nikkhamma nagarā muni;

<!-- @segment snp3.1:10.3 -->

Paṇḍavaṁ abhihāresi,

<!-- @segment snp3.1:10.4 -->

Ettha vāso bhavissati.

<!-- @segment snp3.1:11.1 -->

Disvāna vāsūpagataṁ,

<!-- @segment snp3.1:11.2 -->

Tayo dūtā upāvisuṁ;

<!-- @segment snp3.1:11.3 -->

Tesu ekova āgantvā,

<!-- @segment snp3.1:11.4 -->

Rājino paṭivedayi.

<!-- @segment snp3.1:12.1 -->

“Esa bhikkhu mahārāja,

<!-- @segment snp3.1:12.2 -->

Paṇḍavassa puratthato;

<!-- @segment snp3.1:12.3 -->

Nisinno byagghusabhova,

<!-- @segment snp3.1:12.4 -->

Sīhova girigabbhare”.

<!-- @segment snp3.1:13.1 -->

Sutvāna dūtavacanaṁ,

<!-- @segment snp3.1:13.2 -->

Bhaddayānena khattiyo;

<!-- @segment snp3.1:13.3 -->

Taramānarūpo niyyāsi,

<!-- @segment snp3.1:13.4 -->

Yena paṇḍavapabbato.

<!-- @segment snp3.1:14.1 -->

Sa yānabhūmiṁ yāyitvā,

<!-- @segment snp3.1:14.2 -->

Yānā oruyha khattiyo;

<!-- @segment snp3.1:14.3 -->

Pattiko upasaṅkamma,

<!-- @segment snp3.1:14.4 -->

Āsajja naṁ upāvisi.

<!-- @segment snp3.1:15.1 -->

Nisajja rājā sammodi,

<!-- @segment snp3.1:15.2 -->

Kathaṁ sāraṇīyaṁ tato;

<!-- @segment snp3.1:15.3 -->

Kathaṁ so vītisāretvā,

<!-- @segment snp3.1:15.4 -->

Imamatthaṁ abhāsatha.

<!-- @segment snp3.1:16.1 -->

“Yuvā ca daharo cāsi,

<!-- @segment snp3.1:16.2 -->

Paṭhamuppattiko susu;

<!-- @segment snp3.1:16.3 -->

Vaṇṇārohena sampanno,

<!-- @segment snp3.1:16.4 -->

Jātimā viya khattiyo.

<!-- @segment snp3.1:17.1 -->

Sobhayanto anīkaggaṁ,

<!-- @segment snp3.1:17.2 -->

Nāgasaṅghapurakkhato;

<!-- @segment snp3.1:17.3 -->

Dadāmi bhoge bhuñjassu,

<!-- @segment snp3.1:17.4 -->

Jātiṁ akkhāhi pucchito”.

<!-- @segment snp3.1:18.1 -->

“Ujuṁ janapado rāja,

<!-- @segment snp3.1:18.2 -->

Himavantassa passato;

<!-- @segment snp3.1:18.3 -->

Dhanaviriyena sampanno,

<!-- @segment snp3.1:18.4 -->

Kosalesu niketino.

<!-- @segment snp3.1:19.1 -->

Ādiccā nāma gottena,

<!-- @segment snp3.1:19.2 -->

Sākiyā nāma jātiyā;

<!-- @segment snp3.1:19.3 -->

Tamhā kulā pabbajitomhi,

<!-- @segment snp3.1:19.4 -->

Na kāme abhipatthayaṁ.

<!-- @segment snp3.1:20.1 -->

Kāmesvādīnavaṁ disvā,

<!-- @segment snp3.1:20.2 -->

Nekkhammaṁ daṭṭhu khemato;

<!-- @segment snp3.1:20.3 -->

Padhānāya gamissāmi,

<!-- @segment snp3.1:20.4 -->

Ettha me rañjatī mano”ti.

<!-- @segment snp3.1:20.5 -->

Pabbajjāsuttaṁ paṭhamaṁ.
