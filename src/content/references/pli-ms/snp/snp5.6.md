<!-- @segment snp5.6:1.1 -->

“Pucchāmi taṁ bhagavā brūhi metaṁ,

<!-- @segment snp5.6:1.2 -->

(iccāyasmā dhotako)

<!-- @segment snp5.6:1.3 -->

Vācābhikaṅkhāmi mahesi tuyhaṁ;

<!-- @segment snp5.6:1.4 -->

Tava sutvāna nigghosaṁ,

<!-- @segment snp5.6:1.5 -->

Sikkhe nibbānamattano”.

<!-- @segment snp5.6:2.1 -->

“Tenahātappaṁ karohi,

<!-- @segment snp5.6:2.2 -->

(dhotakāti bhagavā)

<!-- @segment snp5.6:2.3 -->

Idheva nipako sato;

<!-- @segment snp5.6:2.4 -->

Ito sutvāna nigghosaṁ,

<!-- @segment snp5.6:2.5 -->

Sikkhe nibbānamattano”.

<!-- @segment snp5.6:3.1 -->

“Passāmahaṁ devamanussaloke,

<!-- @segment snp5.6:3.2 -->

Akiñcanaṁ brāhmaṇamiriyamānaṁ;

<!-- @segment snp5.6:3.3 -->

Taṁ taṁ namassāmi samantacakkhu,

<!-- @segment snp5.6:3.4 -->

Pamuñca maṁ sakka kathaṅkathāhi”.

<!-- @segment snp5.6:4.1 -->

“Nāhaṁ sahissāmi pamocanāya,

<!-- @segment snp5.6:4.2 -->

Kathaṅkathiṁ dhotaka kañci loke;

<!-- @segment snp5.6:4.3 -->

Dhammañca seṭṭhaṁ abhijānamāno,

<!-- @segment snp5.6:4.4 -->

Evaṁ tuvaṁ oghamimaṁ taresi”.

<!-- @segment snp5.6:5.1 -->

“Anusāsa brahme karuṇāyamāno,

<!-- @segment snp5.6:5.2 -->

Vivekadhammaṁ yamahaṁ vijaññaṁ;

<!-- @segment snp5.6:5.3 -->

Yathāhaṁ ākāsova abyāpajjamāno,

<!-- @segment snp5.6:5.4 -->

Idheva santo asito careyyaṁ”.

<!-- @segment snp5.6:6.1 -->

“Kittayissāmi te santiṁ,

<!-- @segment snp5.6:6.2 -->

(dhotakāti bhagavā)

<!-- @segment snp5.6:6.3 -->

Diṭṭhe dhamme anītihaṁ;

<!-- @segment snp5.6:6.4 -->

Yaṁ viditvā sato caraṁ,

<!-- @segment snp5.6:6.5 -->

Tare loke visattikaṁ”.

<!-- @segment snp5.6:7.1 -->

“Tañcāhaṁ abhinandāmi,

<!-- @segment snp5.6:7.2 -->

mahesi santimuttamaṁ;

<!-- @segment snp5.6:7.3 -->

Yaṁ viditvā sato caraṁ,

<!-- @segment snp5.6:7.4 -->

tare loke visattikaṁ”.

<!-- @segment snp5.6:8.1 -->

“Yaṁ kiñci sampajānāsi,

<!-- @segment snp5.6:8.2 -->

(dhotakāti bhagavā)

<!-- @segment snp5.6:8.3 -->

Uddhaṁ adho tiriyañcāpi majjhe;

<!-- @segment snp5.6:8.4 -->

Etaṁ viditvā saṅgoti loke,

<!-- @segment snp5.6:8.5 -->

Bhavābhavāya mākāsi taṇhan”ti.

<!-- @segment snp5.6:8.6 -->

Dhotakamāṇavapucchā pañcamī.
