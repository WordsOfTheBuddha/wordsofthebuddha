<!-- @segment snp5.16:1.1 -->

“Dvāhaṁ sakkaṁ apucchissaṁ,

<!-- @segment snp5.16:1.2 -->

(iccāyasmā mogharājā)

<!-- @segment snp5.16:1.3 -->

Na me byākāsi cakkhumā;

<!-- @segment snp5.16:1.4 -->

Yāvatatiyañca devīsi,

<!-- @segment snp5.16:1.5 -->

Byākarotīti me sutaṁ.

<!-- @segment snp5.16:2.1 -->

Ayaṁ loko paro loko,

<!-- @segment snp5.16:2.2 -->

brahmaloko sadevako;

<!-- @segment snp5.16:2.3 -->

Diṭṭhiṁ te nābhijānāti,

<!-- @segment snp5.16:2.4 -->

gotamassa yasassino.

<!-- @segment snp5.16:3.1 -->

Evaṁ abhikkantadassāviṁ,

<!-- @segment snp5.16:3.2 -->

atthi pañhena āgamaṁ;

<!-- @segment snp5.16:3.3 -->

Kathaṁ lokaṁ avekkhantaṁ,

<!-- @segment snp5.16:3.4 -->

maccurājā na passati”.

<!-- @segment snp5.16:4.1 -->

“Suññato lokaṁ avekkhassu,

<!-- @segment snp5.16:4.2 -->

Mogharāja sadā sato;

<!-- @segment snp5.16:4.3 -->

Attānudiṭṭhiṁ ūhacca,

<!-- @segment snp5.16:4.4 -->

Evaṁ maccutaro siyā;

<!-- @segment snp5.16:4.5 -->

Evaṁ lokaṁ avekkhantaṁ,

<!-- @segment snp5.16:4.6 -->

Maccurājā na passatī”ti.

<!-- @segment snp5.16:4.7 -->

Mogharājamāṇavapucchā pannarasamā.
