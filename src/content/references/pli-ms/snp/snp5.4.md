<!-- @segment snp5.4:1.1 -->

“Anejaṁ mūladassāviṁ,

<!-- @segment snp5.4:1.2 -->

(iccāyasmā puṇṇako)

<!-- @segment snp5.4:1.3 -->

Atthi pañhena āgamaṁ;

<!-- @segment snp5.4:1.4 -->

Kiṁ nissitā isayo manujā,

<!-- @segment snp5.4:1.5 -->

Khattiyā brāhmaṇā devatānaṁ;

<!-- @segment snp5.4:1.6 -->

Yaññamakappayiṁsu puthūdha loke,

<!-- @segment snp5.4:1.7 -->

Pucchāmi taṁ bhagavā brūhi me taṁ”.

<!-- @segment snp5.4:2.1 -->

“Ye kecime isayo manujā,

<!-- @segment snp5.4:2.2 -->

(puṇṇakāti bhagavā)

<!-- @segment snp5.4:2.3 -->

Khattiyā brāhmaṇā devatānaṁ;

<!-- @segment snp5.4:2.4 -->

Yaññamakappayiṁsu puthūdha loke,

<!-- @segment snp5.4:2.5 -->

Āsīsamānā puṇṇaka itthattaṁ;

<!-- @segment snp5.4:2.6 -->

Jaraṁ sitā yaññamakappayiṁsu”.

<!-- @segment snp5.4:3.1 -->

“Ye kecime isayo manujā,

<!-- @segment snp5.4:3.2 -->

(iccāyasmā puṇṇako)

<!-- @segment snp5.4:3.3 -->

Khattiyā brāhmaṇā devatānaṁ;

<!-- @segment snp5.4:3.4 -->

Yaññamakappayiṁsu puthūdha loke,

<!-- @segment snp5.4:3.5 -->

Kaccissu te bhagavā yaññapathe appamattā;

<!-- @segment snp5.4:3.6 -->

Atāruṁ jātiñca jarañca mārisa,

<!-- @segment snp5.4:3.7 -->

Pucchāmi taṁ bhagavā brūhi me taṁ”.

<!-- @segment snp5.4:4.1 -->

“Āsīsanti thomayanti abhijappanti juhanti,

<!-- @segment snp5.4:4.2 -->

(puṇṇakāti bhagavā)

<!-- @segment snp5.4:4.3 -->

Kāmābhijappanti paṭicca lābhaṁ;

<!-- @segment snp5.4:4.4 -->

Te yājayogā bhavarāgarattā,

<!-- @segment snp5.4:4.5 -->

Nātariṁsu jātijaranti brūmi”.

<!-- @segment snp5.4:5.1 -->

“Te ce nātariṁsu yājayogā,

<!-- @segment snp5.4:5.2 -->

(iccāyasmā puṇṇako)

<!-- @segment snp5.4:5.3 -->

Yaññehi jātiñca jarañca mārisa;

<!-- @segment snp5.4:5.4 -->

Atha ko carahi devamanussaloke,

<!-- @segment snp5.4:5.5 -->

Atāri jātiñca jarañca mārisa;

<!-- @segment snp5.4:5.6 -->

Pucchāmi taṁ bhagavā brūhi me taṁ”.

<!-- @segment snp5.4:6.1 -->

“Saṅkhāya lokasmi paroparāni,

<!-- @segment snp5.4:6.2 -->

(puṇṇakāti bhagavā)

<!-- @segment snp5.4:6.3 -->

Yassiñjitaṁ natthi kuhiñci loke;

<!-- @segment snp5.4:6.4 -->

Santo vidhūmo anīgho nirāso,

<!-- @segment snp5.4:6.5 -->

Atāri so jātijaranti brūmī”ti.

<!-- @segment snp5.4:6.6 -->

Puṇṇakamāṇavapucchā tatiyā.
