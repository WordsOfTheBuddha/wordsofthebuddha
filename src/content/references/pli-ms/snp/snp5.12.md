<!-- @segment snp5.12:1.1 -->

“Sutvānahaṁ vīramakāmakāmiṁ,

<!-- @segment snp5.12:1.2 -->

(iccāyasmā jatukaṇṇi)

<!-- @segment snp5.12:1.3 -->

Oghātigaṁ puṭṭhumakāmamāgamaṁ;

<!-- @segment snp5.12:1.4 -->

Santipadaṁ brūhi sahajanetta,

<!-- @segment snp5.12:1.5 -->

Yathātacchaṁ bhagavā brūhi metaṁ.

<!-- @segment snp5.12:2.1 -->

Bhagavā hi kāme abhibhuyya iriyati,

<!-- @segment snp5.12:2.2 -->

Ādiccova pathaviṁ tejī tejasā;

<!-- @segment snp5.12:2.3 -->

Parittapaññassa me bhūripañña,

<!-- @segment snp5.12:2.4 -->

Ācikkha dhammaṁ yamahaṁ vijaññaṁ;

<!-- @segment snp5.12:2.5 -->

Jātijarāya idha vippahānaṁ”.

<!-- @segment snp5.12:3.1 -->

“Kāmesu vinaya gedhaṁ,

<!-- @segment snp5.12:3.2 -->

(jatukaṇṇīti bhagavā)

<!-- @segment snp5.12:3.3 -->

Nekkhammaṁ daṭṭhu khemato;

<!-- @segment snp5.12:3.4 -->

Uggahītaṁ nirattaṁ vā,

<!-- @segment snp5.12:3.5 -->

Mā te vijjittha kiñcanaṁ.

<!-- @segment snp5.12:4.1 -->

Yaṁ pubbe taṁ visosehi,

<!-- @segment snp5.12:4.2 -->

pacchā te māhu kiñcanaṁ;

<!-- @segment snp5.12:4.3 -->

Majjhe ce no gahessasi,

<!-- @segment snp5.12:4.4 -->

upasanto carissasi.

<!-- @segment snp5.12:5.1 -->

Sabbaso nāmarūpasmiṁ,

<!-- @segment snp5.12:5.2 -->

Vītagedhassa brāhmaṇa;

<!-- @segment snp5.12:5.3 -->

Āsavāssa na vijjanti,

<!-- @segment snp5.12:5.4 -->

Yehi maccuvasaṁ vaje”ti.

<!-- @segment snp5.12:5.5 -->

Jatukaṇṇimāṇavapucchā ekādasamā.
