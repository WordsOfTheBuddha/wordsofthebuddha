<!-- @segment snp2.6:1.1 -->

Dhammacariyaṁ brahmacariyaṁ,

<!-- @segment snp2.6:1.2 -->

Etadāhu vasuttamaṁ;

<!-- @segment snp2.6:1.3 -->

Pabbajitopi ce hoti,

<!-- @segment snp2.6:1.4 -->

Agārā anagāriyaṁ.

<!-- @segment snp2.6:2.1 -->

So ce mukharajātiko,

<!-- @segment snp2.6:2.2 -->

vihesābhirato mago;

<!-- @segment snp2.6:2.3 -->

Jīvitaṁ tassa pāpiyo,

<!-- @segment snp2.6:2.4 -->

rajaṁ vaḍḍheti attano.

<!-- @segment snp2.6:3.1 -->

Kalahābhirato bhikkhu,

<!-- @segment snp2.6:3.2 -->

Mohadhammena āvuto;

<!-- @segment snp2.6:3.3 -->

Akkhātampi na jānāti,

<!-- @segment snp2.6:3.4 -->

Dhammaṁ buddhena desitaṁ.

<!-- @segment snp2.6:4.1 -->

Vihesaṁ bhāvitattānaṁ,

<!-- @segment snp2.6:4.2 -->

avijjāya purakkhato;

<!-- @segment snp2.6:4.3 -->

Saṅkilesaṁ na jānāti,

<!-- @segment snp2.6:4.4 -->

maggaṁ nirayagāminaṁ.

<!-- @segment snp2.6:5.1 -->

Vinipātaṁ samāpanno,

<!-- @segment snp2.6:5.2 -->

gabbhā gabbhaṁ tamā tamaṁ;

<!-- @segment snp2.6:5.3 -->

Sa ve tādisako bhikkhu,

<!-- @segment snp2.6:5.4 -->

pecca dukkhaṁ nigacchati.

<!-- @segment snp2.6:6.1 -->

Gūthakūpo yathā assa,

<!-- @segment snp2.6:6.2 -->

sampuṇṇo gaṇavassiko;

<!-- @segment snp2.6:6.3 -->

Yo ca evarūpo assa,

<!-- @segment snp2.6:6.4 -->

dubbisodho hi sāṅgaṇo.

<!-- @segment snp2.6:7.1 -->

Yaṁ evarūpaṁ jānātha,

<!-- @segment snp2.6:7.2 -->

bhikkhavo gehanissitaṁ;

<!-- @segment snp2.6:7.3 -->

Pāpicchaṁ pāpasaṅkappaṁ,

<!-- @segment snp2.6:7.4 -->

pāpaācāragocaraṁ.

<!-- @segment snp2.6:8.1 -->

Sabbe samaggā hutvāna,

<!-- @segment snp2.6:8.2 -->

abhinibbajjiyātha naṁ;

<!-- @segment snp2.6:8.3 -->

Kāraṇḍavaṁ niddhamatha,

<!-- @segment snp2.6:8.4 -->

kasambuṁ apakassatha.

<!-- @segment snp2.6:9.1 -->

Tato palāpe vāhetha,

<!-- @segment snp2.6:9.2 -->

Assamaṇe samaṇamānine;

<!-- @segment snp2.6:9.3 -->

Niddhamitvāna pāpicche,

<!-- @segment snp2.6:9.4 -->

Pāpaācāragocare.

<!-- @segment snp2.6:10.1 -->

Suddhā suddhehi saṁvāsaṁ,

<!-- @segment snp2.6:10.2 -->

Kappayavho patissatā;

<!-- @segment snp2.6:10.3 -->

Tato samaggā nipakā,

<!-- @segment snp2.6:10.4 -->

Dukkhassantaṁ karissathāti.

<!-- @segment snp2.6:10.5 -->

Dhammacariyasuttaṁ chaṭṭhaṁ.
