<!-- @segment snp1.8:1.1 -->

“Karaṇīyamatthakusalena,

<!-- @segment snp1.8:1.2 -->

Yanta santaṁ padaṁ abhisamecca;

<!-- @segment snp1.8:1.3 -->

Sakko ujū ca suhujū ca,

<!-- @segment snp1.8:1.4 -->

Sūvaco cassa mudu anatimānī.

<!-- @segment snp1.8:2.1 -->

Santussako ca subharo ca,

<!-- @segment snp1.8:2.2 -->

Appakicco ca sallahukavutti;

<!-- @segment snp1.8:2.3 -->

Santindriyo ca nipako ca,

<!-- @segment snp1.8:2.4 -->

Appagabbho kulesvananugiddho.

<!-- @segment snp1.8:3.1 -->

Na ca khuddamācare kiñci,

<!-- @segment snp1.8:3.2 -->

Yena viññū pare upavadeyyuṁ;

<!-- @segment snp1.8:3.3 -->

Sukhino va khemino hontu,

<!-- @segment snp1.8:3.4 -->

Sabbasattā bhavantu sukhitattā.

<!-- @segment snp1.8:4.1 -->

Ye keci pāṇabhūtatthi,

<!-- @segment snp1.8:4.2 -->

Tasā vā thāvarā vanavasesā;

<!-- @segment snp1.8:4.3 -->

Dīghā vā ye va mahantā,

<!-- @segment snp1.8:4.4 -->

Majjhimā rassakā aṇukathūlā.

<!-- @segment snp1.8:5.1 -->

Diṭṭhā vā ye va adiṭṭhā,

<!-- @segment snp1.8:5.2 -->

Ye va dūre vasanti avidūre;

<!-- @segment snp1.8:5.3 -->

Bhūtā va sambhavesī va,

<!-- @segment snp1.8:5.4 -->

Sabbasattā bhavantu sukhitattā.

<!-- @segment snp1.8:6.1 -->

Na paro paraṁ nikubbetha,

<!-- @segment snp1.8:6.2 -->

Nātimaññetha katthaci na kañci;

<!-- @segment snp1.8:6.3 -->

Byārosanā paṭighasañña,

<!-- @segment snp1.8:6.4 -->

Nāññamaññassa dukkhamiccheyya.

<!-- @segment snp1.8:7.1 -->

Mātā yathā niyaṁ puttam

<!-- @segment snp1.8:7.2 -->

Āyusā ekaputtamanurakkhe;

<!-- @segment snp1.8:7.3 -->

Evampi sabbabhūtesu,

<!-- @segment snp1.8:7.4 -->

Mānasaṁ bhāvaye aparimāṇaṁ.

<!-- @segment snp1.8:8.1 -->

Mettañca sabbalokasmi,

<!-- @segment snp1.8:8.2 -->

Mānasaṁ bhāvaye aparimāṇaṁ;

<!-- @segment snp1.8:8.3 -->

Uddhaṁ adho ca tiriyañca,

<!-- @segment snp1.8:8.4 -->

Asambādhaṁ averamasapattaṁ.

<!-- @segment snp1.8:9.1 -->

Tiṭṭhaṁ caraṁ nisinno va,

<!-- @segment snp1.8:9.2 -->

Sayāno yāvatāssa vitamiddho;

<!-- @segment snp1.8:9.3 -->

Etaṁ satiṁ adhiṭṭheyya,

<!-- @segment snp1.8:9.4 -->

Brahmametaṁ vihāramidhamāhu.

<!-- @segment snp1.8:10.1 -->

Diṭṭhiñca anupaggamma,

<!-- @segment snp1.8:10.2 -->

Sīlavā dassanena sampanno;

<!-- @segment snp1.8:10.3 -->

Kāmesu vinaya gedhaṁ,

<!-- @segment snp1.8:10.4 -->

Na hi jātuggabbhaseyya punaretī”ti.

<!-- @segment snp1.8:10.5 -->

Mettasuttaṁ aṭṭhamaṁ.
