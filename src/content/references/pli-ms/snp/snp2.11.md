<!-- @segment snp2.11:1.1 -->

“Kacci abhiṇhasaṁvāsā,

<!-- @segment snp2.11:1.2 -->

nāvajānāsi paṇḍitaṁ;

<!-- @segment snp2.11:1.3 -->

Ukkādhāro manussānaṁ,

<!-- @segment snp2.11:1.4 -->

kacci apacito tayā”.

<!-- @segment snp2.11:2.1 -->

“Nāhaṁ abhiṇhasaṁvāsā,

<!-- @segment snp2.11:2.2 -->

avajānāmi paṇḍitaṁ;

<!-- @segment snp2.11:2.3 -->

Ukkādhāro manussānaṁ,

<!-- @segment snp2.11:2.4 -->

niccaṁ apacito mayā”.

<!-- @segment snp2.11:3.1 -->

“Pañca kāmaguṇe hitvā,

<!-- @segment snp2.11:3.2 -->

piyarūpe manorame;

<!-- @segment snp2.11:3.3 -->

Saddhāya gharā nikkhamma,

<!-- @segment snp2.11:3.4 -->

dukkhassantakaro bhava.

<!-- @segment snp2.11:4.1 -->

Mitte bhajassu kalyāṇe,

<!-- @segment snp2.11:4.2 -->

pantañca sayanāsanaṁ;

<!-- @segment snp2.11:4.3 -->

Vivittaṁ appanigghosaṁ,

<!-- @segment snp2.11:4.4 -->

mattaññū hohi bhojane.

<!-- @segment snp2.11:5.1 -->

Cīvare piṇḍapāte ca,

<!-- @segment snp2.11:5.2 -->

paccaye sayanāsane;

<!-- @segment snp2.11:5.3 -->

Etesu taṇhaṁ mākāsi,

<!-- @segment snp2.11:5.4 -->

mā lokaṁ punarāgami.

<!-- @segment snp2.11:6.1 -->

Saṁvuto pātimokkhasmiṁ,

<!-- @segment snp2.11:6.2 -->

indriyesu ca pañcasu;

<!-- @segment snp2.11:6.3 -->

Sati kāyagatā tyatthu,

<!-- @segment snp2.11:6.4 -->

nibbidābahulo bhava.

<!-- @segment snp2.11:7.1 -->

Nimittaṁ parivajjehi,

<!-- @segment snp2.11:7.2 -->

subhaṁ rāgūpasañhitaṁ;

<!-- @segment snp2.11:7.3 -->

Asubhāya cittaṁ bhāvehi,

<!-- @segment snp2.11:7.4 -->

ekaggaṁ susamāhitaṁ.

<!-- @segment snp2.11:8.1 -->

Animittañca bhāvehi,

<!-- @segment snp2.11:8.2 -->

Mānānusayamujjaha;

<!-- @segment snp2.11:8.3 -->

Tato mānābhisamayā,

<!-- @segment snp2.11:8.4 -->

Upasanto carissatī”ti.

<!-- @segment snp2.11:9.1 -->

Itthaṁ sudaṁ bhagavā āyasmantaṁ rāhulaṁ imāhi gāthāhi abhiṇhaṁ ovadatīti.

<!-- @segment snp2.11:9.2 -->

Rāhulasuttaṁ ekādasamaṁ.
