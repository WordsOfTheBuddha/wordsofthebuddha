<!-- @segment snp1.11:1.1 -->

Caraṁ vā yadi vā tiṭṭhaṁ,

<!-- @segment snp1.11:1.2 -->

nisinno uda vā sayaṁ;

<!-- @segment snp1.11:1.3 -->

Samiñjeti pasāreti,

<!-- @segment snp1.11:1.4 -->

esā kāyassa iñjanā.

<!-- @segment snp1.11:2.1 -->

Aṭṭhinahārusaṁyutto,

<!-- @segment snp1.11:2.2 -->

tacamaṁsāvalepano;

<!-- @segment snp1.11:2.3 -->

Chaviyā kāyo paṭicchanno,

<!-- @segment snp1.11:2.4 -->

yathābhūtaṁ na dissati.

<!-- @segment snp1.11:3.1 -->

Antapūro udarapūro,

<!-- @segment snp1.11:3.2 -->

yakanapeḷassa vatthino;

<!-- @segment snp1.11:3.3 -->

Hadayassa papphāsassa,

<!-- @segment snp1.11:3.4 -->

vakkassa pihakassa ca.

<!-- @segment snp1.11:4.1 -->

Siṅghāṇikāya kheḷassa,

<!-- @segment snp1.11:4.2 -->

sedassa ca medassa ca;

<!-- @segment snp1.11:4.3 -->

Lohitassa lasikāya,

<!-- @segment snp1.11:4.4 -->

pittassa ca vasāya ca.

<!-- @segment snp1.11:5.1 -->

Athassa navahi sotehi,

<!-- @segment snp1.11:5.2 -->

asucī savati sabbadā;

<!-- @segment snp1.11:5.3 -->

Akkhimhā akkhigūthako,

<!-- @segment snp1.11:5.4 -->

kaṇṇamhā kaṇṇagūthako.

<!-- @segment snp1.11:6.1 -->

Siṅghāṇikā ca nāsato,

<!-- @segment snp1.11:6.2 -->

mukhena vamatekadā;

<!-- @segment snp1.11:6.3 -->

Pittaṁ semhañca vamati,

<!-- @segment snp1.11:6.4 -->

kāyamhā sedajallikā.

<!-- @segment snp1.11:7.1 -->

Athassa susiraṁ sīsaṁ,

<!-- @segment snp1.11:7.2 -->

matthaluṅgassa pūritaṁ;

<!-- @segment snp1.11:7.3 -->

Subhato naṁ maññati bālo,

<!-- @segment snp1.11:7.4 -->

avijjāya purakkhato.

<!-- @segment snp1.11:8.1 -->

Yadā ca so mato seti,

<!-- @segment snp1.11:8.2 -->

Uddhumāto vinīlako;

<!-- @segment snp1.11:8.3 -->

Apaviddho susānasmiṁ,

<!-- @segment snp1.11:8.4 -->

Anapekkhā honti ñātayo.

<!-- @segment snp1.11:9.1 -->

Khādanti naṁ suvānā ca,

<!-- @segment snp1.11:9.2 -->

siṅgālā ca vakā kimī;

<!-- @segment snp1.11:9.3 -->

Kākā gijjhā ca khādanti,

<!-- @segment snp1.11:9.4 -->

ye caññe santi pāṇino.

<!-- @segment snp1.11:10.1 -->

Sutvāna buddhavacanaṁ,

<!-- @segment snp1.11:10.2 -->

Bhikkhu paññāṇavā idha;

<!-- @segment snp1.11:10.3 -->

So kho naṁ parijānāti,

<!-- @segment snp1.11:10.4 -->

Yathābhūtañhi passati.

<!-- @segment snp1.11:11.1 -->

Yathā idaṁ tathā etaṁ,

<!-- @segment snp1.11:11.2 -->

Yathā etaṁ tathā idaṁ;

<!-- @segment snp1.11:11.3 -->

Ajjhattañca bahiddhā ca,

<!-- @segment snp1.11:11.4 -->

Kāye chandaṁ virājaye.

<!-- @segment snp1.11:12.1 -->

Chandarāgaviratto so,

<!-- @segment snp1.11:12.2 -->

Bhikkhu paññāṇavā idha;

<!-- @segment snp1.11:12.3 -->

Ajjhagā amataṁ santiṁ,

<!-- @segment snp1.11:12.4 -->

Nibbānaṁ padamaccutaṁ.

<!-- @segment snp1.11:13.1 -->

Dvipādakoyaṁ asuci,

<!-- @segment snp1.11:13.2 -->

Duggandho parihīrati;

<!-- @segment snp1.11:13.3 -->

Nānākuṇapaparipūro,

<!-- @segment snp1.11:13.4 -->

Vissavanto tato tato.

<!-- @segment snp1.11:14.1 -->

Etādisena kāyena,

<!-- @segment snp1.11:14.2 -->

Yo maññe uṇṇametave;

<!-- @segment snp1.11:14.3 -->

Paraṁ vā avajāneyya,

<!-- @segment snp1.11:14.4 -->

Kimaññatra adassanāti.

<!-- @segment snp1.11:14.5 -->

Vijayasuttaṁ ekādasamaṁ.
