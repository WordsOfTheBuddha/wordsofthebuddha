<!-- @segment snp3.2:1.1 -->

“Taṁ maṁ padhānapahitattaṁ,

<!-- @segment snp3.2:1.2 -->

nadiṁ nerañjaraṁ pati;

<!-- @segment snp3.2:1.3 -->

Viparakkamma jhāyantaṁ,

<!-- @segment snp3.2:1.4 -->

yogakkhemassa pattiyā.

<!-- @segment snp3.2:2.1 -->

Namucī karuṇaṁ vācaṁ,

<!-- @segment snp3.2:2.2 -->

bhāsamāno upāgami;

<!-- @segment snp3.2:2.3 -->

‘Kiso tvamasi dubbaṇṇo,

<!-- @segment snp3.2:2.4 -->

santike maraṇaṁ tava.

<!-- @segment snp3.2:3.1 -->

Sahassabhāgo maraṇassa,

<!-- @segment snp3.2:3.2 -->

ekaṁso tava jīvitaṁ;

<!-- @segment snp3.2:3.3 -->

Jīva bho jīvitaṁ seyyo,

<!-- @segment snp3.2:3.4 -->

jīvaṁ puññāni kāhasi.

<!-- @segment snp3.2:4.1 -->

Carato ca te brahmacariyaṁ,

<!-- @segment snp3.2:4.2 -->

Aggihuttañca jūhato;

<!-- @segment snp3.2:4.3 -->

Pahūtaṁ cīyate puññaṁ,

<!-- @segment snp3.2:4.4 -->

Kiṁ padhānena kāhasi.

<!-- @segment snp3.2:5.1 -->

Duggo maggo padhānāya,

<!-- @segment snp3.2:5.2 -->

dukkaro durabhisambhavo’”;

<!-- @segment snp3.2:5.3 -->

Imā gāthā bhaṇaṁ māro,

<!-- @segment snp3.2:5.4 -->

aṭṭhā buddhassa santike.

<!-- @segment snp3.2:6.1 -->

Taṁ tathāvādinaṁ māraṁ,

<!-- @segment snp3.2:6.2 -->

bhagavā etadabravi;

<!-- @segment snp3.2:6.3 -->

“Pamattabandhu pāpima,

<!-- @segment snp3.2:6.4 -->

yenatthena idhāgato.

<!-- @segment snp3.2:7.1 -->

Aṇumattopi puññena,

<!-- @segment snp3.2:7.2 -->

Attho mayhaṁ na vijjati;

<!-- @segment snp3.2:7.3 -->

Yesañca attho puññena,

<!-- @segment snp3.2:7.4 -->

Te māro vattumarahati.

<!-- @segment snp3.2:8.1 -->

Atthi saddhā tathā viriyaṁ,

<!-- @segment snp3.2:8.2 -->

paññā ca mama vijjati;

<!-- @segment snp3.2:8.3 -->

Evaṁ maṁ pahitattampi,

<!-- @segment snp3.2:8.4 -->

kiṁ jīvamanupucchasi.

<!-- @segment snp3.2:9.1 -->

Nadīnamapi sotāni,

<!-- @segment snp3.2:9.2 -->

ayaṁ vāto visosaye;

<!-- @segment snp3.2:9.3 -->

Kiñca me pahitattassa,

<!-- @segment snp3.2:9.4 -->

lohitaṁ nupasussaye.

<!-- @segment snp3.2:10.1 -->

Lohite sussamānamhi,

<!-- @segment snp3.2:10.2 -->

Pittaṁ semhañca sussati;

<!-- @segment snp3.2:10.3 -->

Maṁsesu khīyamānesu,

<!-- @segment snp3.2:10.4 -->

Bhiyyo cittaṁ pasīdati;

<!-- @segment snp3.2:10.5 -->

Bhiyyo sati ca paññā ca,

<!-- @segment snp3.2:10.6 -->

Samādhi mama tiṭṭhati.

<!-- @segment snp3.2:11.1 -->

Tassa mevaṁ viharato,

<!-- @segment snp3.2:11.2 -->

Pattassuttamavedanaṁ;

<!-- @segment snp3.2:11.3 -->

Kāmesu nāpekkhate cittaṁ,

<!-- @segment snp3.2:11.4 -->

Passa sattassa suddhataṁ.

<!-- @segment snp3.2:12.1 -->

Kāmā te paṭhamā senā,

<!-- @segment snp3.2:12.2 -->

Dutiyā arati vuccati;

<!-- @segment snp3.2:12.3 -->

Tatiyā khuppipāsā te,

<!-- @segment snp3.2:12.4 -->

Catutthī taṇhā pavuccati.

<!-- @segment snp3.2:13.1 -->

Pañcamaṁ thinamiddhaṁ te,

<!-- @segment snp3.2:13.2 -->

Chaṭṭhā bhīrū pavuccati;

<!-- @segment snp3.2:13.3 -->

Sattamī vicikicchā te,

<!-- @segment snp3.2:13.4 -->

Makkho thambho te aṭṭhamo.

<!-- @segment snp3.2:14.1 -->

Lābho siloko sakkāro,

<!-- @segment snp3.2:14.2 -->

Micchāladdho ca yo yaso;

<!-- @segment snp3.2:14.3 -->

Yo cattānaṁ samukkaṁse,

<!-- @segment snp3.2:14.4 -->

Pare ca avajānati.

<!-- @segment snp3.2:15.1 -->

Esā namuci te senā,

<!-- @segment snp3.2:15.2 -->

Kaṇhassābhippahārinī;

<!-- @segment snp3.2:15.3 -->

Na naṁ asūro jināti,

<!-- @segment snp3.2:15.4 -->

Jetvā ca labhate sukhaṁ.

<!-- @segment snp3.2:16.1 -->

Esa muñjaṁ parihare,

<!-- @segment snp3.2:16.2 -->

Dhiratthu mama jīvitaṁ;

<!-- @segment snp3.2:16.3 -->

Saṅgāme me mataṁ seyyo,

<!-- @segment snp3.2:16.4 -->

Yañce jīve parājito.

<!-- @segment snp3.2:17.1 -->

Pagāḷhettha na dissanti,

<!-- @segment snp3.2:17.2 -->

Eke samaṇabrāhmaṇā;

<!-- @segment snp3.2:17.3 -->

Tañca maggaṁ na jānanti,

<!-- @segment snp3.2:17.4 -->

Yena gacchanti subbatā.

<!-- @segment snp3.2:18.1 -->

Samantā dhajiniṁ disvā,

<!-- @segment snp3.2:18.2 -->

Yuttaṁ māraṁ savāhanaṁ;

<!-- @segment snp3.2:18.3 -->

Yuddhāya paccuggacchāmi,

<!-- @segment snp3.2:18.4 -->

Mā maṁ ṭhānā acāvayi.

<!-- @segment snp3.2:19.1 -->

Yaṁ te taṁ nappasahati,

<!-- @segment snp3.2:19.2 -->

Senaṁ loko sadevako;

<!-- @segment snp3.2:19.3 -->

Taṁ te paññāya bhecchāmi,

<!-- @segment snp3.2:19.4 -->

Āmaṁ pattaṁva asmanā.

<!-- @segment snp3.2:20.1 -->

Vasīkaritvā saṅkappaṁ,

<!-- @segment snp3.2:20.2 -->

Satiñca sūpatiṭṭhitaṁ;

<!-- @segment snp3.2:20.3 -->

Raṭṭhā raṭṭhaṁ vicarissaṁ,

<!-- @segment snp3.2:20.4 -->

Sāvake vinayaṁ puthū.

<!-- @segment snp3.2:21.1 -->

Te appamattā pahitattā,

<!-- @segment snp3.2:21.2 -->

Mama sāsanakārakā;

<!-- @segment snp3.2:21.3 -->

Akāmassa te gamissanti,

<!-- @segment snp3.2:21.4 -->

Yattha gantvā na socare”.

<!-- @segment snp3.2:22.1 -->

“Satta vassāni bhagavantaṁ,

<!-- @segment snp3.2:22.2 -->

Anubandhiṁ padāpadaṁ;

<!-- @segment snp3.2:22.3 -->

Otāraṁ nādhigacchissaṁ,

<!-- @segment snp3.2:22.4 -->

Sambuddhassa satīmato.

<!-- @segment snp3.2:23.1 -->

Medavaṇṇaṁva pāsāṇaṁ,

<!-- @segment snp3.2:23.2 -->

Vāyaso anupariyagā;

<!-- @segment snp3.2:23.3 -->

Apettha muduṁ vindema,

<!-- @segment snp3.2:23.4 -->

Api assādanā siyā.

<!-- @segment snp3.2:24.1 -->

Aladdhā tattha assādaṁ,

<!-- @segment snp3.2:24.2 -->

Vāyasetto apakkami;

<!-- @segment snp3.2:24.3 -->

Kākova selamāsajja,

<!-- @segment snp3.2:24.4 -->

Nibbijjāpema gotamaṁ”.

<!-- @segment snp3.2:25.1 -->

Tassa sokaparetassa,

<!-- @segment snp3.2:25.2 -->

Vīṇā kacchā abhassatha;

<!-- @segment snp3.2:25.3 -->

Tato so dummano yakkho,

<!-- @segment snp3.2:25.4 -->

Tatthevantaradhāyathāti.

<!-- @segment snp3.2:25.5 -->

Padhānasuttaṁ dutiyaṁ.
