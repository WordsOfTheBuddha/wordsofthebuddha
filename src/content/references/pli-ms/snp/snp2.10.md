<!-- @segment snp2.10:1.1 -->

Uṭṭhahatha nisīdatha,

<!-- @segment snp2.10:1.2 -->

ko attho supitena vo;

<!-- @segment snp2.10:1.3 -->

Āturānañhi kā niddā,

<!-- @segment snp2.10:1.4 -->

sallaviddhāna ruppataṁ.

<!-- @segment snp2.10:2.1 -->

Uṭṭhahatha nisīdatha,

<!-- @segment snp2.10:2.2 -->

Daḷhaṁ sikkhatha santiyā;

<!-- @segment snp2.10:2.3 -->

Mā vo pamatte viññāya,

<!-- @segment snp2.10:2.4 -->

Maccurājā amohayittha vasānuge.

<!-- @segment snp2.10:3.1 -->

Yāya devā manussā ca,

<!-- @segment snp2.10:3.2 -->

sitā tiṭṭhanti atthikā;

<!-- @segment snp2.10:3.3 -->

Tarathetaṁ visattikaṁ,

<!-- @segment snp2.10:3.4 -->

khaṇo vo mā upaccagā;

<!-- @segment snp2.10:3.5 -->

Khaṇātītā hi socanti,

<!-- @segment snp2.10:3.6 -->

nirayamhi samappitā.

<!-- @segment snp2.10:4.1 -->

Pamādo rajo pamādo,

<!-- @segment snp2.10:4.2 -->

Pamādānupatito rajo;

<!-- @segment snp2.10:4.3 -->

Appamādena vijjāya,

<!-- @segment snp2.10:4.4 -->

Abbahe sallamattanoti.

<!-- @segment snp2.10:4.5 -->

Uṭṭhānasuttaṁ dasamaṁ.
