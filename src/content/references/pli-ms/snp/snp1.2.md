<!-- @segment snp1.2:1.1 -->

“Pakkodano duddhakhīrohamasmi,

<!-- @segment snp1.2:1.2 -->

(iti dhaniyo gopo)

<!-- @segment snp1.2:1.3 -->

Anutīre mahiyā samānavāso;

<!-- @segment snp1.2:1.4 -->

Channā kuṭi āhito gini,

<!-- @segment snp1.2:1.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:2.1 -->

“Akkodhano vigatakhilohamasmi,

<!-- @segment snp1.2:2.2 -->

(iti bhagavā)

<!-- @segment snp1.2:2.3 -->

Anutīre mahiyekarattivāso;

<!-- @segment snp1.2:2.4 -->

Vivaṭā kuṭi nibbuto gini,

<!-- @segment snp1.2:2.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:3.1 -->

“Andhakamakasā na vijjare,

<!-- @segment snp1.2:3.2 -->

(iti dhaniyo gopo)

<!-- @segment snp1.2:3.3 -->

Kacche rūḷhatiṇe caranti gāvo;

<!-- @segment snp1.2:3.4 -->

Vuṭṭhimpi saheyyumāgataṁ,

<!-- @segment snp1.2:3.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:4.1 -->

“Baddhāsi bhisī susaṅkhatā,

<!-- @segment snp1.2:4.2 -->

(iti bhagavā)

<!-- @segment snp1.2:4.3 -->

Tiṇṇo pāragato vineyya oghaṁ;

<!-- @segment snp1.2:4.4 -->

Attho bhisiyā na vijjati,

<!-- @segment snp1.2:4.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:5.1 -->

“Gopī mama assavā alolā,

<!-- @segment snp1.2:5.2 -->

(iti dhaniyo gopo)

<!-- @segment snp1.2:5.3 -->

Dīgharattaṁ saṁvāsiyā manāpā;

<!-- @segment snp1.2:5.4 -->

Tassā na suṇāmi kiñci pāpaṁ,

<!-- @segment snp1.2:5.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:6.1 -->

“Cittaṁ mama assavaṁ vimuttaṁ,

<!-- @segment snp1.2:6.2 -->

(iti bhagavā)

<!-- @segment snp1.2:6.3 -->

Dīgharattaṁ paribhāvitaṁ sudantaṁ;

<!-- @segment snp1.2:6.4 -->

Pāpaṁ pana me na vijjati,

<!-- @segment snp1.2:6.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:7.1 -->

“Attavetanabhatohamasmi,

<!-- @segment snp1.2:7.2 -->

(iti dhaniyo gopo)

<!-- @segment snp1.2:7.3 -->

Puttā ca me samāniyā arogā;

<!-- @segment snp1.2:7.4 -->

Tesaṁ na suṇāmi kiñci pāpaṁ,

<!-- @segment snp1.2:7.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:8.1 -->

“Nāhaṁ bhatakosmi kassaci,

<!-- @segment snp1.2:8.2 -->

(iti bhagavā)

<!-- @segment snp1.2:8.3 -->

Nibbiṭṭhena carāmi sabbaloke;

<!-- @segment snp1.2:8.4 -->

Attho bhatiyā na vijjati,

<!-- @segment snp1.2:8.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:9.1 -->

“Atthi vasā atthi dhenupā,

<!-- @segment snp1.2:9.2 -->

(iti dhaniyo gopo)

<!-- @segment snp1.2:9.3 -->

Godharaṇiyo paveṇiyopi atthi;

<!-- @segment snp1.2:9.4 -->

Usabhopi gavampatīdha atthi,

<!-- @segment snp1.2:9.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:10.1 -->

“Natthi vasā natthi dhenupā,

<!-- @segment snp1.2:10.2 -->

(iti bhagavā)

<!-- @segment snp1.2:10.3 -->

Godharaṇiyo paveṇiyopi natthi;

<!-- @segment snp1.2:10.4 -->

Usabhopi gavampatīdha natthi,

<!-- @segment snp1.2:10.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:11.1 -->

“Khilā nikhātā asampavedhī,

<!-- @segment snp1.2:11.2 -->

(iti dhaniyo gopo)

<!-- @segment snp1.2:11.3 -->

Dāmā muñjamayā navā susaṇṭhānā;

<!-- @segment snp1.2:11.4 -->

Na hi sakkhinti dhenupāpi chettuṁ,

<!-- @segment snp1.2:11.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:12.1 -->

“Usabhoriva chetva bandhanāni,

<!-- @segment snp1.2:12.2 -->

(iti bhagavā)

<!-- @segment snp1.2:12.3 -->

Nāgo pūtilataṁva dālayitvā;

<!-- @segment snp1.2:12.4 -->

Nāhaṁ punupessaṁ gabbhaseyyaṁ,

<!-- @segment snp1.2:12.5 -->

Atha ce patthayasī pavassa deva”.

<!-- @segment snp1.2:13.1 -->

Ninnañca thalañca pūrayanto,

<!-- @segment snp1.2:13.2 -->

Mahāmegho pavassi tāvadeva;

<!-- @segment snp1.2:13.3 -->

Sutvā devassa vassato,

<!-- @segment snp1.2:13.4 -->

Imamatthaṁ dhaniyo abhāsatha.

<!-- @segment snp1.2:14.1 -->

“Lābhā vata no anappakā,

<!-- @segment snp1.2:14.2 -->

Ye mayaṁ bhagavantaṁ addasāma;

<!-- @segment snp1.2:14.3 -->

Saraṇaṁ taṁ upema cakkhuma,

<!-- @segment snp1.2:14.4 -->

Satthā no hohi tuvaṁ mahāmuni.

<!-- @segment snp1.2:15.1 -->

Gopī ca ahañca assavā,

<!-- @segment snp1.2:15.2 -->

Brahmacariyaṁ sugate carāmase;

<!-- @segment snp1.2:15.3 -->

Jātimaraṇassa pāragū,

<!-- @segment snp1.2:15.4 -->

Dukkhassantakarā bhavāmase”.

<!-- @segment snp1.2:16.1 -->

“Nandati puttehi puttimā,

<!-- @segment snp1.2:16.2 -->

(iti māro pāpimā)

<!-- @segment snp1.2:16.3 -->

Gomā gohi tatheva nandati;

<!-- @segment snp1.2:16.4 -->

Upadhī hi narassa nandanā,

<!-- @segment snp1.2:16.5 -->

Na hi so nandati yo nirūpadhi”.

<!-- @segment snp1.2:17.1 -->

“Socati puttehi puttimā,

<!-- @segment snp1.2:17.2 -->

(iti bhagavā)

<!-- @segment snp1.2:17.3 -->

Gomā gohi tatheva socati;

<!-- @segment snp1.2:17.4 -->

Upadhī hi narassa socanā,

<!-- @segment snp1.2:17.5 -->

Na hi so socati yo nirūpadhī”ti.

<!-- @segment snp1.2:17.6 -->

Dhaniyasuttaṁ dutiyaṁ.
