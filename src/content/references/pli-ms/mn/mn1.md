<!-- @segment mn1:1.1 -->

Evaṁ me sutaṁ—

<!-- @segment mn1:1.2 -->

ekaṁ samayaṁ bhagavā ukkaṭṭhāyaṁ viharati subhagavane sālarājamūle.

<!-- @segment mn1:1.3 -->

<!-- @segment mn1:1.4 -->

<!-- @segment mn1:1.5 -->

<!-- @segment mn1:2.1 -->

“sabbadhammamūlapariyāyaṁ vo, bhikkhave, desessāmi.

<!-- @segment mn1:2.3 -->

“Evaṁ, bhante”ti kho te bhikkhū bhagavato paccassosuṁ.
