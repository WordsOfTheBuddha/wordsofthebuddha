<!-- @segment kp7:1.1 -->

Tirokuṭṭesu tiṭṭhanti,

<!-- @segment kp7:1.2 -->

sandhisiṅghāṭakesu ca;

<!-- @segment kp7:1.3 -->

Dvārabāhāsu tiṭṭhanti,

<!-- @segment kp7:1.4 -->

āgantvāna sakaṁ gharaṁ.

<!-- @segment kp7:2.1 -->

Pahūte annapānamhi,

<!-- @segment kp7:2.2 -->

khajjabhojje upaṭṭhite;

<!-- @segment kp7:2.3 -->

Na tesaṁ koci sarati,

<!-- @segment kp7:2.4 -->

sattānaṁ kammapaccayā.

<!-- @segment kp7:3.1 -->

Evaṁ dadanti ñātīnaṁ,

<!-- @segment kp7:3.2 -->

ye honti anukampakā;

<!-- @segment kp7:3.3 -->

Suciṁ paṇītaṁ kālena,

<!-- @segment kp7:3.4 -->

kappiyaṁ pānabhojanaṁ.

<!-- @segment kp7:4.1 -->

Idaṁ vo ñātīnaṁ hotu,

<!-- @segment kp7:4.2 -->

sukhitā hontu ñātayo;

<!-- @segment kp7:4.3 -->

Te ca tattha samāgantvā,

<!-- @segment kp7:4.4 -->

ñātipetā samāgatā.

<!-- @segment kp7:5.1 -->

Pahūte annapānamhi,

<!-- @segment kp7:5.2 -->

sakkaccaṁ anumodare;

<!-- @segment kp7:5.3 -->

Ciraṁ jīvantu no ñātī,

<!-- @segment kp7:5.4 -->

yesaṁ hetu labhāmase.

<!-- @segment kp7:6.1 -->

Amhākañca katā pūjā,

<!-- @segment kp7:6.2 -->

dāyakā ca anipphalā;

<!-- @segment kp7:6.3 -->

Na hi tattha kasī atthi,

<!-- @segment kp7:6.4 -->

gorakkhettha na vijjati.

<!-- @segment kp7:7.1 -->

Vaṇijjā tādisī natthi,

<!-- @segment kp7:7.2 -->

hiraññena kayākayaṁ;

<!-- @segment kp7:7.3 -->

Ito dinnena yāpenti,

<!-- @segment kp7:7.4 -->

petā kālagatā tahiṁ.

<!-- @segment kp7:8.1 -->

Unname udakaṁ vuṭṭhaṁ,

<!-- @segment kp7:8.2 -->

yathā ninnaṁ pavattati;

<!-- @segment kp7:8.3 -->

Evamevaṁ ito dinnaṁ,

<!-- @segment kp7:8.4 -->

petānaṁ upakappati.

<!-- @segment kp7:9.1 -->

Yathā vārivahā pūrā,

<!-- @segment kp7:9.2 -->

paripūrenti sāgaraṁ;

<!-- @segment kp7:9.3 -->

Evamevaṁ ito dinnaṁ,

<!-- @segment kp7:9.4 -->

petānaṁ upakappati.

<!-- @segment kp7:10.1 -->

Adāsi me akāsi me,

<!-- @segment kp7:10.2 -->

ñāti mittā sakhā ca me;

<!-- @segment kp7:10.3 -->

Petānaṁ dakkhiṇaṁ dajjā,

<!-- @segment kp7:10.4 -->

pubbe katamanussaraṁ.

<!-- @segment kp7:11.1 -->

Na hi ruṇṇaṁ vā soko vā,

<!-- @segment kp7:11.2 -->

yā caññā paridevanā;

<!-- @segment kp7:11.3 -->

Na taṁ petānamatthāya,

<!-- @segment kp7:11.4 -->

evaṁ tiṭṭhanti ñātayo.

<!-- @segment kp7:12.1 -->

Ayañca kho dakkhiṇā dinnā,

<!-- @segment kp7:12.2 -->

saṅghamhi suppatiṭṭhitā;

<!-- @segment kp7:12.3 -->

Dīgharattaṁ hitāyassa,

<!-- @segment kp7:12.4 -->

ṭhānaso upakappati.

<!-- @segment kp7:13.1 -->

So ñātidhammo ca ayaṁ nidassito,

<!-- @segment kp7:13.2 -->

Petāna pūjā ca katā uḷārā;

<!-- @segment kp7:13.3 -->

Balañca bhikkhūnamanuppadinnaṁ,

<!-- @segment kp7:13.4 -->

Tumhehi puññaṁ pasutaṁ anappakanti.

<!-- @segment kp7:14.1 -->

Tirokuṭṭasuttaṁ.
