<!-- @segment kp8:1.1 -->

Nidhiṁ nidheti puriso,

<!-- @segment kp8:1.2 -->

gambhīre odakantike;

<!-- @segment kp8:1.3 -->

Atthe kicce samuppanne,

<!-- @segment kp8:1.4 -->

atthāya me bhavissati.

<!-- @segment kp8:2.1 -->

Rājato vā duruttassa,

<!-- @segment kp8:2.2 -->

corato pīḷitassa vā;

<!-- @segment kp8:2.3 -->

Iṇassa vā pamokkhāya,

<!-- @segment kp8:2.4 -->

dubbhikkhe āpadāsu vā;

<!-- @segment kp8:2.5 -->

Etadatthāya lokasmiṁ,

<!-- @segment kp8:2.6 -->

nidhi nāma nidhīyati.

<!-- @segment kp8:3.1 -->

Tāvassunihito santo,

<!-- @segment kp8:3.2 -->

gambhīre odakantike;

<!-- @segment kp8:3.3 -->

Na sabbo sabbadā eva,

<!-- @segment kp8:3.4 -->

tassa taṁ upakappati.

<!-- @segment kp8:4.1 -->

Nidhi vā ṭhānā cavati,

<!-- @segment kp8:4.2 -->

saññā vāssa vimuyhati;

<!-- @segment kp8:4.3 -->

Nāgā vā apanāmenti,

<!-- @segment kp8:4.4 -->

yakkhā vāpi haranti naṁ.

<!-- @segment kp8:5.1 -->

Appiyā vāpi dāyādā,

<!-- @segment kp8:5.2 -->

uddharanti apassato;

<!-- @segment kp8:5.3 -->

Yadā puññakkhayo hoti,

<!-- @segment kp8:5.4 -->

sabbametaṁ vinassati.

<!-- @segment kp8:6.1 -->

Yassa dānena sīlena,

<!-- @segment kp8:6.2 -->

saṁyamena damena ca;

<!-- @segment kp8:6.3 -->

Nidhī sunihito hoti,

<!-- @segment kp8:6.4 -->

itthiyā purisassa vā.

<!-- @segment kp8:7.1 -->

Cetiyamhi ca saṅghe vā,

<!-- @segment kp8:7.2 -->

puggale atithīsu vā;

<!-- @segment kp8:7.3 -->

Mātari pitari cāpi,

<!-- @segment kp8:7.4 -->

atho jeṭṭhamhi bhātari.

<!-- @segment kp8:8.1 -->

Eso nidhi sunihito,

<!-- @segment kp8:8.2 -->

ajeyyo anugāmiko;

<!-- @segment kp8:8.3 -->

Pahāya gamanīyesu,

<!-- @segment kp8:8.4 -->

etaṁ ādāya gacchati.

<!-- @segment kp8:9.1 -->

Asādhāraṇamaññesaṁ,

<!-- @segment kp8:9.2 -->

acorāharaṇo nidhi;

<!-- @segment kp8:9.3 -->

Kayirātha dhīro puññāni,

<!-- @segment kp8:9.4 -->

yo nidhi anugāmiko.

<!-- @segment kp8:10.1 -->

Esa devamanussānaṁ,

<!-- @segment kp8:10.2 -->

sabbakāmadado nidhi;

<!-- @segment kp8:10.3 -->

Yaṁ yadevābhipatthenti,

<!-- @segment kp8:10.4 -->

sabbametena labbhati.

<!-- @segment kp8:11.1 -->

Suvaṇṇatā susaratā,

<!-- @segment kp8:11.2 -->

susaṇṭhānā surūpatā;

<!-- @segment kp8:11.3 -->

Ādhipaccaparivāro,

<!-- @segment kp8:11.4 -->

sabbametena labbhati.

<!-- @segment kp8:12.1 -->

Padesarajjaṁ issariyaṁ,

<!-- @segment kp8:12.2 -->

cakkavattisukhaṁ piyaṁ;

<!-- @segment kp8:12.3 -->

Devarajjampi dibbesu,

<!-- @segment kp8:12.4 -->

sabbametena labbhati.

<!-- @segment kp8:13.1 -->

Mānussikā ca sampatti,

<!-- @segment kp8:13.2 -->

devaloke ca yā rati;

<!-- @segment kp8:13.3 -->

Yā ca nibbānasampatti,

<!-- @segment kp8:13.4 -->

sabbametena labbhati.

<!-- @segment kp8:14.1 -->

Mittasampadamāgamma,

<!-- @segment kp8:14.2 -->

yonisova payuñjato;

<!-- @segment kp8:14.3 -->

Vijjāvimuttivasībhāvo,

<!-- @segment kp8:14.4 -->

sabbametena labbhati.

<!-- @segment kp8:15.1 -->

Paṭisambhidā vimokkhā ca,

<!-- @segment kp8:15.2 -->

yā ca sāvakapāramī;

<!-- @segment kp8:15.3 -->

Paccekabodhi buddhabhūmi,

<!-- @segment kp8:15.4 -->

sabbametena labbhati.

<!-- @segment kp8:16.1 -->

Evaṁ mahatthikā esā,

<!-- @segment kp8:16.2 -->

yadidaṁ puññasampadā;

<!-- @segment kp8:16.3 -->

Tasmā dhīrā pasaṁsanti,

<!-- @segment kp8:16.4 -->

paṇḍitā katapuññatanti.

<!-- @segment kp8:17.1 -->

Nidhikaṇḍasuttaṁ.
