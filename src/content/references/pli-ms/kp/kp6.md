<!-- @segment kp6:1.1 -->

Yānīdha bhūtāni samāgatāni,

<!-- @segment kp6:1.2 -->

Bhummāni vā yāni va antalikkhe;

<!-- @segment kp6:1.3 -->

Sabbeva bhūtā sumanā bhavantu,

<!-- @segment kp6:1.4 -->

Athopi sakkacca suṇantu bhāsitaṁ.

<!-- @segment kp6:2.1 -->

Tasmā hi bhūtā nisāmetha sabbe,

<!-- @segment kp6:2.2 -->

Mettaṁ karotha mānusiyā pajāya;

<!-- @segment kp6:2.3 -->

Divā ca ratto ca haranti ye baliṁ,

<!-- @segment kp6:2.4 -->

Tasmā hi ne rakkhatha appamattā.

<!-- @segment kp6:3.1 -->

Yaṁ kiñci vittaṁ idha vā huraṁ vā,

<!-- @segment kp6:3.2 -->

Saggesu vā yaṁ ratanaṁ paṇītaṁ;

<!-- @segment kp6:3.3 -->

Na no samaṁ atthi tathāgatena,

<!-- @segment kp6:3.4 -->

Idampi buddhe ratanaṁ paṇītaṁ;

<!-- @segment kp6:3.5 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:4.1 -->

Khayaṁ virāgaṁ amataṁ paṇītaṁ,

<!-- @segment kp6:4.2 -->

Yadajjhagā sakyamunī samāhito;

<!-- @segment kp6:4.3 -->

Na tena dhammena samatthi kiñci,

<!-- @segment kp6:4.4 -->

Idampi dhamme ratanaṁ paṇītaṁ;

<!-- @segment kp6:4.5 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:5.1 -->

Yaṁ buddhaseṭṭho parivaṇṇayī suciṁ,

<!-- @segment kp6:5.2 -->

Samādhimānantarikaññamāhu;

<!-- @segment kp6:5.3 -->

Samādhinā tena samo na vijjati,

<!-- @segment kp6:5.4 -->

Idampi dhamme ratanaṁ paṇītaṁ;

<!-- @segment kp6:5.5 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:6.1 -->

Ye puggalā aṭṭha sataṁ pasatthā,

<!-- @segment kp6:6.2 -->

Cattāri etāni yugāni honti;

<!-- @segment kp6:6.3 -->

Te dakkhiṇeyyā sugatassa sāvakā,

<!-- @segment kp6:6.4 -->

Etesu dinnāni mahapphalāni;

<!-- @segment kp6:6.5 -->

Idampi saṅghe ratanaṁ paṇītaṁ,

<!-- @segment kp6:6.6 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:7.1 -->

Ye suppayuttā manasā daḷhena,

<!-- @segment kp6:7.2 -->

Nikkāmino gotamasāsanamhi;

<!-- @segment kp6:7.3 -->

Te pattipattā amataṁ vigayha,

<!-- @segment kp6:7.4 -->

Laddhā mudhā nibbutiṁ bhuñjamānā;

<!-- @segment kp6:7.5 -->

Idampi saṅghe ratanaṁ paṇītaṁ,

<!-- @segment kp6:7.6 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:8.1 -->

Yathindakhīlo pathavissito siyā,

<!-- @segment kp6:8.2 -->

Catubbhi vātehi asampakampiyo;

<!-- @segment kp6:8.3 -->

Tathūpamaṁ sappurisaṁ vadāmi,

<!-- @segment kp6:8.4 -->

Yo ariyasaccāni avecca passati;

<!-- @segment kp6:8.5 -->

Idampi saṅghe ratanaṁ paṇītaṁ,

<!-- @segment kp6:8.6 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:9.1 -->

Ye ariyasaccāni vibhāvayanti,

<!-- @segment kp6:9.2 -->

Gambhīrapaññena sudesitāni;

<!-- @segment kp6:9.3 -->

Kiñcāpi te honti bhusaṁ pamattā,

<!-- @segment kp6:9.4 -->

Na te bhavaṁ aṭṭhamamādiyanti;

<!-- @segment kp6:9.5 -->

Idampi saṅghe ratanaṁ paṇītaṁ,

<!-- @segment kp6:9.6 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:10.1 -->

Sahāvassa dassanasampadāya,

<!-- @segment kp6:10.2 -->

Tayassu dhammā jahitā bhavanti;

<!-- @segment kp6:10.3 -->

Sakkāyadiṭṭhī vicikicchitañca,

<!-- @segment kp6:10.4 -->

Sīlabbataṁ vāpi yadatthi kiñci.

<!-- @segment kp6:11.1 -->

Catūhapāyehi ca vippamutto,

<!-- @segment kp6:11.2 -->

Chaccābhiṭhānāni abhabba kātuṁ;

<!-- @segment kp6:11.3 -->

Idampi saṅghe ratanaṁ paṇītaṁ,

<!-- @segment kp6:11.4 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:12.1 -->

Kiñcāpi so kamma karoti pāpakaṁ,

<!-- @segment kp6:12.2 -->

Kāyena vācā uda cetasā vā;

<!-- @segment kp6:12.3 -->

Abhabba so tassa paṭicchadāya,

<!-- @segment kp6:12.4 -->

Abhabbatā diṭṭhapadassa vuttā;

<!-- @segment kp6:12.5 -->

Idampi saṅghe ratanaṁ paṇītaṁ,

<!-- @segment kp6:12.6 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:13.1 -->

Vanappagumbe yatha phussitagge,

<!-- @segment kp6:13.2 -->

Gimhānamāse paṭhamasmiṁ gimhe;

<!-- @segment kp6:13.3 -->

Tathūpamaṁ dhammavaraṁ adesayi,

<!-- @segment kp6:13.4 -->

Nibbānagāmiṁ paramaṁ hitāya;

<!-- @segment kp6:13.5 -->

Idampi buddhe ratanaṁ paṇītaṁ,

<!-- @segment kp6:13.6 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:14.1 -->

Varo varaññū varado varāharo,

<!-- @segment kp6:14.2 -->

Anuttaro dhammavaraṁ adesayi;

<!-- @segment kp6:14.3 -->

Idampi buddhe ratanaṁ paṇītaṁ,

<!-- @segment kp6:14.4 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:15.1 -->

Khīṇaṁ purāṇaṁ nava natthi sambhavaṁ,

<!-- @segment kp6:15.2 -->

Virattacittāyatike bhavasmiṁ;

<!-- @segment kp6:15.3 -->

Te khīṇabījā avirūḷhichandā,

<!-- @segment kp6:15.4 -->

Nibbanti dhīrā yathāyaṁ padīpo;

<!-- @segment kp6:15.5 -->

Idampi saṅghe ratanaṁ paṇītaṁ,

<!-- @segment kp6:15.6 -->

Etena saccena suvatthi hotu.

<!-- @segment kp6:16.1 -->

Yānīdha bhūtāni samāgatāni,

<!-- @segment kp6:16.2 -->

Bhummāni vā yāni va antalikkhe;

<!-- @segment kp6:16.3 -->

Tathāgataṁ devamanussapūjitaṁ,

<!-- @segment kp6:16.4 -->

Buddhaṁ namassāma suvatthi hotu.

<!-- @segment kp6:17.1 -->

Yānīdha bhūtāni samāgatāni,

<!-- @segment kp6:17.2 -->

Bhummāni vā yāni va antalikkhe;

<!-- @segment kp6:17.3 -->

Tathāgataṁ devamanussapūjitaṁ,

<!-- @segment kp6:17.4 -->

Dhammaṁ namassāma suvatthi hotu.

<!-- @segment kp6:18.1 -->

Yānīdha bhūtāni samāgatāni,

<!-- @segment kp6:18.2 -->

Bhummāni vā yāni va antalikkhe;

<!-- @segment kp6:18.3 -->

Tathāgataṁ devamanussapūjitaṁ,

<!-- @segment kp6:18.4 -->

Saṅghaṁ namassāma suvatthi hotūti.

<!-- @segment kp6:19.1 -->

Ratanasuttaṁ.
