<!-- @segment an5.245:1.1 -->

“Pañcime, bhikkhave, ādīnavā duccarite.

<!-- @segment an5.245:1.2 -->

Katame pañca?

<!-- @segment an5.245:1.3 -->

Attāpi attānaṁ upavadati;

<!-- @segment an5.245:1.4 -->

anuvicca viññū garahanti;

<!-- @segment an5.245:1.5 -->

pāpako kittisaddo abbhuggacchati;

<!-- @segment an5.245:1.6 -->

saddhammā vuṭṭhāti;

<!-- @segment an5.245:1.7 -->

asaddhamme patiṭṭhāti.

<!-- @segment an5.245:1.8 -->

Ime kho, bhikkhave, pañca ādīnavā duccarite.

<!-- @segment an5.245:2.1 -->

Pañcime, bhikkhave, ānisaṁsā sucarite.

<!-- @segment an5.245:2.2 -->

Katame pañca?

<!-- @segment an5.245:2.3 -->

Attāpi attānaṁ na upavadati;

<!-- @segment an5.245:2.4 -->

anuvicca viññū pasaṁsanti;

<!-- @segment an5.245:2.5 -->

kalyāṇo kittisaddo abbhuggacchati;

<!-- @segment an5.245:2.6 -->

asaddhammā vuṭṭhāti;

<!-- @segment an5.245:2.7 -->

saddhamme patiṭṭhāti.

<!-- @segment an5.245:2.8 -->

Ime kho, bhikkhave, pañca ānisaṁsā sucarite”ti.

<!-- @segment an5.245:2.9 -->

Pañcamaṁ.
