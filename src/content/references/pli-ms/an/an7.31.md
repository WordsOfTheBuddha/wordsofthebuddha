<!-- @segment an7.31:1.1 -->

“Sattime, bhikkhave, upāsakassa parābhavā …pe…

<!-- @segment an7.31:1.2 -->

sattime, bhikkhave, upāsakassa sambhavā.

<!-- @segment an7.31:1.3 -->

Katame satta?

<!-- @segment an7.31:1.4 -->

Bhikkhudassanaṁ na hāpeti,

<!-- @segment an7.31:1.5 -->

saddhammassavanaṁ nappamajjati,

<!-- @segment an7.31:1.6 -->

adhisīle sikkhati,

<!-- @segment an7.31:1.7 -->

pasādabahulo hoti,

<!-- @segment an7.31:1.8 -->

bhikkhūsu theresu ceva navesu ca majjhimesu ca anupārambhacitto dhammaṁ suṇāti na randhagavesī,

<!-- @segment an7.31:1.9 -->

na ito bahiddhā dakkhiṇeyyaṁ gavesati,

<!-- @segment an7.31:1.10 -->

idha ca pubbakāraṁ karoti.

<!-- @segment an7.31:1.11 -->

Ime kho, bhikkhave, satta upāsakassa sambhavāti.

<!-- @segment an7.31:2.1 -->

Dassanaṁ bhāvitattānaṁ,

<!-- @segment an7.31:2.2 -->

Yo hāpeti upāsako;

<!-- @segment an7.31:2.3 -->

Savanañca ariyadhammānaṁ,

<!-- @segment an7.31:2.4 -->

Adhisīle na sikkhati.

<!-- @segment an7.31:3.1 -->

Appasādo ca bhikkhūsu,

<!-- @segment an7.31:3.2 -->

bhiyyo bhiyyo pavaḍḍhati;

<!-- @segment an7.31:3.3 -->

Upārambhakacitto ca,

<!-- @segment an7.31:3.4 -->

saddhammaṁ sotumicchati.

<!-- @segment an7.31:4.1 -->

Ito ca bahiddhā aññaṁ,

<!-- @segment an7.31:4.2 -->

dakkhiṇeyyaṁ gavesati;

<!-- @segment an7.31:4.3 -->

Tattheva ca pubbakāraṁ,

<!-- @segment an7.31:4.4 -->

yo karoti upāsako.

<!-- @segment an7.31:5.1 -->

Ete kho parihāniye,

<!-- @segment an7.31:5.2 -->

satta dhamme sudesite;

<!-- @segment an7.31:5.3 -->

Upāsako sevamāno,

<!-- @segment an7.31:5.4 -->

saddhammā parihāyati.

<!-- @segment an7.31:6.1 -->

Dassanaṁ bhāvitattānaṁ,

<!-- @segment an7.31:6.2 -->

Yo na hāpeti upāsako;

<!-- @segment an7.31:6.3 -->

Savanañca ariyadhammānaṁ,

<!-- @segment an7.31:6.4 -->

Adhisīle ca sikkhati.

<!-- @segment an7.31:7.1 -->

Pasādo cassa bhikkhūsu,

<!-- @segment an7.31:7.2 -->

bhiyyo bhiyyo pavaḍḍhati;

<!-- @segment an7.31:7.3 -->

Anupārambhacitto ca,

<!-- @segment an7.31:7.4 -->

saddhammaṁ sotumicchati.

<!-- @segment an7.31:8.1 -->

Na ito bahiddhā aññaṁ,

<!-- @segment an7.31:8.2 -->

dakkhiṇeyyaṁ gavesati;

<!-- @segment an7.31:8.3 -->

Idheva ca pubbakāraṁ,

<!-- @segment an7.31:8.4 -->

yo karoti upāsako.

<!-- @segment an7.31:9.1 -->

Ete kho aparihāniye,

<!-- @segment an7.31:9.2 -->

Satta dhamme sudesite;

<!-- @segment an7.31:9.3 -->

Upāsako sevamāno,

<!-- @segment an7.31:9.4 -->

Saddhammā na parihāyatī”ti.

<!-- @segment an7.31:9.5 -->

Ekādasamaṁ.

<!-- @segment an7.31:9.6 -->

Vajjisattakavaggo tatiyo.

<!-- @segment an7.31:10.0 -->

Tassuddānaṁ

<!-- @segment an7.31:10.1 -->

Sāranda vassakāro ca,

<!-- @segment an7.31:10.2 -->

tisattakāni bhikkhukā;

<!-- @segment an7.31:10.3 -->

Bodhisaññā dve ca hāni,

<!-- @segment an7.31:10.4 -->

vipatti ca parābhavoti.
