<!-- @segment sn1.49:1.1 -->

“Yedha maccharino loke,

<!-- @segment sn1.49:1.2 -->

kadariyā paribhāsakā;

<!-- @segment sn1.49:1.3 -->

Aññesaṁ dadamānānaṁ,

<!-- @segment sn1.49:1.4 -->

antarāyakarā narā.

<!-- @segment sn1.49:2.1 -->

Kīdiso tesaṁ vipāko,

<!-- @segment sn1.49:2.2 -->

samparāyo ca kīdiso;

<!-- @segment sn1.49:2.3 -->

Bhagavantaṁ puṭṭhumāgamma,

<!-- @segment sn1.49:2.4 -->

kathaṁ jānemu taṁ mayan”ti.

<!-- @segment sn1.49:3.1 -->

“Yedha maccharino loke,

<!-- @segment sn1.49:3.2 -->

kadariyā paribhāsakā;

<!-- @segment sn1.49:3.3 -->

Aññesaṁ dadamānānaṁ,

<!-- @segment sn1.49:3.4 -->

antarāyakarā narā.

<!-- @segment sn1.49:4.1 -->

Nirayaṁ tiracchānayoniṁ,

<!-- @segment sn1.49:4.2 -->

yamalokaṁ upapajjare;

<!-- @segment sn1.49:4.3 -->

Sace enti manussattaṁ,

<!-- @segment sn1.49:4.4 -->

dalidde jāyare kule.

<!-- @segment sn1.49:5.1 -->

Coḷaṁ piṇḍo ratī khiḍḍā,

<!-- @segment sn1.49:5.2 -->

yattha kicchena labbhati;

<!-- @segment sn1.49:5.3 -->

Parato āsīsare bālā,

<!-- @segment sn1.49:5.4 -->

tampi tesaṁ na labbhati;

<!-- @segment sn1.49:5.5 -->

Diṭṭhe dhammesa vipāko,

<!-- @segment sn1.49:5.6 -->

samparāye ca duggatī”ti.

<!-- @segment sn1.49:6.1 -->

“Itihetaṁ vijānāma,

<!-- @segment sn1.49:6.2 -->

aññaṁ pucchāma gotama;

<!-- @segment sn1.49:6.3 -->

Yedha laddhā manussattaṁ,

<!-- @segment sn1.49:6.4 -->

vadaññū vītamaccharā.

<!-- @segment sn1.49:7.1 -->

Buddhe pasannā dhamme ca,

<!-- @segment sn1.49:7.2 -->

saṅghe ca tibbagāravā;

<!-- @segment sn1.49:7.3 -->

Kīdiso tesaṁ vipāko,

<!-- @segment sn1.49:7.4 -->

samparāyo ca kīdiso;

<!-- @segment sn1.49:7.5 -->

Bhagavantaṁ puṭṭhumāgamma,

<!-- @segment sn1.49:7.6 -->

kathaṁ jānemu taṁ mayan”ti.

<!-- @segment sn1.49:8.1 -->

“Yedha laddhā manussattaṁ,

<!-- @segment sn1.49:8.2 -->

vadaññū vītamaccharā;

<!-- @segment sn1.49:8.3 -->

Buddhe pasannā dhamme ca,

<!-- @segment sn1.49:8.4 -->

saṅghe ca tibbagāravā;

<!-- @segment sn1.49:8.5 -->

Ete saggā pakāsanti,

<!-- @segment sn1.49:8.6 -->

yattha te upapajjare.

<!-- @segment sn1.49:9.1 -->

Sace enti manussattaṁ,

<!-- @segment sn1.49:9.2 -->

aḍḍhe ājāyare kule;

<!-- @segment sn1.49:9.3 -->

Coḷaṁ piṇḍo ratī khiḍḍā,

<!-- @segment sn1.49:9.4 -->

yatthākicchena labbhati.

<!-- @segment sn1.49:10.1 -->

Parasambhatesu bhogesu,

<!-- @segment sn1.49:10.2 -->

vasavattīva modare;

<!-- @segment sn1.49:10.3 -->

Diṭṭhe dhammesa vipāko,

<!-- @segment sn1.49:10.4 -->

samparāye ca suggatī”ti.
