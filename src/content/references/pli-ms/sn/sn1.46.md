<!-- @segment sn1.46:1.1 -->

“Accharāgaṇasaṅghuṭṭhaṁ,

<!-- @segment sn1.46:1.2 -->

pisācagaṇasevitaṁ;

<!-- @segment sn1.46:1.3 -->

Vanantaṁ mohanaṁ nāma,

<!-- @segment sn1.46:1.4 -->

kathaṁ yātrā bhavissatī”ti.

<!-- @segment sn1.46:2.1 -->

“Ujuko nāma so maggo,

<!-- @segment sn1.46:2.2 -->

abhayā nāma sā disā;

<!-- @segment sn1.46:2.3 -->

Ratho akūjano nāma,

<!-- @segment sn1.46:2.4 -->

dhammacakkehi saṁyuto.

<!-- @segment sn1.46:3.1 -->

Hirī tassa apālambo,

<!-- @segment sn1.46:3.2 -->

satyassa parivāraṇaṁ;

<!-- @segment sn1.46:3.3 -->

Dhammāhaṁ sārathiṁ brūmi,

<!-- @segment sn1.46:3.4 -->

sammādiṭṭhipurejavaṁ.

<!-- @segment sn1.46:4.1 -->

Yassa etādisaṁ yānaṁ,

<!-- @segment sn1.46:4.2 -->

itthiyā purisassa vā;

<!-- @segment sn1.46:4.3 -->

Sa ve etena yānena,

<!-- @segment sn1.46:4.4 -->

nibbānasseva santike”ti.
