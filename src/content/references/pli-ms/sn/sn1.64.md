<!-- @segment sn1.64:1.1 -->

“Kiṁsu saṁyojano loko,

<!-- @segment sn1.64:1.2 -->

kiṁsu tassa vicāraṇaṁ;

<!-- @segment sn1.64:1.3 -->

Kissassu vippahānena,

<!-- @segment sn1.64:1.4 -->

nibbānaṁ iti vuccatī”ti.

<!-- @segment sn1.64:2.1 -->

“Nandīsaṁyojano loko,

<!-- @segment sn1.64:2.2 -->

vitakkassa vicāraṇaṁ;

<!-- @segment sn1.64:2.3 -->

Taṇhāya vippahānena,

<!-- @segment sn1.64:2.4 -->

nibbānaṁ iti vuccatī”ti.
