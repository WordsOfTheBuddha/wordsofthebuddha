<!-- @segment sn1.65:1.1 -->

“Kiṁsu sambandhano loko,

<!-- @segment sn1.65:1.2 -->

kiṁsu tassa vicāraṇaṁ;

<!-- @segment sn1.65:1.3 -->

Kissassu vippahānena,

<!-- @segment sn1.65:1.4 -->

sabbaṁ chindati bandhanan”ti.

<!-- @segment sn1.65:2.1 -->

“Nandīsambandhano loko,

<!-- @segment sn1.65:2.2 -->

vitakkassa vicāraṇaṁ;

<!-- @segment sn1.65:2.3 -->

Taṇhāya vippahānena,

<!-- @segment sn1.65:2.4 -->

sabbaṁ chindati bandhanan”ti.
