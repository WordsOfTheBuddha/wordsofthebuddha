<!-- @segment sn1.72:1.1 -->

“Kiṁsu rathassa paññāṇaṁ,

<!-- @segment sn1.72:1.2 -->

kiṁsu paññāṇamaggino;

<!-- @segment sn1.72:1.3 -->

Kiṁsu raṭṭhassa paññāṇaṁ,

<!-- @segment sn1.72:1.4 -->

kiṁsu paññāṇamitthiyā”ti.

<!-- @segment sn1.72:2.1 -->

“Dhajo rathassa paññāṇaṁ,

<!-- @segment sn1.72:2.2 -->

dhūmo paññāṇamaggino;

<!-- @segment sn1.72:2.3 -->

Rājā raṭṭhassa paññāṇaṁ,

<!-- @segment sn1.72:2.4 -->

bhattā paññāṇamitthiyā”ti.
