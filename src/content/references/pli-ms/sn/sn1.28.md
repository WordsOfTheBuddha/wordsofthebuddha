<!-- @segment sn1.28:1.1 -->

“Mahaddhanā mahābhogā,

<!-- @segment sn1.28:1.2 -->

raṭṭhavantopi khattiyā;

<!-- @segment sn1.28:1.3 -->

Aññamaññābhigijjhanti,

<!-- @segment sn1.28:1.4 -->

kāmesu analaṅkatā.

<!-- @segment sn1.28:2.1 -->

Tesu ussukkajātesu,

<!-- @segment sn1.28:2.2 -->

bhavasotānusārisu;

<!-- @segment sn1.28:2.3 -->

Kedha taṇhaṁ pajahiṁsu,

<!-- @segment sn1.28:2.4 -->

ke lokasmiṁ anussukā”ti.

<!-- @segment sn1.28:3.1 -->

“Hitvā agāraṁ pabbajitā,

<!-- @segment sn1.28:3.2 -->

hitvā puttaṁ pasuṁ piyaṁ;

<!-- @segment sn1.28:3.3 -->

Hitvā rāgañca dosañca,

<!-- @segment sn1.28:3.4 -->

avijjañca virājiya;

<!-- @segment sn1.28:3.5 -->

Khīṇāsavā arahanto,

<!-- @segment sn1.28:3.6 -->

te lokasmiṁ anussukā”ti.
