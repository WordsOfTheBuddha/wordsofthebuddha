<!-- @segment sn1.48:1.1 -->

“Idañhi taṁ jetavanaṁ,

<!-- @segment sn1.48:1.2 -->

isisaṅghanisevitaṁ;

<!-- @segment sn1.48:1.3 -->

Āvutthaṁ dhammarājena,

<!-- @segment sn1.48:1.4 -->

pītisañjananaṁ mama.

<!-- @segment sn1.48:2.1 -->

Kammaṁ vijjā ca dhammo ca,

<!-- @segment sn1.48:2.2 -->

sīlaṁ jīvitamuttamaṁ;

<!-- @segment sn1.48:2.3 -->

Etena maccā sujjhanti,

<!-- @segment sn1.48:2.4 -->

na gottena dhanena vā.

<!-- @segment sn1.48:3.1 -->

Tasmā hi paṇḍito poso,

<!-- @segment sn1.48:3.2 -->

sampassaṁ atthamattano;

<!-- @segment sn1.48:3.3 -->

Yoniso vicine dhammaṁ,

<!-- @segment sn1.48:3.4 -->

evaṁ tattha visujjhati.

<!-- @segment sn1.48:4.1 -->

Sāriputtova paññāya,

<!-- @segment sn1.48:4.2 -->

sīlena upasamena ca;

<!-- @segment sn1.48:4.3 -->

Yopi pāraṅgato bhikkhu,

<!-- @segment sn1.48:4.4 -->

etāvaparamo siyā”ti.
