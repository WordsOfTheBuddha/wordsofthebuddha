<!-- @segment sn1.79:1.1 -->

“Kiṁsu bandhati pātheyyaṁ,

<!-- @segment sn1.79:1.2 -->

kiṁsu bhogānamāsayo;

<!-- @segment sn1.79:1.3 -->

Kiṁsu naraṁ parikassati,

<!-- @segment sn1.79:1.4 -->

kiṁsu lokasmi dujjahaṁ;

<!-- @segment sn1.79:1.5 -->

Kismiṁ baddhā puthū sattā,

<!-- @segment sn1.79:1.6 -->

pāsena sakuṇī yathā”ti.

<!-- @segment sn1.79:2.1 -->

“Saddhā bandhati pātheyyaṁ,

<!-- @segment sn1.79:2.2 -->

sirī bhogānamāsayo;

<!-- @segment sn1.79:2.3 -->

Icchā naraṁ parikassati,

<!-- @segment sn1.79:2.4 -->

icchā lokasmi dujjahā;

<!-- @segment sn1.79:2.5 -->

Icchābaddhā puthū sattā,

<!-- @segment sn1.79:2.6 -->

pāsena sakuṇī yathā”ti.
