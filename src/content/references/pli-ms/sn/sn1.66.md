<!-- @segment sn1.66:1.1 -->

“Kenassubbhāhato loko,

<!-- @segment sn1.66:1.2 -->

kenassu parivārito;

<!-- @segment sn1.66:1.3 -->

Kena sallena otiṇṇo,

<!-- @segment sn1.66:1.4 -->

kissa dhūpāyito sadā”ti.

<!-- @segment sn1.66:2.1 -->

“Maccunābbhāhato loko,

<!-- @segment sn1.66:2.2 -->

jarāya parivārito;

<!-- @segment sn1.66:2.3 -->

Taṇhāsallena otiṇṇo,

<!-- @segment sn1.66:2.4 -->

icchādhūpāyito sadā”ti.
