<!-- @segment sn1.30:1.1 -->

“Eṇijaṅghaṁ kisaṁ vīraṁ,

<!-- @segment sn1.30:1.2 -->

appāhāraṁ alolupaṁ;

<!-- @segment sn1.30:1.3 -->

Sīhaṁvekacaraṁ nāgaṁ,

<!-- @segment sn1.30:1.4 -->

kāmesu anapekkhinaṁ;

<!-- @segment sn1.30:1.5 -->

Upasaṅkamma pucchāma,

<!-- @segment sn1.30:1.6 -->

kathaṁ dukkhā pamuccatī”ti.

<!-- @segment sn1.30:2.1 -->

“Pañca kāmaguṇā loke,

<!-- @segment sn1.30:2.2 -->

manochaṭṭhā paveditā;

<!-- @segment sn1.30:2.3 -->

Ettha chandaṁ virājetvā,

<!-- @segment sn1.30:2.4 -->

evaṁ dukkhā pamuccatī”ti.

<!-- @segment sn1.30:2.5 -->

Sattivaggo tatiyo.

<!-- @segment sn1.30:3.0 -->

Tassuddānaṁ

<!-- @segment sn1.30:3.1 -->

Sattiyā phusati ceva,

<!-- @segment sn1.30:3.2 -->

jaṭā manonivāraṇā;

<!-- @segment sn1.30:3.3 -->

Arahantena pajjoto,

<!-- @segment sn1.30:3.4 -->

sarā mahaddhanena ca;

<!-- @segment sn1.30:3.5 -->

Catucakkena navamaṁ,

<!-- @segment sn1.30:3.6 -->

eṇijaṅghena te dasāti.
