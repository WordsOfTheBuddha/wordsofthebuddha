<!-- @segment sn1.77:1.1 -->

“Kiṁsu issariyaṁ loke,

<!-- @segment sn1.77:1.2 -->

kiṁsu bhaṇḍānamuttamaṁ;

<!-- @segment sn1.77:1.3 -->

Kiṁsu satthamalaṁ loke,

<!-- @segment sn1.77:1.4 -->

kiṁsu lokasmimabbudaṁ.

<!-- @segment sn1.77:2.1 -->

Kiṁsu harantaṁ vārenti,

<!-- @segment sn1.77:2.2 -->

haranto pana ko piyo;

<!-- @segment sn1.77:2.3 -->

Kiṁsu punappunāyantaṁ,

<!-- @segment sn1.77:2.4 -->

abhinandanti paṇḍitā”ti.

<!-- @segment sn1.77:3.1 -->

“Vaso issariyaṁ loke,

<!-- @segment sn1.77:3.2 -->

itthī bhaṇḍānamuttamaṁ;

<!-- @segment sn1.77:3.3 -->

Kodho satthamalaṁ loke,

<!-- @segment sn1.77:3.4 -->

corā lokasmimabbudā.

<!-- @segment sn1.77:4.1 -->

Coraṁ harantaṁ vārenti,

<!-- @segment sn1.77:4.2 -->

haranto samaṇo piyo;

<!-- @segment sn1.77:4.3 -->

Samaṇaṁ punappunāyantaṁ,

<!-- @segment sn1.77:4.4 -->

abhinandanti paṇḍitā”ti.
