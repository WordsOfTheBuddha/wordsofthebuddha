<!-- @segment sn1.62:1.1 -->

“Kenassu nīyati loko,

<!-- @segment sn1.62:1.2 -->

kenassu parikassati;

<!-- @segment sn1.62:1.3 -->

Kissassu ekadhammassa,

<!-- @segment sn1.62:1.4 -->

sabbeva vasamanvagū”ti.

<!-- @segment sn1.62:2.1 -->

“Cittena nīyati loko,

<!-- @segment sn1.62:2.2 -->

cittena parikassati;

<!-- @segment sn1.62:2.3 -->

Cittassa ekadhammassa,

<!-- @segment sn1.62:2.4 -->

sabbeva vasamanvagū”ti.
