<!-- @segment sn1.81:1.1 -->

“Kesūdha araṇā loke,

<!-- @segment sn1.81:1.2 -->

kesaṁ vusitaṁ na nassati;

<!-- @segment sn1.81:1.3 -->

Kedha icchaṁ parijānanti,

<!-- @segment sn1.81:1.4 -->

kesaṁ bhojissiyaṁ sadā.

<!-- @segment sn1.81:2.1 -->

Kiṁsu mātā pitā bhātā,

<!-- @segment sn1.81:2.2 -->

vandanti naṁ patiṭṭhitaṁ;

<!-- @segment sn1.81:2.3 -->

Kiṁsu idha jātihīnaṁ,

<!-- @segment sn1.81:2.4 -->

abhivādenti khattiyā”ti.

<!-- @segment sn1.81:3.1 -->

“Samaṇīdha araṇā loke,

<!-- @segment sn1.81:3.2 -->

Samaṇānaṁ vusitaṁ na nassati;

<!-- @segment sn1.81:3.3 -->

Samaṇā icchaṁ parijānanti,

<!-- @segment sn1.81:3.4 -->

Samaṇānaṁ bhojissiyaṁ sadā.

<!-- @segment sn1.81:4.1 -->

Samaṇaṁ mātā pitā bhātā,

<!-- @segment sn1.81:4.2 -->

vandanti naṁ patiṭṭhitaṁ;

<!-- @segment sn1.81:4.3 -->

Samaṇīdha jātihīnaṁ,

<!-- @segment sn1.81:4.4 -->

abhivādenti khattiyā”ti.

<!-- @segment sn1.81:4.5 -->

Chetvāvaggo aṭṭhamo.

<!-- @segment sn1.81:5.0 -->

Tassuddānaṁ

<!-- @segment sn1.81:5.1 -->

Chetvā rathañca cittañca,

<!-- @segment sn1.81:5.2 -->

vuṭṭhi bhītā najīrati;

<!-- @segment sn1.81:5.3 -->

Issaraṁ kāmaṁ pātheyyaṁ,

<!-- @segment sn1.81:5.4 -->

pajjoto araṇena cāti.

<!-- @segment sn1.81:5.5 -->

Devatāsaṁyuttaṁ samattaṁ.
