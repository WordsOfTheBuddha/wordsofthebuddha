<!-- @segment sn1.51:1.1 -->

“Kiṁsu yāva jarā sādhu,

<!-- @segment sn1.51:1.2 -->

kiṁsu sādhu patiṭṭhitaṁ;

<!-- @segment sn1.51:1.3 -->

Kiṁsu narānaṁ ratanaṁ,

<!-- @segment sn1.51:1.4 -->

kiṁsu corehi dūharan”ti.

<!-- @segment sn1.51:2.1 -->

“Sīlaṁ yāva jarā sādhu,

<!-- @segment sn1.51:2.2 -->

saddhā sādhu patiṭṭhitā;

<!-- @segment sn1.51:2.3 -->

Paññā narānaṁ ratanaṁ,

<!-- @segment sn1.51:2.4 -->

puññaṁ corehi dūharan”ti.
