<!-- @segment sn1.50:1.1 -->

“Avihaṁ upapannāse,

<!-- @segment sn1.50:1.2 -->

vimuttā satta bhikkhavo;

<!-- @segment sn1.50:1.3 -->

Rāgadosaparikkhīṇā,

<!-- @segment sn1.50:1.4 -->

tiṇṇā loke visattikan”ti.

<!-- @segment sn1.50:2.1 -->

“Ke ca te ataruṁ paṅkaṁ,

<!-- @segment sn1.50:2.2 -->

maccudheyyaṁ suduttaraṁ;

<!-- @segment sn1.50:2.3 -->

Ke hitvā mānusaṁ dehaṁ,

<!-- @segment sn1.50:2.4 -->

dibbayogaṁ upaccagun”ti.

<!-- @segment sn1.50:3.1 -->

“Upako palagaṇḍo ca,

<!-- @segment sn1.50:3.2 -->

pukkusāti ca te tayo;

<!-- @segment sn1.50:3.3 -->

Bhaddiyo khaṇḍadevo ca,

<!-- @segment sn1.50:3.4 -->

bāhuraggi ca siṅgiyo;

<!-- @segment sn1.50:3.5 -->

Te hitvā mānusaṁ dehaṁ,

<!-- @segment sn1.50:3.6 -->

dibbayogaṁ upaccagun”ti.

<!-- @segment sn1.50:4.1 -->

“Kusalī bhāsasī tesaṁ,

<!-- @segment sn1.50:4.2 -->

mārapāsappahāyinaṁ;

<!-- @segment sn1.50:4.3 -->

Kassa te dhammamaññāya,

<!-- @segment sn1.50:4.4 -->

acchiduṁ bhavabandhanan”ti.

<!-- @segment sn1.50:5.1 -->

“Na aññatra bhagavatā,

<!-- @segment sn1.50:5.2 -->

nāññatra tava sāsanā;

<!-- @segment sn1.50:5.3 -->

Yassa te dhammamaññāya,

<!-- @segment sn1.50:5.4 -->

acchiduṁ bhavabandhanaṁ.

<!-- @segment sn1.50:6.1 -->

Yattha nāmañca rūpañca,

<!-- @segment sn1.50:6.2 -->

asesaṁ uparujjhati;

<!-- @segment sn1.50:6.3 -->

Taṁ te dhammaṁ idhaññāya,

<!-- @segment sn1.50:6.4 -->

acchiduṁ bhavabandhanan”ti.

<!-- @segment sn1.50:7.1 -->

“Gambhīraṁ bhāsasī vācaṁ,

<!-- @segment sn1.50:7.2 -->

dubbijānaṁ sudubbudhaṁ;

<!-- @segment sn1.50:7.3 -->

Kassa tvaṁ dhammamaññāya,

<!-- @segment sn1.50:7.4 -->

vācaṁ bhāsasi īdisan”ti.

<!-- @segment sn1.50:8.1 -->

“Kumbhakāro pure āsiṁ,

<!-- @segment sn1.50:8.2 -->

vekaḷiṅge ghaṭīkaro;

<!-- @segment sn1.50:8.3 -->

Mātāpettibharo āsiṁ,

<!-- @segment sn1.50:8.4 -->

kassapassa upāsako.

<!-- @segment sn1.50:9.1 -->

Virato methunā dhammā,

<!-- @segment sn1.50:9.2 -->

brahmacārī nirāmiso;

<!-- @segment sn1.50:9.3 -->

Ahuvā te sagāmeyyo,

<!-- @segment sn1.50:9.4 -->

ahuvā te pure sakhā.

<!-- @segment sn1.50:10.1 -->

Sohamete pajānāmi,

<!-- @segment sn1.50:10.2 -->

vimutte satta bhikkhavo;

<!-- @segment sn1.50:10.3 -->

Rāgadosaparikkhīṇe,

<!-- @segment sn1.50:10.4 -->

tiṇṇe loke visattikan”ti.

<!-- @segment sn1.50:11.1 -->

“Evametaṁ tadā āsi,

<!-- @segment sn1.50:11.2 -->

yathā bhāsasi bhaggava;

<!-- @segment sn1.50:11.3 -->

Kumbhakāro pure āsi,

<!-- @segment sn1.50:11.4 -->

vekaḷiṅge ghaṭīkaro;

<!-- @segment sn1.50:11.5 -->

Mātāpettibharo āsi,

<!-- @segment sn1.50:11.6 -->

kassapassa upāsako.

<!-- @segment sn1.50:12.1 -->

Virato methunā dhammā,

<!-- @segment sn1.50:12.2 -->

brahmacārī nirāmiso;

<!-- @segment sn1.50:12.3 -->

Ahuvā me sagāmeyyo,

<!-- @segment sn1.50:12.4 -->

ahuvā me pure sakhā”ti.

<!-- @segment sn1.50:13.1 -->

“Evametaṁ purāṇānaṁ,

<!-- @segment sn1.50:13.2 -->

sahāyānaṁ ahu saṅgamo;

<!-- @segment sn1.50:13.3 -->

Ubhinnaṁ bhāvitattānaṁ,

<!-- @segment sn1.50:13.4 -->

sarīrantimadhārinan”ti.

<!-- @segment sn1.50:13.5 -->

Ādittavaggo pañcamo.

<!-- @segment sn1.50:14.0 -->

Tassuddānaṁ

<!-- @segment sn1.50:14.1 -->

Ādittaṁ kiṁdadaṁ annaṁ,

<!-- @segment sn1.50:14.2 -->

ekamūlaanomiyaṁ;

<!-- @segment sn1.50:14.3 -->

Accharāvanaropajetaṁ,

<!-- @segment sn1.50:14.4 -->

maccharena ghaṭīkaroti.
