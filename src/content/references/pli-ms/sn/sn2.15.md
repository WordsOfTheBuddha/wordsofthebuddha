<!-- @segment sn2.15:1.1 -->

Ekamantaṁ ṭhito kho candano devaputto bhagavantaṁ gāthāya ajjhabhāsi:

<!-- @segment sn2.15:2.1 -->

“Kathaṁsu tarati oghaṁ,

<!-- @segment sn2.15:2.2 -->

rattindivamatandito;

<!-- @segment sn2.15:2.3 -->

Appatiṭṭhe anālambe,

<!-- @segment sn2.15:2.4 -->

ko gambhīre na sīdatī”ti.

<!-- @segment sn2.15:3.1 -->

“Sabbadā sīlasampanno,

<!-- @segment sn2.15:3.2 -->

Paññavā susamāhito;

<!-- @segment sn2.15:3.3 -->

Āraddhavīriyo pahitatto,

<!-- @segment sn2.15:3.4 -->

Oghaṁ tarati duttaraṁ.

<!-- @segment sn2.15:4.1 -->

Virato kāmasaññāya,

<!-- @segment sn2.15:4.2 -->

rūpasaṁyojanātigo;

<!-- @segment sn2.15:4.3 -->

Nandīrāgaparikkhīṇo,

<!-- @segment sn2.15:4.4 -->

so gambhīre na sīdatī”ti.
