<!-- @segment sn51.67-76:1.1 -->

(Esanāvaggo vitthāretabbo.)

<!-- @segment sn51.67-76:2.0 -->

Tassuddānaṁ

<!-- @segment sn51.67-76:2.1 -->

Esanā vidhā āsavo,

<!-- @segment sn51.67-76:2.2 -->

Bhavo ca dukkhatā tisso;

<!-- @segment sn51.67-76:2.3 -->

Khilaṁ malañca nīgho ca,

<!-- @segment sn51.67-76:2.4 -->

Vedanā taṇhā tasinā cāti.
