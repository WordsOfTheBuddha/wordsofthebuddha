<!-- @segment sn1.60:1.1 -->

“Kiṁsu nidānaṁ gāthānaṁ,

<!-- @segment sn1.60:1.2 -->

kiṁsu tāsaṁ viyañjanaṁ;

<!-- @segment sn1.60:1.3 -->

Kiṁsu sannissitā gāthā,

<!-- @segment sn1.60:1.4 -->

kiṁsu gāthānamāsayo”ti.

<!-- @segment sn1.60:2.1 -->

“Chando nidānaṁ gāthānaṁ,

<!-- @segment sn1.60:2.2 -->

akkharā tāsaṁ viyañjanaṁ;

<!-- @segment sn1.60:2.3 -->

Nāmasannissitā gāthā,

<!-- @segment sn1.60:2.4 -->

kavi gāthānamāsayo”ti.

<!-- @segment sn1.60:2.5 -->

Jarāvaggo chaṭṭho.

<!-- @segment sn1.60:3.0 -->

Tassuddānaṁ

<!-- @segment sn1.60:3.1 -->

Jarā ajarasā mittaṁ,

<!-- @segment sn1.60:3.2 -->

vatthu tīṇi janāni ca;

<!-- @segment sn1.60:3.3 -->

Uppatho ca dutiyo ca,

<!-- @segment sn1.60:3.4 -->

kavinā pūrito vaggoti.
