<!-- @segment sn1.63:1.1 -->

“Kenassu nīyati loko,

<!-- @segment sn1.63:1.2 -->

kenassu parikassati;

<!-- @segment sn1.63:1.3 -->

Kissassu ekadhammassa,

<!-- @segment sn1.63:1.4 -->

sabbeva vasamanvagū”ti.

<!-- @segment sn1.63:2.1 -->

“Taṇhāya nīyati loko,

<!-- @segment sn1.63:2.2 -->

taṇhāya parikassati;

<!-- @segment sn1.63:2.3 -->

Taṇhāya ekadhammassa,

<!-- @segment sn1.63:2.4 -->

sabbeva vasamanvagū”ti.
