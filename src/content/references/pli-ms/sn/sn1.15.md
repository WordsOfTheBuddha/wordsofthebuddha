<!-- @segment sn1.15:1.1 -->

“Ṭhite majjhanhike kāle,

<!-- @segment sn1.15:1.2 -->

sannisīvesu pakkhisu;

<!-- @segment sn1.15:1.3 -->

Saṇateva brahāraññaṁ,

<!-- @segment sn1.15:1.4 -->

taṁ bhayaṁ paṭibhāti man”ti.

<!-- @segment sn1.15:2.1 -->

“Ṭhite majjhanhike kāle,

<!-- @segment sn1.15:2.2 -->

sannisīvesu pakkhisu;

<!-- @segment sn1.15:2.3 -->

Saṇateva brahāraññaṁ,

<!-- @segment sn1.15:2.4 -->

sā rati paṭibhāti man”ti.
