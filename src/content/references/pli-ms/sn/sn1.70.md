<!-- @segment sn1.70:1.1 -->

“Kismiṁ loko samuppanno,

<!-- @segment sn1.70:1.2 -->

kismiṁ kubbati santhavaṁ;

<!-- @segment sn1.70:1.3 -->

Kissa loko upādāya,

<!-- @segment sn1.70:1.4 -->

kismiṁ loko vihaññatī”ti.

<!-- @segment sn1.70:2.1 -->

“Chasu loko samuppanno,

<!-- @segment sn1.70:2.2 -->

chasu kubbati santhavaṁ;

<!-- @segment sn1.70:2.3 -->

Channameva upādāya,

<!-- @segment sn1.70:2.4 -->

chasu loko vihaññatī”ti.

<!-- @segment sn1.70:2.5 -->

Addhavaggo sattamo.

<!-- @segment sn1.70:3.0 -->

Tassuddānaṁ

<!-- @segment sn1.70:3.1 -->

Nāmaṁ cittañca taṇhā ca,

<!-- @segment sn1.70:3.2 -->

saṁyojanañca bandhanā;

<!-- @segment sn1.70:3.3 -->

Abbhāhatuḍḍito pihito,

<!-- @segment sn1.70:3.4 -->

icchā lokena te dasāti.
