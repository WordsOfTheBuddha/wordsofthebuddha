<!-- @segment sn1.43:1.1 -->

“Annamevābhinandanti,

<!-- @segment sn1.43:1.2 -->

ubhaye devamānusā;

<!-- @segment sn1.43:1.3 -->

Atha ko nāma so yakkho,

<!-- @segment sn1.43:1.4 -->

yaṁ annaṁ nābhinandatī”ti.

<!-- @segment sn1.43:2.1 -->

“Ye naṁ dadanti saddhāya,

<!-- @segment sn1.43:2.2 -->

vippasannena cetasā;

<!-- @segment sn1.43:2.3 -->

Tameva annaṁ bhajati,

<!-- @segment sn1.43:2.4 -->

asmiṁ loke paramhi ca.

<!-- @segment sn1.43:3.1 -->

Tasmā vineyya maccheraṁ,

<!-- @segment sn1.43:3.2 -->

dajjā dānaṁ malābhibhū;

<!-- @segment sn1.43:3.3 -->

Puññāni paralokasmiṁ,

<!-- @segment sn1.43:3.4 -->

patiṭṭhā honti pāṇinan”ti.
