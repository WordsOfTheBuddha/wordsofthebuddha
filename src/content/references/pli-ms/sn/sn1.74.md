<!-- @segment sn1.74:1.1 -->

“Kiṁsu uppatataṁ seṭṭhaṁ,

<!-- @segment sn1.74:1.2 -->

kiṁsu nipatataṁ varaṁ;

<!-- @segment sn1.74:1.3 -->

Kiṁsu pavajamānānaṁ,

<!-- @segment sn1.74:1.4 -->

kiṁsu pavadataṁ varan”ti.

<!-- @segment sn1.74:2.1 -->

“Bījaṁ uppatataṁ seṭṭhaṁ,

<!-- @segment sn1.74:2.2 -->

vuṭṭhi nipatataṁ varā;

<!-- @segment sn1.74:2.3 -->

Gāvo pavajamānānaṁ,

<!-- @segment sn1.74:2.4 -->

putto pavadataṁ varo”ti.

<!-- @segment sn1.74:3.1 -->

“Vijjā uppatataṁ seṭṭhā,

<!-- @segment sn1.74:3.2 -->

avijjā nipatataṁ varā;

<!-- @segment sn1.74:3.3 -->

Saṅgho pavajamānānaṁ,

<!-- @segment sn1.74:3.4 -->

buddho pavadataṁ varo”ti.
