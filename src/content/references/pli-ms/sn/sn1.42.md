<!-- @segment sn1.42:1.1 -->

“Kiṁdado balado hoti,

<!-- @segment sn1.42:1.2 -->

kiṁdado hoti vaṇṇado;

<!-- @segment sn1.42:1.3 -->

Kiṁdado sukhado hoti,

<!-- @segment sn1.42:1.4 -->

kiṁdado hoti cakkhudo;

<!-- @segment sn1.42:1.5 -->

Ko ca sabbadado hoti,

<!-- @segment sn1.42:1.6 -->

taṁ me akkhāhi pucchito”ti.

<!-- @segment sn1.42:2.1 -->

“Annado balado hoti,

<!-- @segment sn1.42:2.2 -->

vatthado hoti vaṇṇado;

<!-- @segment sn1.42:2.3 -->

Yānado sukhado hoti,

<!-- @segment sn1.42:2.4 -->

dīpado hoti cakkhudo.

<!-- @segment sn1.42:3.1 -->

So ca sabbadado hoti,

<!-- @segment sn1.42:3.2 -->

yo dadāti upassayaṁ;

<!-- @segment sn1.42:3.3 -->

Amataṁ dado ca so hoti,

<!-- @segment sn1.42:3.4 -->

yo dhammamanusāsatī”ti.
