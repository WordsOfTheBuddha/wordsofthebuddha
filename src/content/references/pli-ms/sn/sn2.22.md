<!-- @segment sn2.22:1.1 -->

Ekamantaṁ ṭhito kho khemo devaputto bhagavato santike imā gāthāyo abhāsi:

<!-- @segment sn2.22:2.1 -->

“Caranti bālā dummedhā,

<!-- @segment sn2.22:2.2 -->

amitteneva attanā;

<!-- @segment sn2.22:2.3 -->

Karontā pāpakaṁ kammaṁ,

<!-- @segment sn2.22:2.4 -->

yaṁ hoti kaṭukapphalaṁ.

<!-- @segment sn2.22:3.1 -->

Na taṁ kammaṁ kataṁ sādhu,

<!-- @segment sn2.22:3.2 -->

yaṁ katvā anutappati;

<!-- @segment sn2.22:3.3 -->

Yassa assumukho rodaṁ,

<!-- @segment sn2.22:3.4 -->

vipākaṁ paṭisevati.

<!-- @segment sn2.22:4.1 -->

Tañca kammaṁ kataṁ sādhu,

<!-- @segment sn2.22:4.2 -->

yaṁ katvā nānutappati;

<!-- @segment sn2.22:4.3 -->

Yassa patīto sumano,

<!-- @segment sn2.22:4.4 -->

vipākaṁ paṭisevati.

<!-- @segment sn2.22:5.1 -->

Paṭikacceva taṁ kayirā,

<!-- @segment sn2.22:5.2 -->

yaṁ jaññā hitamattano;

<!-- @segment sn2.22:5.3 -->

Na sākaṭikacintāya,

<!-- @segment sn2.22:5.4 -->

mantā dhīro parakkame.

<!-- @segment sn2.22:6.1 -->

Yathā sākaṭiko maṭṭhaṁ,

<!-- @segment sn2.22:6.2 -->

samaṁ hitvā mahāpathaṁ;

<!-- @segment sn2.22:6.3 -->

Visamaṁ maggamāruyha,

<!-- @segment sn2.22:6.4 -->

akkhacchinnova jhāyati.

<!-- @segment sn2.22:7.1 -->

Evaṁ dhammā apakkamma,

<!-- @segment sn2.22:7.2 -->

adhammamanuvattiya;

<!-- @segment sn2.22:7.3 -->

Mando maccumukhaṁ patto,

<!-- @segment sn2.22:7.4 -->

akkhacchinnova jhāyatī”ti.
