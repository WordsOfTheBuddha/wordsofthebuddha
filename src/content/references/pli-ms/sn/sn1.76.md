<!-- @segment sn1.76:1.1 -->

“Kiṁ jīrati kiṁ na jīrati,

<!-- @segment sn1.76:1.2 -->

kiṁsu uppathoti vuccati;

<!-- @segment sn1.76:1.3 -->

Kiṁsu dhammānaṁ paripantho,

<!-- @segment sn1.76:1.4 -->

kiṁsu rattindivakkhayo;

<!-- @segment sn1.76:1.5 -->

Kiṁ malaṁ brahmacariyassa,

<!-- @segment sn1.76:1.6 -->

kiṁ sinānamanodakaṁ.

<!-- @segment sn1.76:2.1 -->

Kati lokasmiṁ chiddāni,

<!-- @segment sn1.76:2.2 -->

yattha vittaṁ na tiṭṭhati;

<!-- @segment sn1.76:2.3 -->

Bhagavantaṁ puṭṭhumāgamma,

<!-- @segment sn1.76:2.4 -->

kathaṁ jānemu taṁ mayan”ti.

<!-- @segment sn1.76:3.1 -->

“Rūpaṁ jīrati maccānaṁ,

<!-- @segment sn1.76:3.2 -->

nāmagottaṁ na jīrati;

<!-- @segment sn1.76:3.3 -->

Rāgo uppathoti vuccati.

<!-- @segment sn1.76:4.1 -->

Lobho dhammānaṁ paripantho,

<!-- @segment sn1.76:4.2 -->

Vayo rattindivakkhayo;

<!-- @segment sn1.76:4.3 -->

Itthī malaṁ brahmacariyassa,

<!-- @segment sn1.76:4.4 -->

Etthāyaṁ sajjate pajā;

<!-- @segment sn1.76:4.5 -->

Tapo ca brahmacariyañca,

<!-- @segment sn1.76:4.6 -->

Taṁ sinānamanodakaṁ.

<!-- @segment sn1.76:5.1 -->

Cha lokasmiṁ chiddāni,

<!-- @segment sn1.76:5.2 -->

yattha vittaṁ na tiṭṭhati;

<!-- @segment sn1.76:5.3 -->

Ālasyañca pamādo ca,

<!-- @segment sn1.76:5.4 -->

anuṭṭhānaṁ asaṁyamo;

<!-- @segment sn1.76:5.5 -->

Niddā tandī ca te chidde,

<!-- @segment sn1.76:5.6 -->

sabbaso taṁ vivajjaye”ti.
