<!-- @segment sn1.18:1.1 -->

“Hirīnisedho puriso,

<!-- @segment sn1.18:1.2 -->

koci lokasmiṁ vijjati;

<!-- @segment sn1.18:1.3 -->

Yo nindaṁ apabodhati,

<!-- @segment sn1.18:1.4 -->

asso bhadro kasāmivā”ti.

<!-- @segment sn1.18:2.1 -->

“Hirīnisedhā tanuyā,

<!-- @segment sn1.18:2.2 -->

ye caranti sadā satā;

<!-- @segment sn1.18:2.3 -->

Antaṁ dukkhassa pappuyya,

<!-- @segment sn1.18:2.4 -->

caranti visame saman”ti.
