<!-- @segment sn1.57:1.1 -->

“Kiṁsu janeti purisaṁ,

<!-- @segment sn1.57:1.2 -->

kiṁsu tassa vidhāvati;

<!-- @segment sn1.57:1.3 -->

Kiṁsu saṁsāramāpādi,

<!-- @segment sn1.57:1.4 -->

kiṁsu tassa parāyanan”ti.

<!-- @segment sn1.57:2.1 -->

“Taṇhā janeti purisaṁ,

<!-- @segment sn1.57:2.2 -->

cittamassa vidhāvati;

<!-- @segment sn1.57:2.3 -->

Satto saṁsāramāpādi,

<!-- @segment sn1.57:2.4 -->

kammaṁ tassa parāyanan”ti.
