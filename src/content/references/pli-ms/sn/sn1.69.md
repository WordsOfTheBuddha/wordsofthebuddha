<!-- @segment sn1.69:1.1 -->

“Kenassu bajjhatī loko,

<!-- @segment sn1.69:1.2 -->

kissa vinayāya muccati;

<!-- @segment sn1.69:1.3 -->

Kissassu vippahānena,

<!-- @segment sn1.69:1.4 -->

sabbaṁ chindati bandhanan”ti.

<!-- @segment sn1.69:2.1 -->

“Icchāya bajjhatī loko,

<!-- @segment sn1.69:2.2 -->

icchāvinayāya muccati;

<!-- @segment sn1.69:2.3 -->

Icchāya vippahānena,

<!-- @segment sn1.69:2.4 -->

sabbaṁ chindati bandhanan”ti.
