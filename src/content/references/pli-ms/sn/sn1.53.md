<!-- @segment sn1.53:1.1 -->

“Kiṁsu pavasato mittaṁ,

<!-- @segment sn1.53:1.2 -->

kiṁsu mittaṁ sake ghare;

<!-- @segment sn1.53:1.3 -->

Kiṁ mittaṁ atthajātassa,

<!-- @segment sn1.53:1.4 -->

kiṁ mittaṁ samparāyikan”ti.

<!-- @segment sn1.53:2.1 -->

“Sattho pavasato mittaṁ,

<!-- @segment sn1.53:2.2 -->

mātā mittaṁ sake ghare;

<!-- @segment sn1.53:2.3 -->

Sahāyo atthajātassa,

<!-- @segment sn1.53:2.4 -->

hoti mittaṁ punappunaṁ;

<!-- @segment sn1.53:2.5 -->

Sayaṅkatāni puññāni,

<!-- @segment sn1.53:2.6 -->

taṁ mittaṁ samparāyikan”ti.
