<!-- @segment sn1.56:1.1 -->

“Kiṁsu janeti purisaṁ,

<!-- @segment sn1.56:1.2 -->

kiṁsu tassa vidhāvati;

<!-- @segment sn1.56:1.3 -->

Kiṁsu saṁsāramāpādi,

<!-- @segment sn1.56:1.4 -->

kismā na parimuccatī”ti.

<!-- @segment sn1.56:2.1 -->

“Taṇhā janeti purisaṁ,

<!-- @segment sn1.56:2.2 -->

cittamassa vidhāvati;

<!-- @segment sn1.56:2.3 -->

Satto saṁsāramāpādi,

<!-- @segment sn1.56:2.4 -->

dukkhā na parimuccatī”ti.
