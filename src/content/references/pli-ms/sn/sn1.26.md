<!-- @segment sn1.26:1.1 -->

“Kati lokasmiṁ pajjotā,

<!-- @segment sn1.26:1.2 -->

Tehi loko pakāsati;

<!-- @segment sn1.26:1.3 -->

Bhagavantaṁ puṭṭhumāgamma,

<!-- @segment sn1.26:1.4 -->

Kathaṁ jānemu taṁ mayan”ti.

<!-- @segment sn1.26:2.1 -->

“Cattāro loke pajjotā,

<!-- @segment sn1.26:2.2 -->

pañcamettha na vijjati;

<!-- @segment sn1.26:2.3 -->

Divā tapati ādicco,

<!-- @segment sn1.26:2.4 -->

rattimābhāti candimā.

<!-- @segment sn1.26:3.1 -->

Atha aggi divārattiṁ,

<!-- @segment sn1.26:3.2 -->

tattha tattha pakāsati;

<!-- @segment sn1.26:3.3 -->

Sambuddho tapataṁ seṭṭho,

<!-- @segment sn1.26:3.4 -->

esā ābhā anuttarā”ti.
