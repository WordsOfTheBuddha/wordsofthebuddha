<!-- @segment sn1.80:1.1 -->

“Kiṁsu lokasmi pajjoto,

<!-- @segment sn1.80:1.2 -->

kiṁsu lokasmi jāgaro;

<!-- @segment sn1.80:1.3 -->

Kiṁsu kamme sajīvānaṁ,

<!-- @segment sn1.80:1.4 -->

kimassa iriyāpatho.

<!-- @segment sn1.80:2.1 -->

Kiṁsu alasaṁ analasañca,

<!-- @segment sn1.80:2.2 -->

Mātā puttaṁva posati;

<!-- @segment sn1.80:2.3 -->

Kiṁ bhūtā upajīvanti,

<!-- @segment sn1.80:2.4 -->

Ye pāṇā pathavissitā”ti.

<!-- @segment sn1.80:3.1 -->

“Paññā lokasmi pajjoto,

<!-- @segment sn1.80:3.2 -->

sati lokasmi jāgaro;

<!-- @segment sn1.80:3.3 -->

Gāvo kamme sajīvānaṁ,

<!-- @segment sn1.80:3.4 -->

sītassa iriyāpatho.

<!-- @segment sn1.80:4.1 -->

Vuṭṭhi alasaṁ analasañca,

<!-- @segment sn1.80:4.2 -->

Mātā puttaṁva posati;

<!-- @segment sn1.80:4.3 -->

Vuṭṭhiṁ bhūtā upajīvanti,

<!-- @segment sn1.80:4.4 -->

Ye pāṇā pathavissitā”ti.
