<!-- @segment sn1.68:1.1 -->

“Kenassu pihito loko,

<!-- @segment sn1.68:1.2 -->

kismiṁ loko patiṭṭhito;

<!-- @segment sn1.68:1.3 -->

Kenassu uḍḍito loko,

<!-- @segment sn1.68:1.4 -->

kenassu parivārito”ti.

<!-- @segment sn1.68:2.1 -->

“Maccunā pihito loko,

<!-- @segment sn1.68:2.2 -->

dukkhe loko patiṭṭhito;

<!-- @segment sn1.68:2.3 -->

Taṇhāya uḍḍito loko,

<!-- @segment sn1.68:2.4 -->

jarāya parivārito”ti.
