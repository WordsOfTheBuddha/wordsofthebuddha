<!-- @segment sn2.24:1.1 -->

Ekamantaṁ ṭhito kho ghaṭīkāro devaputto bhagavato santike imaṁ gāthaṁ abhāsi:

<!-- @segment sn2.24:2.1 -->

“Avihaṁ upapannāse,

<!-- @segment sn2.24:2.2 -->

vimuttā satta bhikkhavo;

<!-- @segment sn2.24:2.3 -->

Rāgadosaparikkhīṇā,

<!-- @segment sn2.24:2.4 -->

tiṇṇā loke visattikan”ti.

<!-- @segment sn2.24:3.1 -->

“Ke ca te ataruṁ paṅkaṁ,

<!-- @segment sn2.24:3.2 -->

maccudheyyaṁ suduttaraṁ;

<!-- @segment sn2.24:3.3 -->

Ke hitvā mānusaṁ dehaṁ,

<!-- @segment sn2.24:3.4 -->

dibbayogaṁ upaccagun”ti.

<!-- @segment sn2.24:4.1 -->

“Upako palagaṇḍo ca,

<!-- @segment sn2.24:4.2 -->

pukkusāti ca te tayo;

<!-- @segment sn2.24:4.3 -->

Bhaddiyo khaṇḍadevo ca,

<!-- @segment sn2.24:4.4 -->

bāhuraggi ca saṅgiyo;

<!-- @segment sn2.24:4.5 -->

Te hitvā mānusaṁ dehaṁ,

<!-- @segment sn2.24:4.6 -->

dibbayogaṁ upaccagun”ti.

<!-- @segment sn2.24:5.1 -->

“Kusalī bhāsasī tesaṁ,

<!-- @segment sn2.24:5.2 -->

mārapāsappahāyinaṁ;

<!-- @segment sn2.24:5.3 -->

Kassa te dhammamaññāya,

<!-- @segment sn2.24:5.4 -->

acchiduṁ bhavabandhanan”ti.

<!-- @segment sn2.24:6.1 -->

“Na aññatra bhagavatā,

<!-- @segment sn2.24:6.2 -->

nāññatra tava sāsanā;

<!-- @segment sn2.24:6.3 -->

Yassa te dhammamaññāya,

<!-- @segment sn2.24:6.4 -->

acchiduṁ bhavabandhanaṁ.

<!-- @segment sn2.24:7.1 -->

Yattha nāmañca rūpañca,

<!-- @segment sn2.24:7.2 -->

asesaṁ uparujjhati;

<!-- @segment sn2.24:7.3 -->

Taṁ te dhammaṁ idhaññāya,

<!-- @segment sn2.24:7.4 -->

acchiduṁ bhavabandhanan”ti.

<!-- @segment sn2.24:8.1 -->

“Gambhīraṁ bhāsasī vācaṁ,

<!-- @segment sn2.24:8.2 -->

dubbijānaṁ sudubbudhaṁ;

<!-- @segment sn2.24:8.3 -->

Kassa tvaṁ dhammamaññāya,

<!-- @segment sn2.24:8.4 -->

vācaṁ bhāsasi īdisan”ti.

<!-- @segment sn2.24:9.1 -->

“Kumbhakāro pure āsiṁ,

<!-- @segment sn2.24:9.2 -->

vekaḷiṅge ghaṭīkaro;

<!-- @segment sn2.24:9.3 -->

Mātāpettibharo āsiṁ,

<!-- @segment sn2.24:9.4 -->

kassapassa upāsako.

<!-- @segment sn2.24:10.1 -->

Virato methunā dhammā,

<!-- @segment sn2.24:10.2 -->

brahmacārī nirāmiso;

<!-- @segment sn2.24:10.3 -->

Ahuvā te sagāmeyyo,

<!-- @segment sn2.24:10.4 -->

ahuvā te pure sakhā.

<!-- @segment sn2.24:11.1 -->

Sohamete pajānāmi,

<!-- @segment sn2.24:11.2 -->

vimutte satta bhikkhavo;

<!-- @segment sn2.24:11.3 -->

Rāgadosaparikkhīṇe,

<!-- @segment sn2.24:11.4 -->

tiṇṇe loke visattikan”ti.

<!-- @segment sn2.24:12.1 -->

“Evametaṁ tadā āsi,

<!-- @segment sn2.24:12.2 -->

yathā bhāsasi bhaggava;

<!-- @segment sn2.24:12.3 -->

Kumbhakāro pure āsi,

<!-- @segment sn2.24:12.4 -->

vekaḷiṅge ghaṭīkaro.

<!-- @segment sn2.24:13.1 -->

Mātāpettibharo āsi,

<!-- @segment sn2.24:13.2 -->

kassapassa upāsako;

<!-- @segment sn2.24:13.3 -->

Virato methunā dhammā,

<!-- @segment sn2.24:13.4 -->

brahmacārī nirāmiso;

<!-- @segment sn2.24:13.5 -->

Ahuvā me sagāmeyyo,

<!-- @segment sn2.24:13.6 -->

ahuvā me pure sakhā”ti.

<!-- @segment sn2.24:14.1 -->

“Evametaṁ purāṇānaṁ,

<!-- @segment sn2.24:14.2 -->

sahāyānaṁ ahu saṅgamo;

<!-- @segment sn2.24:14.3 -->

Ubhinnaṁ bhāvitattānaṁ,

<!-- @segment sn2.24:14.4 -->

sarīrantimadhārinan”ti.
