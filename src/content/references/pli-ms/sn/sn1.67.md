<!-- @segment sn1.67:1.1 -->

“Kenassu uḍḍito loko,

<!-- @segment sn1.67:1.2 -->

kenassu parivārito;

<!-- @segment sn1.67:1.3 -->

Kenassu pihito loko,

<!-- @segment sn1.67:1.4 -->

kismiṁ loko patiṭṭhito”ti.

<!-- @segment sn1.67:2.1 -->

“Taṇhāya uḍḍito loko,

<!-- @segment sn1.67:2.2 -->

jarāya parivārito;

<!-- @segment sn1.67:2.3 -->

Maccunā pihito loko,

<!-- @segment sn1.67:2.4 -->

dukkhe loko patiṭṭhito”ti.
